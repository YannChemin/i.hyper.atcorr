8/* Compute Rayleigh optical depth at wavelength wl (µm) using Edlen 1966 formula.
9 * Requires ctx->atm to be initialized. */
10void sixs_odrayl(const SixsCtx *ctx, float wl, float *tray) {
11 double awl = wl;
12 double ak = 1.0 / awl;
13 double a1 = 130.0 - ak * ak;
14 double a2 = 38.9 - ak * ak;
15 double a3 = 2406030.0 / a1;
16 double a4 = 15997.0 / a2;
17 double an = (8342.13 + a3 + a4) * 1.0e-8 + 1.0;
18
19 double delta = ctx->del.delta;
20 double a = (24.0 * M_PI * M_PI * M_PI) * ((an * an - 1.0) * (an * an - 1.0)) *
21 (6.0 + 3.0 * delta) / (6.0 - 7.0 * delta) /
22 ((an * an + 2.0) * (an * an + 2.0));
23
24 double tr = 0.0;
25 const float *z = ctx->atm.z;
26 const float *p = ctx->atm.p;
27 const float *t = ctx->atm.t;
28 float ns = 2.54743e19f;
29 for (int k = 0; k < NATM - 1; k++) {
30 double dppt = (288.15 / 1013.25) * ((double)(p[k] / t[k]) + (double)(p[k+1] / t[k+1])) / 2.0;
31 double sr = (a * dppt / (awl * awl * awl * awl) / ns * 1.0e16) * 1.0e5;
32 tr += (double)(z[k+1] - z[k]) * sr;
33 }
34 *tray = (float)tr;
