23class TestSolarSpectrum(TestCase):
24 """Tests for solar_E0(): Thuillier solar irradiance spectrum."""
25
26 def test_e0_visible_positive(self):
27 """E0 must be positive at every 50 nm step across the VIS/NIR range."""
28 wl = np.arange(0.40, 0.91, 0.05, dtype=np.float32)
29 E0 = ac.solar_E0(wl)
30 for w, e in zip(wl, E0):
31 self.assertGreater(float(e), 0.0, msg=f"E0 ≤ 0 at {w:.2f} µm")
32
33 def test_e0_swir_positive(self):
34 """E0 must be positive in the SWIR range (1.0–2.5 µm)."""
35 wl = np.arange(1.0, 2.51, 0.25, dtype=np.float32)
36 E0 = ac.solar_E0(wl)
37 self.assertTrue(np.all(E0 > 0), msg=f"Non-positive E0 in SWIR: {E0}")
38
39 def test_e0_reference_visible(self):
40 """E0 at key VIS wavelengths must be within 15% of Thuillier reference."""
41 # Approximate published values (W m⁻² µm⁻¹)
42 references = {0.45: 1980.0, 0.55: 1900.0, 0.65: 1630.0, 0.87: 950.0}
43 for wl_um, ref in references.items():
44 E0 = ac.solar_E0(float(wl_um))
45 self.assertAlmostEqual(
46 E0, ref, delta=ref * 0.15,
47 msg=f"E0({wl_um} µm) = {E0:.1f}, expected ~{ref:.0f}",
48 )
49
50 def test_e0_scalar_returns_float(self):
51 """Scalar wavelength input must return a Python float."""
52 E0 = ac.solar_E0(0.55)
53 self.assertIsInstance(E0, float)
54 self.assertGreater(E0, 0.0)
55
56 def test_e0_array_returns_array(self):
57 """Array input must return an array of matching length."""
58 wl = np.array([0.45, 0.55, 0.65, 0.87], dtype=np.float32)
59 E0 = ac.solar_E0(wl)
60 self.assertEqual(len(E0), len(wl))
61 self.assertTrue(np.all(np.isfinite(E0)))
62
63 def test_e0_blue_greater_than_swir(self):
64 """Broad spectral trend: E0 at 0.45 µm must exceed E0 at 2.0 µm."""
65 self.assertGreater(ac.solar_E0(0.45), ac.solar_E0(2.0))
66
67 def test_e0_all_finite(self):
68 """No NaN or Inf in E0 across 0.25–4.0 µm."""
69 wl = np.arange(0.25, 4.01, 0.05, dtype=np.float32)
70 E0 = ac.solar_E0(wl)
71 self.assertTrue(
72 np.all(np.isfinite(E0)),
73 msg=f"Non-finite E0 at: {wl[~np.isfinite(E0)]}",
74 )

77class TestEarthSunDist(TestCase):
78 """Tests for earth_sun_dist2(): seasonal Earth-Sun distance squared."""
79
80 def test_perihelion_less_than_one(self):
81 """Near perihelion (DOY 3) Earth is closest: d² must be < 1."""
82 d2 = ac.earth_sun_dist2(3)
83 self.assertLess(
84 d2, 1.0, msg=f"d²(DOY=3) = {d2:.6f}, expected < 1.0"
85 )
86
87 def test_aphelion_greater_than_one(self):
88 """Near aphelion (DOY 185) Earth is farthest: d² must be > 1."""
89 d2 = ac.earth_sun_dist2(185)
90 self.assertGreater(
91 d2, 1.0, msg=f"d²(DOY=185) = {d2:.6f}, expected > 1.0"
92 )
93
94 def test_all_days_positive_finite(self):
95 """d² must be positive and finite for every calendar day."""
96 for doy in range(1, 366):
97 d2 = ac.earth_sun_dist2(doy)
98 self.assertGreater(d2, 0.0, msg=f"d² ≤ 0 at DOY {doy}")
99 self.assertTrue(math.isfinite(d2), msg=f"d² non-finite at DOY {doy}")
100
101 def test_annual_variation_within_4pct(self):
102 """d² must stay within 4% of 1 AU² all year (Earth eccentricity ~0.0167)."""
103 for doy in range(1, 366):
104 d2 = ac.earth_sun_dist2(doy)
105 self.assertAlmostEqual(
106 d2, 1.0, delta=0.04,
107 msg=f"d²(DOY={doy}) = {d2:.4f} outside ±4% of 1 AU²",
108 )
109
110 def test_eccentricity_range(self):
111 """Peak-to-peak variation in d² must be ~3.3% (orbital eccentricity 0.0167)."""
112 d2_all = [ac.earth_sun_dist2(d) for d in range(1, 366)]
113 rng = max(d2_all) - min(d2_all)
114 self.assertGreater(rng, 0.030, msg=f"Annual d² range {rng:.4f} < 3.0%")
115 self.assertLess(rng, 0.040, msg=f"Annual d² range {rng:.4f} > 4.0%")
116
117 def test_symmetry_perihelion_aphelion(self):
118 """d²(perihelion) + d²(aphelion) ≈ 2 (symmetric about 1 AU²)."""
119 d2_min = min(ac.earth_sun_dist2(d) for d in range(1, 366))
120 d2_max = max(ac.earth_sun_dist2(d) for d in range(1, 366))
121 self.assertAlmostEqual(
122 d2_min + d2_max, 2.0, delta=0.01,
123 msg=f"d²_min + d²_max = {d2_min + d2_max:.4f}, expected ≈ 2.0",
124 )
