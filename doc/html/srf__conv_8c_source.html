1/* srf_conv.c — Spectral Response Function (SRF) convolution for gas transmittance.
2 *
3 * For hyperspectral sensors with FWHM < 5 nm, the 6SV Curtis-Godson
4 * parameterisation (10 cm⁻¹ = ~0.5–2 nm) introduces errors in gas absorption
5 * bands (H₂O at 720/820/940/1380 nm, O₂-A at 760 nm, CO₂ at 1600/2000 nm).
6 * Errors reach 20–40% transmittance at 1 nm FWHM, and up to 80% at the O₂-A
7 * band for sensors narrower than 0.5 nm.
8 *
9 * Correction method:
10 * 1. Call libRadtran uvspec with mol_abs_param reptran fine (~0.05 nm)
11 * for solar and viewing paths. T_eff(λ) = edir_surface / edir_toa.
12 * (Rayleigh scattering is present but cancels in the fine/coarse ratio
13 * because it is spectrally smooth over any sensor FWHM.)
14 * 2. Convolve T_gas_fine with the sensor Gaussian SRF per band.
15 * 3. Repeat with reptran coarse — this matches the resolution the LUT was
16 * built at (6SV Curtis-Godson at ~10 cm⁻¹).
17 * 4. Correction = T_gas_fine_SRF / T_gas_coarse_at_band_centre
18 * 5. Apply multiplicatively to all T_down and T_up in the LUT.
19 * T_scat is smooth spectrally; only T_gas needs SRF correction.
20 *
21 * OpenMP: 4 × n_h2o uvspec calls are run in parallel (one call per
22 * {resolution, direction, h2o} triplet). Each thread uses its own temp
23 * directory so subprocesses never collide.
24 *
25 * libRadtran dependency: runtime only (subprocess via popen). No link-time
26 * dependency. Set LIBRADTRAN_DATA to override default data path.
27 */

138/* ─── Gaussian SRF convolution ──────────────────────────────────────────── */
139
140/* Convolve spectrum (wl[], T[], n) with Gaussian SRF centred at centre_nm
141 * with given fwhm_nm. Weights: w = exp(-0.5*(Δλ/σ)²), σ = fwhm/(2√(2ln2)).
142 * Returns SRF-weighted mean transmittance; falls back to nearest if no points
143 * fall within the ±4σ window. */
144static float gaussian_srf_convolve(const float *wl, const float *T, int n,
145 float centre_nm, float fwhm_nm)
146{
147 float sigma = fwhm_nm / (2.0f * sqrtf(2.0f * logf(2.0f)));
148 float half_win = SRF_WINDOW_SIGMA * sigma;
149 float lo = centre_nm - half_win;
150 float hi = centre_nm + half_win;
151
152 /* Binary search for start of window */
153 int l = 0, r = n - 1;
154 while (l < r) {
155 int m = (l + r) / 2;
156 if (wl[m] < lo) l = m + 1; else r = m;
157 }
158
159 double sum_w = 0.0, sum_wT = 0.0;
160 for (int i = l; i < n && wl[i] <= hi; i++) {
161 double d = (double)(wl[i] - centre_nm) / (double)sigma;
162 double w = exp(-0.5 * d * d);
163 sum_w += w;
164 sum_wT += w * (double)T[i];
165 }
166
167 if (sum_w < 1e-30) {
168 /* No points in window — return nearest */
169 int nearest = 0;
170 float best = fabsf(wl[0] - centre_nm);
171 for (int i = 1; i < n; i++) {
172 float dist = fabsf(wl[i] - centre_nm);
173 if (dist < best) { best = dist; nearest = i; }
174 }
175 return T[nearest];
176 }
177 float result = (float)(sum_wT / sum_w);
178 if (result < 0.0f) result = 0.0f;
179 if (result > 1.0f) result = 1.0f;
180 return result;
181}

183/* ─── Linear interpolation ──────────────────────────────────────────────── */
184
185static float interp1d_nm(const float *x, const float *y, int n, float xi)
186{
187 if (n <= 0) return 1.0f;
188 if (xi <= x[0]) return y[0];
189 if (xi >= x[n - 1]) return y[n - 1];
190
191 int lo = 0, hi = n - 1;
192 while (hi - lo > 1) {
193 int mid = (lo + hi) / 2;
194 if (x[mid] <= xi) lo = mid; else hi = mid;
195 }
196 float t = (xi - x[lo]) / (x[hi] - x[lo]);
197 float v = y[lo] + t * (y[hi] - y[lo]);
198 if (v < 0.0f) v = 0.0f;
199 if (v > 1.0f) v = 1.0f;
200 return v;
201}

203/* ─── Run uvspec for gas-only transmittance ─────────────────────────────── */
204
205/* Runs uvspec with mol_abs_param reptran {resolution}, no_rayleigh, no
206 * aerosol. Returns T_gas = edir_surface / edir_toa.
207 *
208 * sza_deg: zenith angle of the beam path (solar for down, view for up).
209 * h2o_gcm2: water vapour column (g/cm²).
210 * o3_du: ozone column (Dobson units).
211 * wl_min_nm, wl_max_nm: spectral range requested.
212 * resolution: "fine" (~0.05 nm) or "coarse" (~0.16 nm).
213 * thread_id: used to make temp-dir name unique per OMP thread.
214 *
215 * Returns GasSpectrum* on success (caller frees), NULL on error. */
216static GasSpectrum *run_uvspec_gas(float sza_deg, float h2o_gcm2, float o3_du,
217 float wl_min_nm, float wl_max_nm,
218 const char *resolution,
219 const char *uvspec_path,
220 const char *data_path,
221 int thread_id)
222{
223 /* ── Create temp directory ── */
224 char tmpdir[PATH_MAX];
225 snprintf(tmpdir, sizeof(tmpdir), "/tmp/atcorr_srf_%d_XXXXXX", thread_id);
226 if (!mkdtemp(tmpdir)) return NULL;
227
228 char inp_path[PATH_MAX + 16]; /* +16 for "/uvspec.inp\0" */
229 snprintf(inp_path, sizeof(inp_path), "%s/uvspec.inp", tmpdir);
230
231 /* ── Write uvspec input ── */
232 FILE *f = fopen(inp_path, "w");
233 if (!f) { rmdir(tmpdir); return NULL; }
234
235 /* mol_modify H2O takes precipitable water in mm (1 g/cm² = 10 mm) */
236 float h2o_mm = h2o_gcm2 * 10.0f;
237 float wl_lo = wl_min_nm - (float)SRF_MARGIN_NM;
238 float wl_hi = wl_max_nm + (float)SRF_MARGIN_NM;
239 if (wl_lo < 200.0f) wl_lo = 200.0f;
240 if (wl_hi > 4900.0f) wl_hi = 4900.0f;
241
242 /* Clamp SZA: disort fails at 90° */
243 if (sza_deg > 89.0f) sza_deg = 89.0f;
244 if (sza_deg < 0.0f) sza_deg = 0.0f;
245
246 fprintf(f,
247 "data_files_path %s\n"
248 "atmosphere_file %s/atmmod/afglus.dat\n"
249 "source solar %s/solar_flux/kurudz_1.0nm.dat per_nm\n"
250 "wavelength %.2f %.2f\n"
251 "mol_abs_param reptran %s\n"
252 "mol_modify H2O %.4f MM\n"
253 "mol_modify O3 %.2f DU\n"
254 "sza %.4f\n"
255 "phi0 0.0\n"
256 "umu 1.0\n"
257 "albedo 0.0\n"
258 "pressure 1013.25\n"
259 "rte_solver disort\n"
260 "number_of_streams 2\n"
261 "zout sur toa\n"
262 "output_user lambda edir\n"
263 "quiet\n",
264 data_path, data_path, data_path,
265 (double)wl_lo, (double)wl_hi,
266 resolution,
267 (double)h2o_mm,
268 (double)(o3_du > 0.0f ? o3_du : 300.0f),
269 (double)sza_deg);
270 fclose(f);
271
272 /* ── Run uvspec ── */
273 char cmd[PATH_MAX * 2 + 16];
274 snprintf(cmd, sizeof(cmd), "%s < %s 2>/dev/null", uvspec_path, inp_path);
275
276 FILE *pipe = popen(cmd, "r");
277 if (!pipe) {
278 unlink(inp_path); rmdir(tmpdir);
279 return NULL;
280 }
281
282 /* ── Allocate output ── */
283 GasSpectrum *gs = malloc(sizeof(GasSpectrum));
284 if (!gs) { pclose(pipe); unlink(inp_path); rmdir(tmpdir); return NULL; }
285 gs->wl = malloc((size_t)SRF_MAX_PTS * sizeof(float));
286 gs->T_gas = malloc((size_t)SRF_MAX_PTS * sizeof(float));
287 gs->n_pts = 0;
288 if (!gs->wl || !gs->T_gas) {
289 gas_spectrum_free(gs); pclose(pipe);
290 unlink(inp_path); rmdir(tmpdir);
291 return NULL;
292 }
293
294 /* ── Parse 2-row-per-wavelength output ──
295 * zout sur toa → rows alternate: surface (even), TOA (odd). */
296 char line[128];
297 int is_sur = 1;
298 float edir_sur = 0.0f;
299 float last_wl = -1.0f;
300
301 while (fgets(line, sizeof(line), pipe)) {
302 if (line[0] == '#' || line[0] == '\n') continue;
303 float wl_v, edir_v;
304 if (sscanf(line, "%f %f", &wl_v, &edir_v) != 2) continue;
305
306 if (is_sur) {
307 edir_sur = edir_v;
308 last_wl = wl_v;
309 is_sur = 0;
310 } else {
311 /* TOA row: compute T_gas = edir_sur / edir_toa */
312 float edir_toa = (edir_v > 1e-30f) ? edir_v : 1e-30f;
313 float T = edir_sur / edir_toa;
314 if (T < 0.0f) T = 0.0f;
315 if (T > 1.0f) T = 1.0f;
316
317 /* Only store points within requested range */
318 if (last_wl >= wl_min_nm && last_wl <= wl_max_nm
319 && gs->n_pts < SRF_MAX_PTS) {
320 gs->wl[gs->n_pts] = last_wl;
321 gs->T_gas[gs->n_pts] = T;
322 gs->n_pts++;
323 }
324 is_sur = 1;
325 }
326 }
327
328 pclose(pipe);
329 unlink(inp_path);
330 rmdir(tmpdir);
331
332 if (gs->n_pts == 0) {
333 gas_spectrum_free(gs);
334 return NULL;
335 }
336 return gs;
337}

341/* Compute SRF correction table.
342 *
343 * For each h2o grid point, runs 4 uvspec calls:
344 * fine × solar zenith → T_gas_down_fine[wl]
345 * fine × view zenith → T_gas_up_fine[wl]
346 * coarse × solar zenith → T_gas_down_coarse[wl]
347 * coarse × view zenith → T_gas_up_coarse[wl]
348 *
349 * Correction[iw, ih] = gaussian_convolve(fine[wl]) / interp(coarse, wl[iw])
350 *
351 * All 4 × n_h2o calls are distributed across OpenMP threads.
352 *
353 * Returns NULL if uvspec/data not found or all runs failed. */
354SrfCorrection *atcorr_srf_compute(const SrfConfig *srf_cfg,
355 const LutConfig *lut_cfg)
356{
357 char uvspec_path[PATH_MAX], data_path[PATH_MAX];
358
359 if (find_uvspec(uvspec_path, sizeof(uvspec_path)) != 0) {
360 fprintf(stderr, "[srf_conv] uvspec not found — set LIBRADTRAN_DIR\n");
361 return NULL;
362 }
363 if (find_data_path(data_path, sizeof(data_path)) != 0) {
364 fprintf(stderr, "[srf_conv] libRadtran data not found — set LIBRADTRAN_DATA\n");
365 return NULL;
366 }
367
368 int n_wl = lut_cfg->n_wl;
369 int n_h2o = lut_cfg->n_h2o;
370
371 /* Sensor range in nm */
372 float wl_min_nm = lut_cfg->wl[0] * 1000.0f; /* µm → nm */
373 float wl_max_nm = lut_cfg->wl[n_wl - 1] * 1000.0f;
374
375 float o3_du = (lut_cfg->ozone_du > 0.0f)
376 ? lut_cfg->ozone_du : 300.0f;
377 float sza = lut_cfg->sza;
378 float vza = lut_cfg->vza;
379 float thresh = (srf_cfg && srf_cfg->threshold_um > 0.0f)
380 ? srf_cfg->threshold_um : 0.005f; /* 5 nm */
381
382 /* Count bands needing correction */
383 int n_to_correct = 0;
384 if (srf_cfg && srf_cfg->fwhm_um) {
385 for (int iw = 0; iw < n_wl; iw++)
386 if (srf_cfg->fwhm_um[iw] < thresh) n_to_correct++;
387 } else {
388 n_to_correct = n_wl; /* no FWHM provided: correct all */
389 }
390 if (n_to_correct == 0) {
391 fprintf(stderr, "[srf_conv] no bands below %.1f nm threshold — "
392 "skipping SRF correction\n", thresh * 1000.0f);
393 return NULL;
394 }
395 fprintf(stderr, "[srf_conv] SRF correction: %d/%d bands need reptran fine "
396 "(threshold %.1f nm)\n",
397 n_to_correct, n_wl, thresh * 1000.0f);
398
399 /* ── Allocate output ── */
400 SrfCorrection *srf = calloc(1, sizeof(SrfCorrection));
401 if (!srf) return NULL;
402 srf->n_wl = n_wl;
403 srf->n_h2o = n_h2o;
404 srf->corr_down = malloc((size_t)n_wl * n_h2o * sizeof(float));
405 srf->corr_up = malloc((size_t)n_wl * n_h2o * sizeof(float));
406 if (!srf->corr_down || !srf->corr_up) {
407 atcorr_srf_free(srf); return NULL;
408 }
409 /* Default: no correction */
410 for (int i = 0; i < n_wl * n_h2o; i++) {
411 srf->corr_down[i] = 1.0f;
412 srf->corr_up[i] = 1.0f;
413 }
414
415 /* ── Schedule 4 × n_h2o uvspec runs ──
416 * run_id = i_res * 2*n_h2o + i_dir * n_h2o + ih
417 * i_res : 0 = fine, 1 = coarse
418 * i_dir : 0 = down (sza), 1 = up (vza)
419 * ih : h2o grid index */
420 int total_runs = 4 * n_h2o;
421 GasSpectrum **spectra = calloc((size_t)total_runs, sizeof(GasSpectrum *));
422 if (!spectra) { atcorr_srf_free(srf); return NULL; }
423
424 #pragma omp parallel for schedule(dynamic)
425 for (int run = 0; run < total_runs; run++) {
426 int i_res = run / (2 * n_h2o);
427 int i_dir = (run % (2 * n_h2o)) / n_h2o;
428 int ih = run % n_h2o;
429
430 const char *res = (i_res == 0) ? "fine" : "coarse";
431 float sza_r = (i_dir == 0) ? sza : vza;
432 float h2o_v = lut_cfg->h2o[ih];
433
434 int tid = omp_get_thread_num();
435
436 spectra[run] = run_uvspec_gas(
437 sza_r, h2o_v, o3_du,
438 wl_min_nm, wl_max_nm,
439 res, uvspec_path, data_path, tid);
440
441 if (!spectra[run])
442 fprintf(stderr, "[srf_conv] run %d failed "
443 "(%s, %s, H2O=%.2f)\n",
444 run, res,
445 i_dir == 0 ? "down" : "up", h2o_v);
446 }
447
448 /* ── Compute correction factors ── */
449 for (int ih = 0; ih < n_h2o; ih++) {
450 int r_fd = 0 * 2 * n_h2o + 0 * n_h2o + ih; /* fine, down */
451 int r_fu = 0 * 2 * n_h2o + 1 * n_h2o + ih; /* fine, up */
452 int r_cd = 1 * 2 * n_h2o + 0 * n_h2o + ih; /* coarse, down */
453 int r_cu = 1 * 2 * n_h2o + 1 * n_h2o + ih; /* coarse, up */
454
455 GasSpectrum *fd = spectra[r_fd];
456 GasSpectrum *fu = spectra[r_fu];
457 GasSpectrum *cd = spectra[r_cd];
458 GasSpectrum *cu = spectra[r_cu];
459
460 if (!fd || !fu || !cd || !cu) continue;
461
462 for (int iw = 0; iw < n_wl; iw++) {
463 float wl_um = lut_cfg->wl[iw];
464 float wl_nm = wl_um * 1000.0f;
465 float fwhm_nm = (srf_cfg && srf_cfg->fwhm_um)
466 ? srf_cfg->fwhm_um[iw] * 1000.0f
467 : 0.0f; /* 0 → treat as delta: use nearest */
468
469 /* Skip bands above threshold */
470 if (srf_cfg && srf_cfg->fwhm_um
471 && srf_cfg->fwhm_um[iw] >= thresh)
472 continue;
473
474 int idx = iw * n_h2o + ih;
475
476 if (fwhm_nm > 0.05f) {
477 /* Gaussian SRF convolution of fine spectrum */
478 float Tfd_srf = gaussian_srf_convolve(
479 fd->wl, fd->T_gas, fd->n_pts, wl_nm, fwhm_nm);
480 float Tfu_srf = gaussian_srf_convolve(
481 fu->wl, fu->T_gas, fu->n_pts, wl_nm, fwhm_nm);
482 /* Coarse reference at band centre */
483 float Tcd_ref = interp1d_nm(cd->wl, cd->T_gas, cd->n_pts, wl_nm);
484 float Tcu_ref = interp1d_nm(cu->wl, cu->T_gas, cu->n_pts, wl_nm);
485
486 srf->corr_down[idx] = (Tcd_ref > 1e-6f) ? Tfd_srf / Tcd_ref : 1.0f;
487 srf->corr_up[idx] = (Tcu_ref > 1e-6f) ? Tfu_srf / Tcu_ref : 1.0f;
488 } else {
489 /* Delta SRF (or negligible FWHM): use fine at nearest point */
490 float Tfd = interp1d_nm(fd->wl, fd->T_gas, fd->n_pts, wl_nm);
491 float Tfu = interp1d_nm(fu->wl, fu->T_gas, fu->n_pts, wl_nm);
492 float Tcd = interp1d_nm(cd->wl, cd->T_gas, cd->n_pts, wl_nm);
493 float Tcu = interp1d_nm(cu->wl, cu->T_gas, cu->n_pts, wl_nm);
494
495 srf->corr_down[idx] = (Tcd > 1e-6f) ? Tfd / Tcd : 1.0f;
496 srf->corr_up[idx] = (Tcu > 1e-6f) ? Tfu / Tcu : 1.0f;
497 }
498
499 /* Physical clamp: correction shouldn't be outside [0.1, 3.0].
500 * Values outside this range indicate a uvspec or LUT anomaly. */
501 srf->corr_down[idx] = fmaxf(0.1f, fminf(3.0f, srf->corr_down[idx]));
502 srf->corr_up[idx] = fmaxf(0.1f, fminf(3.0f, srf->corr_up[idx]));
503 }
504 }
505
506 /* ── Cleanup ── */
507 for (int run = 0; run < total_runs; run++)
508 gas_spectrum_free(spectra[run]);
509 free(spectra);
510
511 return srf;
512}

514/* Apply SRF correction multiplicatively to all T_down and T_up in the LUT.
515 * Must be called before pixel-level inversion.
516 * LUT arrays are modified in place: T_down[*] *= corr_down, T_up[*] *= corr_up. */
517void atcorr_srf_apply(const SrfCorrection *srf,
518 const LutConfig *cfg, LutArrays *lut)
519{
520 if (!srf || !cfg || !lut) return;
521
522 int n_aod = cfg->n_aod;
523 int n_h2o = cfg->n_h2o;
524 int n_wl = cfg->n_wl;
525
526 for (int ia = 0; ia < n_aod; ia++) {
527 for (int ih = 0; ih < n_h2o; ih++) {
528 for (int iw = 0; iw < n_wl; iw++) {
529 size_t lut_idx = (size_t)ia * n_h2o * n_wl
530 + (size_t)ih * n_wl
531 + (size_t)iw;
532 int srf_idx = iw * n_h2o + ih;
533
534 float td = lut->T_down[lut_idx] * srf->corr_down[srf_idx];
535 float tu = lut->T_up[lut_idx] * srf->corr_up[srf_idx];
536
537 lut->T_down[lut_idx] = fmaxf(0.0f, fminf(1.0f, td));
538 lut->T_up[lut_idx] = fmaxf(0.0f, fminf(1.0f, tu));
539 }
540 }
541 }
