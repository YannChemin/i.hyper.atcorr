8/* Returns solar irradiance E0 in W/(m² µm) at wavelength wl_um.
9 * Source: 6SV2.1 Thuillier 2003 spectrum, 1501 points 0.25–4.0 µm at 0.0025
10 * µm step. Linear interpolation; clamped at boundaries. */
11float sixs_E0(float wl_um)
12{
13 float t = (wl_um - (float)SOLAR_TABLE_WL_START) / (float)SOLAR_TABLE_STEP;
14 int i = (int)t;
15 float f = t - (float)i;
16
17 if (i < 0) return solar_si[0];
18 if (i >= SOLAR_TABLE_N - 1) return solar_si[SOLAR_TABLE_N - 1];
19 return solar_si[i] * (1.0f - f) + solar_si[i + 1] * f;
20}

22/* Returns d² (squared Earth-Sun distance in AU) for day-of-year doy (1–365).
23 * Formula: d = 1 − 0.01670963 × cos(2π(doy−3)/365)
24 * Same convention as i.hyper.smac's radtran.py. */
25double sixs_earth_sun_dist2(int doy)
26{
27 double beta = 2.0 * M_PI * (doy - 3) / 365.0;
28 double d = 1.0 - 0.01670963 * cos(beta);
29 return d * d;
30}
