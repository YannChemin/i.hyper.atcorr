36def _make_lut(**kwargs):
37 """Helper: build LutConfig and compute LUT, merging kwargs over _BASE_CFG."""

139 f"{r[k]:.4f} → {r[k+1]:.4f}"

141 )

147 T = (
148 self.lut.T_down[:, i_h, i_w]
149 * self.lut.T_up[:, i_h, i_w]

151 for k in range(len(_AOD) - 1):

153 T[k + 1], T[k] + 1e-4,

155 f"T_total increased with AOD at "

245class TestInversion(TestCase):
246 """Tests for invert(): algebraic TOA → BOA reflectance inversion."""
247
248 def test_no_atmosphere_identity(self):
249 """With R_atm=0, T=1, s_alb=0: rho_boa must equal rho_toa."""
250 rho_toa = np.array([0.1, 0.2, 0.3, 0.5], dtype=np.float32)
251 rho_boa = ac.invert(rho_toa, 0.0, 1.0, 1.0, 0.0)
252 np.testing.assert_allclose(rho_boa, rho_toa, atol=1e-5)
253
254 def test_atmosphere_reduces_bright_surface(self):
255 """With R_atm > 0 and T < 1, rho_boa < rho_toa for a bright surface."""
256 rho_boa = ac.invert(0.5, 0.10, 0.80, 0.85, 0.10)
257 self.assertLess(float(rho_boa), 0.5)
258
259 def test_bright_surface_stays_bright(self):
260 """Very bright TOA (≈0.8) corrected must still give rho_boa > 0.5."""
261 rho_boa = ac.invert(0.80, 0.08, 0.85, 0.90, 0.05)
262 self.assertGreater(float(rho_boa), 0.5)
263
264 def test_dark_surface_stays_dark(self):
265 """Dark TOA (≈0.05) corrected must give rho_boa < 0.15."""
266 rho_boa = ac.invert(0.05, 0.03, 0.92, 0.95, 0.02)
267 self.assertLess(float(rho_boa), 0.15)
268
269 def test_full_pipeline_physical_range(self):
270 """Full pipeline: compute LUT → slice → invert; rho_boa must be in [-0.05, 1]."""
271 cfg, lut = _make_lut()
272 sl = ac.lut_slice(cfg, lut, 0.2, 2.0)
273 # Moderate TOA reflectance, realistic for land surface
274 rho_toa = np.array([0.12, 0.10, 0.09, 0.08], dtype=np.float32)
275 rho_boa = ac.invert(rho_toa, sl.R_atm, sl.T_down, sl.T_up, sl.s_alb)
276 self.assertTrue(np.all(np.isfinite(rho_boa)))
277 self.assertTrue(np.all(rho_boa >= -0.05))
278 self.assertTrue(np.all(rho_boa < 1.0))
279
280 def test_vectorised_over_pixels(self):
281 """invert() must work element-wise for an array of 100 pixels."""
282 rho_toa = np.random.uniform(0.05, 0.50, 100).astype(np.float32)
283 rho_boa = ac.invert(rho_toa, 0.05, 0.90, 0.92, 0.08)
284 self.assertEqual(len(rho_boa), 100)
285 self.assertTrue(np.all(np.isfinite(rho_boa)))

321 def test_polar_Q_positive_at_blue(self):
322 """R_atmQ > 0 at 450 nm for backscatter geometry (Rayleigh dominates)."""

324 self.assertGreater(

329 def test_polar_blue_improvement_1_to_10_pct(self):
330 """Polarization improvement at 450 nm must be 1–10% of scalar R_atm."""
331 R_vec = float(self.lut_p.R_atm[0, 0, 0])
332 R_sca = float(self.lut_s.R_atm[0, 0, 0])
333 if R_sca > 0:
334 pct = 100.0 * (R_vec - R_sca) / R_sca
335 self.assertGreater(pct, 1.0, msg=f"Polarization correction {pct:.2f}% < 1%")
336 self.assertLess(pct, 10.0, msg=f"Polarization correction {pct:.2f}% > 10%")
337
338 def test_polar_effect_decreases_from_blue_to_nir(self):
339 """Polarization effect at 450 nm must be larger than at 870 nm."""
340 diff_blue = float(self.lut_p.R_atm[0, 0, 0] - self.lut_s.R_atm[0, 0, 0])
341 diff_nir = float(self.lut_p.R_atm[0, 0, -1] - self.lut_s.R_atm[0, 0, -1])
342 self.assertGreaterEqual(
343 diff_blue, diff_nir,
344 msg=(
345 f"Polarization effect at 450nm ({diff_blue:.4f}) "
346 f"< effect at 870nm ({diff_nir:.4f})"
