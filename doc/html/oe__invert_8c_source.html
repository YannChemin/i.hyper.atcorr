20/* 1-D least-squares line fit: returns slope b in ρ = a + b×wl.
21 * Also sets *a_out to the intercept. n must be ≥ 2. */
22static void _linfit(const float *wl, const float *y, int n,
23 float *a_out, float *b_out)
24{
25 double sw = 0, swl = 0, sy = 0, swl2 = 0, swly = 0;
26 for (int i = 0; i < n; i++) {
27 sw += 1.0;
28 swl += wl[i];
29 sy += y[i];
30 swl2 += (double)wl[i] * wl[i];
31 swly += (double)wl[i] * y[i];
32 }
33 double det = sw * swl2 - swl * swl;
34 if (det == 0.0) { *a_out = (float)(sy / sw); *b_out = 0.0f; return; }
35 *a_out = (float)((sy * swl2 - swl * swly) / det);
36 *b_out = (float)((sw * swly - swl * sy) / det);
37}

39/* H₂O continuum interpolation depth (same physics as retrieve_h2o_940). */
40static float _h2o_depth(float l865, float l940, float l1040)
41{
42 static const float FRAC = (0.940f - 0.865f) / (1.040f - 0.865f);
43 if (!(l865 > 0.0f) || !(l940 > 0.0f) || !(l1040 > 0.0f)) return -1.0f;
44 float L_cont = l865 * (1.0f - FRAC) + l1040 * FRAC;
45 if (L_cont <= 0.0f) return -1.0f;
46 float D = 1.0f - l940 / L_cont;
47 return (D < 0.0f) ? 0.0f : D;
48}

52void oe_invert_aod_h2o(
53 const LutConfig *cfg,
54 const LutArrays *lut,
55 const float *rho_toa_vis,
56 int npix,
57 int n_vis,
58 const float *vis_wl,
59 const float *L_865,
60 const float *L_940,
61 const float *L_1040,
62 float sza_deg,
63 float vza_deg,
64 float aod_prior,
65 float h2o_prior,
66 float sigma_aod,
67 float sigma_h2o,
68 float sigma_spec,
69 float *out_aod,
70 float *out_h2o)
71{
72 /* Precompute H₂O constants for the J_h2o term */
73 static const float K_940 = 0.036f;
74 float cos_sza = cosf(sza_deg * (float)(M_PI / 180.0));
75 float cos_vza = cosf(vza_deg * (float)(M_PI / 180.0));
76 float mu_s = (cos_sza > 0.05f) ? cos_sza : 0.05f;
77 float mu_v = (cos_vza > 0.05f) ? cos_vza : 0.05f;
78 float km = K_940 * (1.0f / mu_s + 1.0f / mu_v);
79
80 int na = cfg->n_aod;
81 int nh = cfg->n_h2o;
82
83 /* Safety: need at least 2 VIS bands for linear fit */
84 int do_spec = (n_vis >= 2);
85
86 /* Pre-allocate per-band BOA work buffer (stack is fine for n_vis ≤ 64) */
87 float rho_boa_buf[64];
88 if (n_vis > 64) n_vis = 64;
89
90 float sigma_spec2 = sigma_spec * sigma_spec;
91 float sigma_aod2 = sigma_aod * sigma_aod;
92 float sigma_h2o2 = sigma_h2o * sigma_h2o;
93
94#ifdef _OPENMP
95#pragma omp parallel for schedule(static) private(rho_boa_buf)
96#endif
97 for (int i = 0; i < npix; i++) {
98
99 /* ── Check that all VIS bands are valid ── */
100 int valid = 1;
101 for (int b = 0; b < n_vis; b++) {
102 float rt = rho_toa_vis[i * n_vis + b];
103 if (!isfinite(rt) || rt <= 0.0f) { valid = 0; break; }
104 }
105 if (!valid) {
106 out_aod[i] = aod_prior;
107 out_h2o[i] = h2o_prior;
108 continue;
109 }
110
111 /* ── H₂O measurement constraint ── */
112 float D_940_meas = -1.0f;
113 if (L_865 && L_940 && L_1040)
114 D_940_meas = _h2o_depth(L_865[i], L_940[i], L_1040[i]);
115
116 float best_cost = FLT_MAX;
117 float best_aod = aod_prior;
118 float best_h2o = h2o_prior;
119 int best_ia = 0;
120 int best_ih = 0;
121
122 /* ── Grid search over (AOD, H₂O) LUT points ── */
123 for (int ia = 0; ia < na; ia++) {
124 float aod = cfg->aod[ia];
125
126 for (int ih = 0; ih < nh; ih++) {
127 float h2o = cfg->h2o[ih];
128
129 /* ── J_spec: invert all VIS bands and measure smoothness ── */
130 float J_spec = 0.0f;
131 if (do_spec) {
132 int ok = 1;
133 for (int b = 0; b < n_vis; b++) {
134 float R, Td, Tu, s;
135 atcorr_lut_interp_pixel(cfg, lut, aod, h2o, vis_wl[b],
136 &R, &Td, &Tu, &s);
137 float rt = rho_toa_vis[i * n_vis + b];
138 float y = (rt - R) / (Td * Tu + 1e-10f);
139 float rboa = y / (1.0f + s * y + 1e-10f);
140 if (!isfinite(rboa) || rboa < -0.05f || rboa > 1.5f) {
141 ok = 0; break;
142 }
143 rho_boa_buf[b] = rboa;
144 }
145 if (!ok) continue;
146
147 /* Linear fit residuals (spectral smoothness) */
148 float a_fit, b_fit;
149 _linfit(vis_wl, rho_boa_buf, n_vis, &a_fit, &b_fit);
150 for (int b = 0; b < n_vis; b++) {
151 float fit = a_fit + b_fit * vis_wl[b];
152 float dr = rho_boa_buf[b] - fit;
153 J_spec += dr * dr;
154 }
155 J_spec /= sigma_spec2;
156 }
157
158 /* ── J_h2o: H₂O band-depth consistency ── */
159 float J_h2o = 0.0f;
160 if (D_940_meas >= 0.0f) {
161 /* Expected D_940 for this H₂O value:
162 * D_expected = km × h2o (Beer-Lambert linear approx) */
163 float D_expected = km * h2o;
164 if (D_expected > 0.95f) D_expected = 0.95f;
165 float dD = D_940_meas - D_expected;
166 /* Normalise by a typical D uncertainty ~0.05 */
167 J_h2o = (dD * dD) / (0.05f * 0.05f);
168 }
169
170 /* ── J_prior: MAP regularisation ── */
171 float da = (logf(aod + 1e-4f) - logf(aod_prior + 1e-4f));
172 float dh = h2o - h2o_prior;
173 float J_prior = (da * da) / sigma_aod2 + (dh * dh) / sigma_h2o2;
174
175 float cost = J_spec + J_h2o + J_prior;
176
177 if (cost < best_cost) {
178 best_cost = cost;
179 best_aod = aod;
180 best_h2o = h2o;
181 best_ia = ia;
182 best_ih = ih;
183 }
184 }
185 }
186
187 /* ── Parabolic sub-grid refinement in AOD dimension ── */
188 if (best_ia > 0 && best_ia < na - 1) {
189 /* Evaluate cost at ia-1, ia, ia+1 with best_ih fixed */
190 float aod_m = cfg->aod[best_ia - 1];
191 float aod_p = cfg->aod[best_ia + 1];
192 float h2o = best_h2o;
193
194 /* Cost at the three points (recompute J terms at ia±1) */
195 float c[3];
196 for (int k = -1; k <= 1; k++) {
197 int ia2 = best_ia + k;
198 float aod2 = cfg->aod[ia2];
199 float J_s = 0.0f;
200 if (do_spec) {
201 int ok = 1;
202 for (int b = 0; b < n_vis; b++) {
203 float R, Td, Tu, s;
204 atcorr_lut_interp_pixel(cfg, lut, aod2, h2o, vis_wl[b],
205 &R, &Td, &Tu, &s);
206 float rt = rho_toa_vis[i * n_vis + b];
207 float y = (rt - R) / (Td * Tu + 1e-10f);
208 float rb = y / (1.0f + s * y + 1e-10f);
209 if (!isfinite(rb) || rb < -0.05f || rb > 1.5f) {
210 ok = 0; break;
211 }
212 rho_boa_buf[b] = rb;
213 }
214 if (ok) {
215 float a_fit, b_fit;
216 _linfit(vis_wl, rho_boa_buf, n_vis, &a_fit, &b_fit);
217 for (int b = 0; b < n_vis; b++) {
218 float dr = rho_boa_buf[b] - (a_fit + b_fit * vis_wl[b]);
219 J_s += dr * dr;
220 }
221 J_s /= sigma_spec2;
222 } else {
223 J_s = FLT_MAX;
224 }
225 }
226 float da2 = (logf(aod2 + 1e-4f) - logf(aod_prior + 1e-4f));
227 float dh2 = h2o - h2o_prior;
228 float J_p = (da2 * da2) / sigma_aod2 + (dh2 * dh2) / sigma_h2o2;
229 c[k + 1] = (J_s < FLT_MAX) ? J_s + J_p : FLT_MAX;
230 }
231
232 /* Fit parabola c(Δ) = c0 + c1×Δ + c2×Δ²; minimum at Δ* = -c1/(2×c2) */
233 float c0 = c[1], cm = c[0], cp = c[2];
234 if (cm < FLT_MAX && cp < FLT_MAX) {
235 float c2 = 0.5f * (cm - 2.0f * c0 + cp);
236 float c1 = 0.5f * (cp - cm);
237 if (c2 > 1e-10f) {
238 float delta = -c1 / (2.0f * c2);
239 if (delta > -1.0f && delta < 1.0f) {
240 /* Linearly interpolate AOD */
241 float ref = best_aod;
242 if (delta < 0.0f)
243 ref = aod_m + (best_aod - aod_m) * (1.0f + delta);
244 else
245 ref = best_aod + (aod_p - best_aod) * delta;
246 best_aod = ref;
247 if (best_aod < cfg->aod[0])
248 best_aod = cfg->aod[0];
249 if (best_aod > cfg->aod[na - 1])
250 best_aod = cfg->aod[na - 1];
251 }
252 }
253 }
254 }
255
256 /* ── Parabolic sub-grid refinement in H₂O dimension ── */
257 if (best_ih > 0 && best_ih < nh - 1) {
258 float h2o_m = cfg->h2o[best_ih - 1];
259 float h2o_p = cfg->h2o[best_ih + 1];
260 float aod = best_aod;
261
262 float c[3];
263 for (int k = -1; k <= 1; k++) {
264 int ih2 = best_ih + k;
265 float h2o2 = cfg->h2o[ih2];
266 float J_h = 0.0f;
267 if (D_940_meas >= 0.0f) {
268 float D_exp = km * h2o2;
269 if (D_exp > 0.95f) D_exp = 0.95f;
270 float dD = D_940_meas - D_exp;
271 J_h = (dD * dD) / (0.05f * 0.05f);
272 }
273 float da2 = (logf(aod + 1e-4f) - logf(aod_prior + 1e-4f));
274 float dh2 = h2o2 - h2o_prior;
275 float J_p = (da2 * da2) / sigma_aod2 + (dh2 * dh2) / sigma_h2o2;
276 c[k + 1] = J_h + J_p;
277 }
278
279 float c2 = 0.5f * (c[0] - 2.0f * c[1] + c[2]);
280 float c1 = 0.5f * (c[2] - c[0]);
281 if (c2 > 1e-10f) {
282 float delta = -c1 / (2.0f * c2);
283 if (delta > -1.0f && delta < 1.0f) {
284 float ref;
285 if (delta < 0.0f)
286 ref = h2o_m + (best_h2o - h2o_m) * (1.0f + delta);
287 else
288 ref = best_h2o + (h2o_p - best_h2o) * delta;
289 best_h2o = ref;
290 if (best_h2o < cfg->h2o[0])
291 best_h2o = cfg->h2o[0];
292 if (best_h2o > cfg->h2o[nh - 1])
293 best_h2o = cfg->h2o[nh - 1];
294 }
295 }
296 }
297
298 out_aod[i] = best_aod;
299 out_h2o[i] = best_h2o;
300 }
