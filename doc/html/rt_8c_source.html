10/* ------------------------------------------------------------------ *
11 * sixs_os: compute atmospheric path reflectance xl(-mu,1)/xmus.
12 *
13 * ctx - per-thread context (polar.betal, del.delta, quad.nquad,
14 * multi.igmax, err used)
15 * iaer_prof - 0 = exponential profile (only supported)
16 * tamoy - aerosol optical depth (truncated, DISCOM applies coeff)
17 * trmoy - Rayleigh optical depth
18 * pizmoy - truncated single scattering albedo
19 * tamoyp - aerosol OD above observer plane
20 * trmoyp - Rayleigh OD above observer plane
21 * palt - observer altitude km (>900 = satellite)
22 * phirad - relative azimuth angle (radians)
23 * nt - number of atmospheric layers
24 * mu - Gauss quadrature points per hemisphere
25 * np - number of azimuth planes stored in xl
26 * nfi - number of azimuth angles in xlphim
27 * rm_off - Gauss cosines, size 2mu+1, rm_off[j+mu]=rm(j)
28 * gb_off - Gauss weights, size 2mu+1, gb_off[j+mu]=gb(j)
29 * rp - azimuth angles for xl planes [np]
30 * xl - output radiance [(2mu+1)*np]; xl[0]=xl(-mu,1)=path refl*xmus
31 * xlphim - path reflectance vs azimuth [nfi]
32 * rolut - angular LUT reflectance [mu*61] (can be NULL to skip)
33 * ------------------------------------------------------------------ */
34void sixs_os(SixsCtx *ctx, int iaer_prof,
35 float tamoy, float trmoy, float pizmoy,
36 float tamoyp, float trmoyp, float palt,
37 float phirad, int nt, int mu, int np, int nfi,
38 const float *rm_off, const float *gb_off, const float *rp,
39 float *xl, float *xlphim, float *rolut)
40{
41 /* Local offset macros using function parameters */
42 #undef RM
43 #undef GB
44 #define RM(j) rm_off[(j)+mu]
45 #define GB(j) gb_off[(j)+mu]
46
47 int snt = nt; /* save to restore at end */
48
49 double ta = tamoy, tr = trmoy;
50 double tap = tamoy - tamoyp, trp = trmoy - trmoyp;
51 double piz = pizmoy;
52 double hr = 8.0, ha = 2.0;
53 int ntp = nt, iplane = 0;
54 double accu = 1e-20, accu2 = 1e-3;
55 int mum1 = mu - 1;
56
57 /* Plane observation: compute aerosol scale height */
58 if (palt <= 900.0 && palt > 0.0) {
59 ha = (tap > 1e-3) ? -palt / log(tap / ta) : 2.0;
60 ntp = nt - 1;
61 }
62
63 /* Allocate working arrays */
64 int sz = nt + 1;
65 int dim = 2 * mu + 1;
66
67 double *h = (double*)calloc(sz, sizeof(double));
68 double *ch = (double*)calloc(sz, sizeof(double));
69 double *xdel = (double*)calloc(sz, sizeof(double));
70 double *ydel = (double*)calloc(sz, sizeof(double));
71 double *altc = (double*)calloc(sz, sizeof(double));
72 double *i1 = (double*)calloc((size_t)sz * dim, sizeof(double));
73 double *i2 = (double*)calloc((size_t)sz * dim, sizeof(double));
74 double *i3 = (double*)calloc(dim, sizeof(double));
75 double *i4 = (double*)calloc(dim, sizeof(double));
76 double *in_ = (double*)calloc(dim, sizeof(double));
77 double *inm1 = (double*)calloc(dim, sizeof(double));
78 double *inm2 = (double*)calloc(dim, sizeof(double));
79 double *xpl = (double*)calloc(dim, sizeof(double));
80 int psl_rows = NQ_MAX + 2;
81 double *psl = (double*)calloc((size_t)psl_rows * dim, sizeof(double));
82 double *bp = (double*)calloc((size_t)(mu + 1) * dim, sizeof(double));
83
84 /* LUT azimuth arrays (optional) */
85 float *filut = rolut ? (float*)calloc((size_t)mu * 61, sizeof(float)) : NULL;
86 int *nfilut = rolut ? (int*) calloc(mu, sizeof(int)) : NULL;
87
88 if (!h || !ch || !xdel || !ydel || !altc || !i1 || !i2 || !i3 || !i4 ||
89 !in_ || !inm1 || !inm2 || !xpl || !psl || !bp)
90 goto cleanup;
91
92 /* Array accessors (negative-indexed via offset) */
93 #define I1(k,j) i1 [(size_t)(k)*dim + (j)+mu]
94 #define I2(k,j) i2 [(size_t)(k)*dim + (j)+mu]
95 #define I3(j) i3 [(j)+mu]
96 #define I4(j) i4 [(j)+mu]
97 #define IN(j) in_[(j)+mu]
98 #define INM1(j) inm1[(j)+mu]
99 #define INM2(j) inm2[(j)+mu]
100 #define XPL(j) xpl [(j)+mu]
101 #define BP(j,k) bp [(j)*dim + (k)+mu]
102 #define XL(m,l) xl [((m)+mu)*np + (l)]
103 #define FILUT(i,j) filut [(i)*61 + (j)]
104 #define ROLUT(i,j) rolut [(i)*61 + (j)]
105
106 double xmus = -(double)RM(0); /* solar cosine (rm(0) = -xmus) */
107
108 /* ===== Layer grid setup ===== */
109 if (ta <= accu2 && tr > ta) {
110 /* Pure Rayleigh */
111 for (int j = 0; j <= ntp; j++) {
112 h[j] = (double)j * tr / ntp;
113 ch[j] = exp(-h[j] / xmus) * 0.5;
114 ydel[j] = 1.0; xdel[j] = 0.0;
115 altc[j] = (j == 0) ? 300.0 : -log(h[j] / tr) * hr;
116 }
117 } else if (tr <= accu2 && ta > tr) {
118 /* Pure aerosol */
119 for (int j = 0; j <= ntp; j++) {
120 h[j] = (double)j * ta / ntp;
121 ch[j] = exp(-h[j] / xmus) * 0.5;
122 ydel[j] = 0.0; xdel[j] = piz;
123 altc[j] = (j == 0) ? 300.0 : -log(h[j] / ta) * ha;
124 }
125 } else {
126 /* Mixed Rayleigh + aerosol: use discre bisection */
127 ydel[0] = 1.0; xdel[0] = 0.0; h[0] = 0.0; ch[0] = 0.5; altc[0] = 300.0;
128 double zx = 300.0;
129 for (int it = 0; it <= ntp; it++) {
130 double yy = (it > 0) ? h[it-1] : 0.0;
131 double dd = (it > 0) ? ydel[it-1] : 0.0;
132 int itp = it;
133 sixs_discre(ctx, ta, ha, tr, hr, itp, ntp, yy, dd, 300.0, 0.0, &zx);
134 if (ctx->err.ier) goto cleanup;
135 double xxx = -zx / ha;
136 double ca = (xxx < -18.0) ? 0.0 : ta * exp(xxx);
137 double cr = tr * exp(-zx / hr);
138 h[it] = ca + cr;
139 altc[it] = zx;
140 ch[it] = exp(-h[it] / xmus) * 0.5;
141 cr /= hr; ca /= ha;
142 double ratio = (cr + ca > 0.0) ? cr / (cr + ca) : 0.5;
143 xdel[it] = (1.0 - ratio) * piz;
144 ydel[it] = ratio;
145 }
146 }
147
148 /* ===== Plane layer update ===== */
149 if (ntp == nt - 1) {
150 double taup = tap + trp;
151 iplane = -1;
152 for (int i = 0; i <= ntp; i++) if (taup >= h[i]) iplane = i;
153 double th = 0.0005;
154 double xt1 = fabs(h[iplane] - taup);
155 double xt2 = fabs(h[iplane+1] - taup);
156 if (xt1 > th && xt2 > th) {
157 for (int i = nt; i >= iplane + 1; i--) {
158 xdel[i] = xdel[i-1]; ydel[i] = ydel[i-1];
159 h[i] = h[i-1]; altc[i] = altc[i-1]; ch[i] = ch[i-1];
160 }
161 } else {
162 nt = ntp;
163 if (xt2 < xt1) iplane++;
164 }
165 h[iplane] = taup;
166 if (tr > accu2 && ta > accu2) {
167 double ca = ta * exp(-palt / ha), cr = tr * exp(-palt / hr);
168 double hi = ca + cr;
169 h[iplane] = hi;
170 ch[iplane] = exp(-hi / xmus) * 0.5;
171 cr /= hr; ca /= ha;
172 double ratio = cr / (cr + ca);
173 xdel[iplane] = (1.0 - ratio) * piz;
174 ydel[iplane] = ratio;
175 altc[iplane] = palt;
176 } else if (tr > accu2) {
177 ydel[iplane] = 1.0; xdel[iplane] = 0.0; altc[iplane] = palt;
178 } else {
179 ydel[iplane] = 0.0; xdel[iplane] = piz; altc[iplane] = palt;
180 }
181 }
182
183 /* ===== Initialize output arrays ===== */
184 double pi = acos(-1.0);
185 for (int l = 0; l < np; l++)
186 for (int m = -mu; m <= mu; m++) XL(m, l) = 0.0f;
187 for (int ifi = 0; ifi < nfi; ifi++) xlphim[ifi] = 0.0f;
188
189 /* LUT azimuth angle table */
190 if (rolut && filut && nfilut) {
191 double its_deg = acos(xmus) * 180.0 / pi;
192 double sin_sza = sqrt(fmax(0.0, 1.0 - xmus * xmus));
193 for (int m2 = 0; m2 < mu; m2++) {
194 nfilut[m2] = 0;
195 for (int j = 0; j < 61; j++) { FILUT(m2,j)=0.f; ROLUT(m2,j)=0.f; }
196 double lutmuv = (double)RM(m2 + 1);
197 double luttv = acos(lutmuv) * 180.0 / pi;
198 double iscama = 180.0 - fabs(luttv - its_deg);
199 double iscami = 180.0 - (luttv + its_deg);
200 int nbisca = (int)((iscama - iscami) / 4.0) + 1;
201 if (nbisca > 61) nbisca = 61;
202 if (nbisca < 1) nbisca = 1;
203 nfilut[m2] = nbisca;
204 FILUT(m2, 0) = 0.0f;
205 FILUT(m2, nbisca - 1) = 180.0f;
206 double scaa = iscama;
207 double sin_vza = sqrt(fmax(0.0, 1.0 - lutmuv * lutmuv));
208 for (int j = 1; j < nbisca - 1; j++) {
209 scaa -= 4.0;
210 double cscaa = cos(scaa * pi / 180.0);
211 if (sin_sza < 1e-9 || sin_vza < 1e-9) { FILUT(m2,j)=0.f; continue; }
212 double cfi = -(cscaa + xmus * lutmuv) / (sin_sza * sin_vza);
213 cfi = fmax(-1.0, fmin(1.0, cfi));
214 FILUT(m2, j) = (float)(acos(cfi) * 180.0 / pi);
215 }
216 }
217 }
218
219 /* ===== Rayleigh depolarization parameters ===== */
220 double aaaa = (double)ctx->del.delta / (2.0 - (double)ctx->del.delta);
221 double ron = (1.0 - aaaa) / (1.0 + 2.0 * aaaa);
222 double beta0 = 1.0, beta2 = 0.5 * ron;
223
224 /* i4 accumulates Fourier sum across all orders */
225 for (int j = -mu; j <= mu; j++) I4(j) = 0.0;
226
227 int iborm = ctx->quad.nquad - 3;
228 if (fabs(xmus - 1.0) < 1e-6) iborm = 0;
229
230 /* ===== Fourier decomposition loop: is = 0 .. iborm ===== */
231 for (int is = 0; is <= iborm; is++) {
232 int ig = 1;
233 double roavion = 0.0, roavion0 = 0.0, roavion1 = 0.0, roavion2 = 0.0;
234 for (int j = -mu; j <= mu; j++) I3(j) = 0.0;
235
236 /* Reset beta0 for is > 0 (Rayleigh has no azimuthal component) */
237 if (is > 0) beta0 = 0.0;
238
239 /* Kernel for this Fourier order */
240 sixs_kernel(ctx, is, mu, rm_off, xpl, psl, bp);
241
242 /* Primary scattering source: i2(k,j) = ch[k]*(sa2*xdel[k]+sa1*ydel[k]) */
243 for (int j = -mu; j <= mu; j++) {
244 double spl = XPL(0);
245 double sa1, sa2;
246 if (is <= 2) {
247 sa1 = beta0 + beta2 * XPL(j) * spl;
248 sa2 = BP(0, j);
249 } else {
250 sa1 = 0.0;
251 sa2 = BP(0, j);
252 }
253 for (int k = 0; k <= nt; k++)
254 I2(k, j) = ch[k] * (sa2 * xdel[k] + sa1 * ydel[k]);
255 }
256
257 /* Vertical integration, primary upward (k = 1..mu) */
258 for (int k = 1; k <= mu; k++) {
259 I1(nt, k) = 0.0;
260 double zi1 = 0.0, yy = (double)RM(k);
261 for (int i = nt - 1; i >= 0; i--) {
262 int jj = i + 1;
263 double f = h[jj] - h[i];
264 double a = (I2(jj,k) - I2(i,k)) / f;
265 double b = I2(i,k) - a * h[i];
266 double c = exp(-f / yy);
267 double d = 1.0 - c;
268 double xx = h[i] - h[jj] * c;
269 zi1 = c * zi1 + (d * (b + a * yy) + a * xx) * 0.5;
270 I1(i, k) = zi1;
271 }
272 }
273
274 /* Vertical integration, primary downward (k = -mu..-1) */
275 for (int k = -mu; k <= -1; k++) {
276 I1(0, k) = 0.0;
277 double zi1 = 0.0, yy = (double)RM(k);
278 for (int i = 1; i <= nt; i++) {
279 int jj = i - 1;
280 double f = h[i] - h[jj];
281 double c = exp(f / yy);
282 double d = 1.0 - c;
283 double a = (I2(i,k) - I2(jj,k)) / f;
284 double b = I2(i,k) - a * h[i];
285 double xx = h[i] - h[jj] * c;
286 zi1 = c * zi1 + (d * (b + a * yy) + a * xx) * 0.5;
287 I1(i, k) = zi1;
288 }
289 }
290
291 /* Collect primary scattering: inm2 = inm1 = i3 = i1 at boundaries */
292 for (int k = -mu; k <= mu; k++) {
293 if (k == 0) continue;
294 int idx = (k < 0) ? nt : 0;
295 INM1(k) = I1(idx, k);
296 INM2(k) = I1(idx, k);
297 I3(k) = I1(idx, k);
298 }
299 roavion2 = I1(iplane, mu);
300 roavion = I1(iplane, mu);
301
302 /* ---- Successive Orders of Scattering ---- */
303 for (;;) {
304 ig++;
305
306 /* Build source function from previous i1 */
307 if (is <= 2) {
308 /* Mixed Rayleigh + aerosol kernel */
309 for (int k = 1; k <= mu; k++) {
310 double xpk = XPL(k), ypk = XPL(-k);
311 for (int i = 0; i <= nt; i++) {
312 double ii1 = 0.0, ii2 = 0.0;
313 double x = xdel[i], y = ydel[i];
314 for (int j = 1; j <= mu; j++) {
315 double xpj = XPL(j), z = (double)GB(j);
316 double xi1 = I1(i, j), xi2 = I1(i, -j);
317 double bpjk = BP(j, k) * x + y * (beta0 + beta2 * xpj * xpk);
318 double bpjmk = BP(j, -k) * x + y * (beta0 + beta2 * xpj * ypk);
319 ii2 += z * (xi1 * bpjk + xi2 * bpjmk);
320 ii1 += z * (xi1 * bpjmk + xi2 * bpjk);
321 }
322 if (fabs(ii2) < 1e-30) ii2 = 0.0;
323 if (fabs(ii1) < 1e-30) ii1 = 0.0;
324 I2(i, k) = ii2;
325 I2(i, -k) = ii1;
326 }
327 }
328 } else {
329 /* Aerosol-only kernel (is > 2, no Rayleigh term) */
330 for (int k = 1; k <= mu; k++) {
331 for (int i = 0; i <= nt; i++) {
332 double ii1 = 0.0, ii2 = 0.0;
333 double x = xdel[i];
334 for (int j = 1; j <= mu; j++) {
335 double z = (double)GB(j);
336 double xi1 = I1(i, j), xi2 = I1(i, -j);
337 double bpjk = BP(j, k) * x;
338 double bpjmk = BP(j, -k) * x;
339 ii2 += z * (xi1 * bpjk + xi2 * bpjmk);
340 ii1 += z * (xi1 * bpjmk + xi2 * bpjk);
341 }
342 if (fabs(ii2) < 1e-30) ii2 = 0.0;
343 if (fabs(ii1) < 1e-30) ii1 = 0.0;
344 I2(i, k) = ii2;
345 I2(i, -k) = ii1;
346 }
347 }
348 }
349
350 /* Vertical integration upward */
351 for (int k = 1; k <= mu; k++) {
352 I1(nt, k) = 0.0;
353 double zi1 = 0.0, yy = (double)RM(k);
354 for (int i = nt - 1; i >= 0; i--) {
355 int jj = i + 1;
356 double f = h[jj] - h[i];
357 double a = (I2(jj,k) - I2(i,k)) / f;
358 double b = I2(i,k) - a * h[i];
359 double c = exp(-f / yy);
360 double d = 1.0 - c;
361 double xx = h[i] - h[jj] * c;
362 zi1 = c * zi1 + (d * (b + a * yy) + a * xx) * 0.5;
363 if (fabs(zi1) < 1e-20) zi1 = 0.0;
364 I1(i, k) = zi1;
365 }
366 }
367
368 /* Vertical integration downward */
369 for (int k = -mu; k <= -1; k++) {
370 I1(0, k) = 0.0;
371 double zi1 = 0.0, yy = (double)RM(k);
372 for (int i = 1; i <= nt; i++) {
373 int jj = i - 1;
374 double f = h[i] - h[jj];
375 double c = exp(f / yy);
376 double d = 1.0 - c;
377 double a = (I2(i,k) - I2(jj,k)) / f;
378 double b = I2(i,k) - a * h[i];
379 double xx = h[i] - h[jj] * c;
380 zi1 = c * zi1 + (d * (b + a * yy) + a * xx) * 0.5;
381 if (fabs(zi1) < 1e-20) zi1 = 0.0;
382 I1(i, k) = zi1;
383 }
384 }
385
386 /* Collect scattering order n */
387 for (int k = -mu; k <= mu; k++) {
388 if (k == 0) continue;
389 IN(k) = I1((k < 0) ? nt : 0, k);
390 }
391 roavion0 = I1(iplane, mu);
392
393 /* Geometric series convergence test */
394 if (ig > 2) {
395 double z = 0.0;
396 if (roavion2 >= accu && roavion1 >= accu && roavion >= accu) {
397 double r = roavion0 / roavion1;
398 double y = fabs(((r - roavion1/roavion2) /
399 ((1.0 - r) * (1.0 - r))) *
400 (roavion0 / roavion));
401 z = fmax(y, z);
402 }
403 for (int l = -mu; l <= mu; l++) {
404 if (l == 0) continue;
405 if (INM2(l) <= accu || INM1(l) <= accu || I3(l) <= accu) continue;
406 double r = IN(l) / INM1(l);
407 double y = fabs(((r - INM1(l)/INM2(l)) /
408 ((1.0 - r) * (1.0 - r))) *
409 (IN(l) / I3(l)));
410 z = fmax(y, z);
411 }
412 if (z < 0.0001) {
413 /* Apply geometric series acceleration */
414 for (int l = -mu; l <= mu; l++) {
415 if (l == 0) continue;
416 double d1 = INM1(l), g1 = IN(l);
417 if (d1 <= accu || fabs(g1 - d1) <= accu) continue;
418 I3(l) += g1 / (1.0 - g1 / d1);
419 }
420 if (roavion1 >= accu && fabs(roavion0 - roavion1) >= accu)
421 roavion += roavion0 / (1.0 - roavion0 / roavion1);
422 goto done_sos;
423 }
424 for (int k = -mu; k <= mu; k++) INM2(k) = INM1(k);
425 roavion2 = roavion1;
426 }
427
428 for (int k = -mu; k <= mu; k++) INM1(k) = IN(k);
429 roavion1 = roavion0;
430 for (int l = -mu; l <= mu; l++) I3(l) += IN(l);
431 roavion += roavion0;
432
433 /* Relative convergence: stop if last order < 1e-5 of sum */
434 {
435 double z = 0.0;
436 for (int l = -mu; l <= mu; l++) {
437 if (l == 0 || fabs(I3(l)) < accu) continue;
438 z = fmax(z, fabs(IN(l) / I3(l)));
439 }
440 if (z < 1e-5) goto done_sos;
441 }
442 if (ig >= ctx->multi.igmax) goto done_sos;
443 } /* end SOS loop */
444 done_sos:;
445
446 /* ---- Accumulate this Fourier component ---- */
447 double delta0s = (is == 0) ? 1.0 : 2.0;
448 for (int l = -mu; l <= mu; l++) I4(l) += delta0s * I3(l);
449
450 /* xl: radiance field at each azimuth plane */
451 for (int l_idx = 0; l_idx < np; l_idx++) {
452 double phi_l = (double)rp[l_idx];
453 for (int m = -mum1; m <= mum1; m++) {
454 double cosfac = (m > 0) ? cos(is * (phi_l + pi)) : cos(is * phi_l);
455 XL(m, l_idx) += (float)(delta0s * I3(m) * cosfac);
456 }
457 }
458
459 /* xl(mu, 1): upward at specific view direction */
460 XL(mu, 0) += (float)(delta0s * I3(mu) * cos(is * (phirad + pi)));
461
462 /* xl(-mu, 1): atmospheric path reflectance (×xmus) */
463 XL(-mu, 0) += (float)(delta0s * roavion * cos(is * (phirad + pi)));
464
465 /* xl(0, 1): downward flux at surface (is=0 only) */
466 if (is == 0) {
467 for (int k = 1; k <= mum1; k++)
468 XL(0, 0) += (float)((double)RM(k) * (double)GB(k) * I3(-k));
469 }
470
471 /* xlphim: path reflectance vs azimuth angle */
472 for (int ifi = 0; ifi < nfi; ifi++) {
473 double phimul = (double)ifi * pi / (double)(nfi - 1);
474 xlphim[ifi] += (float)(delta0s * roavion * cos(is * (phimul + pi)));
475 }
476
477 /* LUT angular table */
478 if (rolut && filut && nfilut) {
479 for (int m2 = 0; m2 < mu; m2++) {
480 for (int l2 = 0; l2 < nfilut[m2]; l2++) {
481 double phimul = (double)FILUT(m2, l2) * pi / 180.0;
482 ROLUT(m2, l2) += (float)(delta0s * I3(m2 + 1) *
483 cos(is * (phimul + pi)));
484 }
485 }
486 }
487
488 /* Fourier convergence: exit early if last term < 0.1% of sum */
489 {
490 double z = 0.0;
491 for (int l = -mu; l <= mu; l++) {
492 if (fabs(I4(l)) < accu) continue;
493 z = fmax(z, fabs(I3(l) / I4(l)));
494 }
495 if (z <= 0.001) break;
496 }
497 } /* end Fourier loop */
498
499 nt = snt; /* restore nt */
500
501cleanup:

503 #undef GB
504 #undef I1
505 #undef I2
506 #undef I3
507 #undef I4

509 #undef INM1
510 #undef INM2
511 #undef XPL
512 #undef BP
513 #undef XL
514 #undef FILUT
515 #undef ROLUT
