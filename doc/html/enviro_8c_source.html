1/* Environmental reflectance correction — ported from 6SV2.1 ENVIRO.f.
2 *
3 * Computes the fraction of diffuse irradiance originating from the
4 * neighbourhood of the target pixel (adjacency effect correction factors)
5 * as a function of sensor altitude and aerosol+Rayleigh optical depths. */

8/* Altitude grid (16 levels, km) and polynomial coefficients.

11 * (nadir-view look-up table, corrected for view-angle effect below). */

37/* Environmental correction factors.
38 *
39 * difr : diffuse Rayleigh optical depth at the target wavelength
40 * difa : diffuse aerosol optical depth at the target wavelength
41 * r : total aerosol OD at 550 nm (for the nadir-view LUT lookup)
42 * palt : sensor altitude (km)
43 * xmuv : cosine of the view zenith angle
44 *
45 * Outputs:
46 * fra : Rayleigh environmental function (fraction of env. contribution)
47 * fae : aerosol environmental function
48 * fr : combined fraction (weighted average of fra and fae) */
49void sixs_enviro(float difr, float difa, float r, float palt, float xmuv,
50 float *fra, float *fae, float *fr)
51{
52 const float a0 = 1.3347f, b0 = 0.57757f;
53 const float a1 = -1.479f, b1 = -1.5275f;
54
55 float fae0, fra0;
56
57 if (palt >= 60.0f) {
58 /* Above 60 km: use asymptotic coefficients */
59 fae0 = 1.0f - 0.448f * expf(-r * 0.27f) - 0.552f * expf(-r * 2.83f);
60 fra0 = 1.0f - 0.930f * expf(-r * 0.080f) - 0.070f * expf(-r * 1.100f);
61 } else {
62 /* Find bracketing altitude levels */
63 int idx = 0;
64 while (idx < 15 && palt >= alt_lut[idx]) idx++;
65 /* idx points to first level with alt_lut[idx] > palt */
66
67 float xcfr1_v, xcfr2_v, xcfa1_v, xcfa2_v, xcfa3_v;
68 if (idx == 0) {
69 /* Below lowest level */
70 xcfr1_v = cfr1[0]; xcfr2_v = cfr2[0];
71 xcfa1_v = cfa1[0]; xcfa2_v = cfa2[0]; xcfa3_v = cfa3[0];
72 } else {
73 /* Linear interpolation between idx-1 and idx */
74 float zmin = alt_lut[idx - 1];
75 float zmax = alt_lut[idx];
76 float t = (palt - zmin) / (zmax - zmin);
77 xcfr1_v = cfr1[idx-1] + (cfr1[idx] - cfr1[idx-1]) * t;
78 xcfr2_v = cfr2[idx-1] + (cfr2[idx] - cfr2[idx-1]) * t;
79 xcfa1_v = cfa1[idx-1] + (cfa1[idx] - cfa1[idx-1]) * t;
80 xcfa2_v = cfa2[idx-1] + (cfa2[idx] - cfa2[idx-1]) * t;
81 xcfa3_v = cfa3[idx-1] + (cfa3[idx] - cfa3[idx-1]) * t;
82 }
83 fra0 = 1.0f - xcfr1_v * expf(-r * xcfr2_v)
84 - (1.0f - xcfr1_v) * expf(-r * 0.08f);
85 fae0 = 1.0f - xcfa1_v * expf(-r * xcfa2_v)
86 - (1.0f - xcfa1_v) * expf(-r * xcfa3_v);
87 }
88
89 /* View-angle correction */
90 float xlnv = logf(xmuv);
91 *fra = fra0 * (xlnv * (1.0f - fra0) + 1.0f);
92 *fae = fae0 * ((1.0f + a0*xlnv + b0*xlnv*xlnv)
93 + fae0 * (a1*xlnv + b1*xlnv*xlnv)
94 + fae0*fae0 * ((-a1 - a0)*xlnv + (-b1 - b0)*xlnv*xlnv));
95
96 /* Combined factor weighted by diffuse OD contributions */
97 if ((difa + difr) > 1.0e-3f) {
98 *fr = (*fae * difa + *fra * difr) / (difa + difr);
99 } else {
100 *fr = 1.0f;
101 }
102}
