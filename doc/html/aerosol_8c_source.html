45/* Initialize aerosol optical properties in context for a given model and AOD.
46 * iaer: 0=none, 1=continental, 2=maritime, 3=urban, 5=desert(treated as continental),
47 * 9=custom Mie log-normal (r_mode,sigma_g,m_r,m_i set externally via sixs_mie_init).
48 * taer55: aerosol optical depth at 550nm
49 * xmud: cosine of scattering angle (for phase function)
50 * For AEROSOL_CUSTOM the caller must have already called sixs_mie_init() to populate
51 * ctx->aer; this function then only checks the no-aerosol path.
52 */
53void sixs_aerosol_init(SixsCtx *ctx, int iaer, float taer55, float xmud) {
54 if (iaer == 0 || taer55 <= 0.0f) {
55 memset(&ctx->aer, 0, sizeof(ctx->aer));
56 /* For pure Rayleigh: ext[8]=1 to avoid divide-by-zero in DISCOM */
57 ctx->aer.ext[7] = 1.0f;
58 return;
59 }
60
61 /* Custom Mie: ctx->aer already populated by sixs_mie_init() */
62 if (iaer == 9) {
63 /* Only rebuild ctx->polar from ctx->aer (re-use the Legendre expansion
64 * that sixs_mie_init already filled). Nothing more to do here. */
65 return;
66 }
67
68 /* Map iaer to mixing index */
69 int mix_idx;
70 switch (iaer) {
71 case 2: mix_idx = 1; break;
72 case 3: mix_idx = 2; break;
73 default: mix_idx = 0; break; /* continental or desert */
74 }
75
76 const float *ci = std_mix[mix_idx];
77
78 /* Compute cij (normalized concentration) using vi */
79 float sigm = 0.0f;
80 for (int j = 0; j < 4; j++) sigm += (ci[j] > 0.0f ? ci[j] / vi[j] : 0.0f);
81 if (sigm == 0.0f) sigm = 1.0f;
82
83 double cij[4] = {0}, sumni = 0.0;
84 for (int j = 0; j < 4; j++) {
85 cij[j] = (ci[j] > 0.0f) ? (ci[j] / vi[j] / sigm) : 0.0;
86 sumni += cij[j] * comp_ext[j][7]; /* normalize at index 8 = 550nm */
87 }
88 double nis = (sumni > 0.0) ? 1.0 / sumni : 1.0;
89
90 /* Build Gauss quadrature grid for phase function interpolation */
91 float cgaus_S[NQ_MAX], pdgs_S[NQ_MAX];
92 int nquad = ctx->quad.nquad;
93 sixs_gauss_setup(nquad, cgaus_S, pdgs_S);
94
95 /* Find Gauss interval for xmud */
96 int j1 = 0;
97 for (int k = 0; k < nquad - 1; k++) {
98 if (xmud >= cgaus_S[k] && xmud < cgaus_S[k + 1]) { j1 = k; break; }
99 }
100 int j2 = j1 + 1;
101 float coef = -(xmud - cgaus_S[j1]) / (cgaus_S[j2] - cgaus_S[j1]);
102
103 /* Mix optical properties at each reference wavelength */
104 float ext[NWL_DISC] = {0}, sca[NWL_DISC] = {0}, gasym[NWL_DISC] = {0};
105 float phase[NWL_DISC] = {0};
106
107 for (int l = 0; l < NWL_DISC; l++) {
108 for (int j = 0; j < 4; j++) {
109 if (cij[j] == 0.0) continue;
110 float dd = comp_pha[j][l][j1] + coef * (comp_pha[j][l][j1] - comp_pha[j][l][j2]);
111 ext[l] += (float)(comp_ext[j][l] * cij[j]);
112 sca[l] += (float)(comp_sca[j][l] * cij[j]);
113 gasym[l] += (float)(comp_sca[j][l] * cij[j] * comp_asy[j][l]);
114 phase[l] += (float)(comp_sca[j][l] * cij[j] * dd);
115 }
116 if (sca[l] > 0.0f) {
117 gasym[l] /= sca[l];
118 phase[l] /= sca[l];
119 }
120 ext[l] *= (float)nis;
121 sca[l] *= (float)nis;
122 }
123
124 for (int l = 0; l < NWL_DISC; l++) {
125 ctx->aer.ext[l] = ext[l];
126 ctx->aer.ome[l] = (ext[l] > 0.0f) ? sca[l] / ext[l] : 0.0f;
127 ctx->aer.gasym[l] = gasym[l];
128 ctx->aer.phase[l] = phase[l];
129 }
130
131 /* Build Legendre expansion in ctx->polar (for TRUNCA equivalent) */
132 /* Compute betal coefficients from phase function at Gauss points */
133 float pha_full[NQ_MAX];
134 /* For simplicity, interpolate from 20-wavelength phase to all nquad points */
135 /* Use the mixed phase function: we need to build phasel[nquad] from component data */
136 /* Mix phase function at all Gauss points (for reference wavelength 8 = 550nm, l=7) */
137 memset(pha_full, 0, nquad * sizeof(float));
138 for (int k = 0; k < nquad; k++) {
139 for (int j = 0; j < 4; j++) {
140 if (cij[j] == 0.0) continue;
141 pha_full[k] += (float)(comp_sca[j][7] * cij[j] * comp_pha[j][7][k]);
142 }
143 if (sca[7] > 0.0f) pha_full[k] /= sca[7];
144 }
145 memcpy(ctx->polar.pha, pha_full, nquad * sizeof(float));
146
147 /* Compute Legendre coefficients (TRUNCA logic) */
148 float pl[NQ_MAX + 2];
149 for (int kk = 0; kk <= nquad - 3; kk++) ctx->polar.betal[kk] = 0.0f;
150 for (int j = 0; j < nquad; j++) {
151 float x = pha_full[j] * pdgs_S[j];
152 float rm = cgaus_S[j];
153 float plm1 = 0.0f, pl0 = 1.0f;
154 for (int k = 0; k <= nquad - 3; k++) {
155 float plnew = ((2.0f * k + 1.0f) * rm * pl0 - k * plm1) / (k + 1.0f);
156 ctx->polar.betal[k] += x * pl0;
157 plm1 = pl0; pl0 = plnew;
158 }
159 }
160 for (int k = 0; k <= nquad - 3; k++) {
161 ctx->polar.betal[k] *= (2.0f * k + 1.0f) * 0.5f;
162 if (ctx->polar.betal[k] < 0.0f) {
163 for (int j = k; j <= nquad - 3; j++) ctx->polar.betal[j] = 0.0f;
164 break;
165 }
166 }
167}
