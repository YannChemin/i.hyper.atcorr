43/* ------------------------------------------------------------------ *
44 * atcorr_compute_lut: fill LutArrays for the given LutConfig.
45 *
46 * LUT layout (C order, last index varies fastest):
47 * R_atm [n_aod][n_h2o][n_wl]
48 * T_down[n_aod][n_h2o][n_wl]
49 * T_up [n_aod][n_h2o][n_wl]
50 * s_alb [n_aod][n_h2o][n_wl] (s_alb independent of H2O, but stored for API)
51 *
52 * Returns 0 on success, -1 on allocation failure.
53 * ------------------------------------------------------------------ */
54int atcorr_compute_lut(const LutConfig *cfg, LutArrays *out)
55{
56 int n_wl = cfg->n_wl;
57 int n_aod = cfg->n_aod;
58 int n_h2o = cfg->n_h2o;
59 size_t n = (size_t)n_aod * n_h2o * n_wl;
60
61 if (!out->R_atm || !out->T_down || !out->T_up || !out->s_alb)
62 return -1;
63
64 /* Initialize output arrays */
65 memset(out->R_atm, 0, n * sizeof(float));
66 memset(out->s_alb, 0, n * sizeof(float));
67 for (size_t i = 0; i < n; i++) {
68 out->T_down[i] = 1.0f;
69 out->T_up[i] = 1.0f;
70 }
71 if (out->T_down_dir)
72 for (size_t i = 0; i < n; i++) out->T_down_dir[i] = 1.0f;
73 if (out->R_atmQ)
74 memset(out->R_atmQ, 0, n * sizeof(float));
75 if (out->R_atmU)
76 memset(out->R_atmU, 0, n * sizeof(float));
77
78 /* Geometry */
79 float xmus = cosf(cfg->sza * (float)M_PI / 180.0f);
80 float xmuv = cosf(cfg->vza * (float)M_PI / 180.0f);
81 float phirad = cfg->raa * (float)M_PI / 180.0f;
82 float phi = cfg->raa;
83 float xmud = -xmus * xmuv -
84 sqrtf(1.0f - xmus*xmus) * sqrtf(1.0f - xmuv*xmuv) *
85 cosf(phirad); /* cos(scattering angle) */
86
87 /* 6SV internal parameters */
88 int nt = NT_P; /* 30 atmospheric layers */
89 int mu = MU_P; /* 25 Gauss points per hemisphere */
90 int np = 1; /* one azimuth plane */
91 int ipol = cfg->enable_polar ? 1 : 0; /* 0=scalar, 1=Stokes(I,Q,U) */
92 int idatmp = (cfg->altitude_km > 900.0f) ? 99 : /* satellite */
93 (cfg->altitude_km > 0.0f) ? 4 : 0; /* plane or ground */
94 float palt = (cfg->altitude_km > 900.0f) ? 1000.0f : cfg->altitude_km;
95 float ftray = 0.0f; /* fraction above plane (0 = ground sensor) */
96
97 /* ===== Outer loop: AOD ===== */
98#ifdef _OPENMP
99#pragma omp parallel for schedule(dynamic,1)
100#endif
101 for (int ia = 0; ia < n_aod; ia++) {
102 /* Each thread needs its own context for thread safety */
103 SixsCtx *ctx = (SixsCtx*)calloc(1, sizeof(SixsCtx));
104 if (!ctx) continue;
105
106 float aod = cfg->aod[ia];
107
108 /* Initialize atmosphere and aerosol */
109 ctx->quad.nquad = NQ_P; /* 83 Gauss points */
110 ctx->multi.igmax = 20;
111 ctx->err.ier = false;
112 sixs_init_atmosphere(ctx, cfg->atmo_model);
113 if (cfg->surface_pressure > 0.0f)
114 sixs_pressure(ctx, cfg->surface_pressure);
115
116 /* Custom Mie: populate ctx->aer before aerosol_init */
117 if (cfg->aerosol_model == AEROSOL_CUSTOM &&
118 cfg->mie_r_mode > 0.0f && cfg->mie_sigma_g > 1.0f) {
119 sixs_mie_init(ctx,
120 (double)cfg->mie_r_mode,
121 (double)cfg->mie_sigma_g,
122 (double)cfg->mie_m_real,
123 (double)cfg->mie_m_imag);
124 }
125
126 sixs_aerosol_init(ctx, cfg->aerosol_model, aod, xmud);
127
128 /* Call DISCOM once per AOD: fills ctx->disc at 20 reference wavelengths */
129 float taer55p = aod; /* satellite: no aerosol above */
130 if (cfg->altitude_km <= 900.0f) taer55p = 0.0f;
131 sixs_discom(ctx, idatmp, cfg->aerosol_model,
132 xmus, xmuv, phi, aod, taer55p, palt, phirad,
133 nt, mu, np, ftray, ipol);
134
135 if (ctx->err.ier) { free(ctx); continue; }
136
137 /* ===== Inner loop: H2O ===== */
138 for (int ih = 0; ih < n_h2o; ih++) {
139 float h2o = cfg->h2o[ih];
140
141 /* ===== Innermost loop: wavelength ===== */
142 for (int iw = 0; iw < n_wl; iw++) {
143 float wl = cfg->wl[iw];
144 size_t idx = ((size_t)ia * n_h2o + ih) * n_wl + iw;
145
146 /* Scattering quantities (no H2O dependence) via interpolation */
147 float roatm, T_down_sca, T_up_sca, s_alb, T_down_dir_sca = 1.0f;
148 sixs_interp(ctx, cfg->aerosol_model, wl,
149 aod, taer55p,
150 &roatm, &T_down_sca, &T_up_sca, &s_alb,
151 NULL, NULL,
152 out->T_down_dir ? &T_down_dir_sca : NULL);
153
154 /* Gas transmittance (H2O-dependent): separate solar/view paths */
155 float T_gas_down = sixs_gas_transmittance(ctx, wl, xmus, xmuv,
156 h2o, cfg->ozone_du);
157 float T_gas_up = sixs_gas_transmittance(ctx, wl, xmuv, xmuv,
158 h2o, cfg->ozone_du);
159
160 /* Combined transmittances */
161 out->R_atm [idx] = roatm;
162 out->T_down[idx] = T_down_sca * T_gas_down;
163 out->T_up [idx] = T_up_sca * T_gas_up;
164 out->s_alb [idx] = s_alb;
165 if (out->T_down_dir)
166 out->T_down_dir[idx] = T_down_dir_sca * T_gas_down;
167
168 /* Polarization Q/U (H2O-independent; only when enable_polar) */
169 if (cfg->enable_polar) {
170 float roatmq = 0.0f, roatmu = 0.0f;
171 sixs_interp_polar(ctx, wl, &roatmq, &roatmu);
172 if (out->R_atmQ) out->R_atmQ[idx] = roatmq;
173 if (out->R_atmU) out->R_atmU[idx] = roatmu;
174 }
175 }
176 }
177
178 free(ctx);
179 }
180
181 return 0;
182}

184/* ── Per-pixel trilinear LUT interpolation ───────────────────────────────────
185 *
186 * Returns R_atm, T_down, T_up, s_alb for a single (aod, h2o, wl) point via
187 * trilinear interpolation in the 3-D LUT. Values are clamped to the grid
188 * boundaries; no extrapolation.
189 *
190 * Used by uncertainty.c (AOD perturbation) and main.c (per-pixel correction
191 * with per-pixel AOD/H2O raster maps).
192 * ──────────────────────────────────────────────────────────────────────────── */
193
194/* Binary search: largest i such that arr[i] <= val.
195 * Returns 0 if val < arr[0], n-2 if val >= arr[n-1]. */
196static int find_bracket(const float *arr, int n, float val)
197{
198 if (val <= arr[0]) return 0;
199 if (val >= arr[n - 1]) return n - 2;
200 int lo = 0, hi = n - 1;
201 while (hi - lo > 1) {
202 int mid = (lo + hi) / 2;
203 if (arr[mid] <= val) lo = mid;
204 else hi = mid;
205 }
206 return lo;
207}

209void atcorr_lut_interp_pixel(const LutConfig *cfg, const LutArrays *lut,
210 float aod_val, float h2o_val, float wl_um,
211 float *R_atm_out, float *T_down_out,
212 float *T_up_out, float *s_alb_out)
213{
214 int n_aod = cfg->n_aod;
215 int n_h2o = cfg->n_h2o;
216 int n_wl = cfg->n_wl;
217
218 /* Clamp to grid */
219 if (aod_val < cfg->aod[0]) aod_val = cfg->aod[0];
220 if (aod_val > cfg->aod[n_aod - 1]) aod_val = cfg->aod[n_aod - 1];
221 if (h2o_val < cfg->h2o[0]) h2o_val = cfg->h2o[0];
222 if (h2o_val > cfg->h2o[n_h2o - 1]) h2o_val = cfg->h2o[n_h2o - 1];
223 if (wl_um < cfg->wl[0]) wl_um = cfg->wl[0];
224 if (wl_um > cfg->wl[n_wl - 1]) wl_um = cfg->wl[n_wl - 1];
225
226 int ia = find_bracket(cfg->aod, n_aod, aod_val);
227 int ih = find_bracket(cfg->h2o, n_h2o, h2o_val);
228 int iw = find_bracket(cfg->wl, n_wl, wl_um);
229
230 /* Fractional weights — guard against zero-width intervals */
231 float dA = cfg->aod[ia + 1] - cfg->aod[ia];
232 float dH = cfg->h2o[ih + 1] - cfg->h2o[ih];
233 float dW = cfg->wl [iw + 1] - cfg->wl [iw];
234
235 float wa1 = (dA > 0.0f) ? (aod_val - cfg->aod[ia]) / dA : 0.0f;
236 float wh1 = (dH > 0.0f) ? (h2o_val - cfg->h2o[ih]) / dH : 0.0f;
237 float ww1 = (dW > 0.0f) ? (wl_um - cfg->wl [iw]) / dW : 0.0f;
238 float wa0 = 1.0f - wa1;
239 float wh0 = 1.0f - wh1;
240 float ww0 = 1.0f - ww1;
241
242/* Flat index into the [n_aod × n_h2o × n_wl] C-order array */
243#define IDX(a, h, w) (((size_t)(a) * n_h2o + (h)) * n_wl + (w))
244
245/* Trilinear interpolation of an array at the 8 surrounding grid corners */
246#define TRILIN(arr) ( \
247 wa0 * (wh0 * (ww0 * (arr)[IDX(ia, ih, iw )] \
248 + ww1 * (arr)[IDX(ia, ih, iw+1)]) \
249 + wh1 * (ww0 * (arr)[IDX(ia, ih+1, iw )] \
250 + ww1 * (arr)[IDX(ia, ih+1, iw+1)])) \
251 + wa1 * (wh0 * (ww0 * (arr)[IDX(ia+1, ih, iw )] \
252 + ww1 * (arr)[IDX(ia+1, ih, iw+1)]) \
253 + wh1 * (ww0 * (arr)[IDX(ia+1, ih+1, iw )] \
254 + ww1 * (arr)[IDX(ia+1, ih+1, iw+1)])))
255
256 *R_atm_out = TRILIN(lut->R_atm);
257 *T_down_out = TRILIN(lut->T_down);
258 *T_up_out = TRILIN(lut->T_up);
259 *s_alb_out = TRILIN(lut->s_alb);
260
261#undef IDX
262#undef TRILIN
263}
