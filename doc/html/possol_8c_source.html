8/* Compute day-of-year from calendar date.
9 * jday: day of month (1-31); month: 1-12; ia: year (0 = non-leap). */
10static int day_number(int jday, int month, int ia)
11{
12 int j;
13 if (month <= 2) {
14 j = 31 * (month - 1) + jday;
15 } else if (month > 8) {
16 j = 31 * (month - 1) - (month - 2) / 2 - 2 + jday;
17 } else {
18 j = 31 * (month - 1) - (month - 1) / 2 - 2 + jday;
19 }
20 if (ia != 0 && (ia % 4) == 0) j += 1;
21 return j;
22}

24/* Compute solar zenith (asol) and azimuth (phi0) angles via Fourier series.
25 * j: day of year (1-365); tu: UTC decimal hours;
26 * xlon: longitude deg (E positive); xlat: latitude deg (N positive). */
27static void pos_fft(int j, float tu, float xlon, float xlat,
28 float *asol, float *phi0)
29{
30 const float fac = (float)(M_PI / 180.0);
31 const float pi = (float)M_PI;
32
33 float tsm = tu + xlon / 15.0f;
34 float xla = xlat * fac;
35 float xj = (float)j;
36 float tet = 2.0f * pi * xj / 365.0f;
37
38 /* Time equation (minutes decimal) */
39 float et = 0.000075f
40 + 0.001868f * cosf(tet) - 0.032077f * sinf(tet)
41 - 0.014615f * cosf(2.0f * tet) - 0.040849f * sinf(2.0f * tet);
42 et = et * 12.0f * 60.0f / pi;
43
44 /* True solar time, hour angle */
45 float tsv = tsm + et / 60.0f - 12.0f;
46 float ah = tsv * 15.0f * fac;
47
48 /* Solar declination (radians) */
49 float delta = 0.006918f
50 - 0.399912f * cosf(tet) + 0.070257f * sinf(tet)
51 - 0.006758f * cosf(2.0f * tet) + 0.000907f * sinf(2.0f * tet)
52 - 0.002697f * cosf(3.0f * tet) + 0.001480f * sinf(3.0f * tet);
53
54 /* Elevation and azimuth */
55 float amuzero = sinf(xla) * sinf(delta) + cosf(xla) * cosf(delta) * cosf(ah);
56 float elev = asinf(amuzero);
57 float az = cosf(delta) * sinf(ah) / cosf(elev);
58 if (fabsf(az) > 1.0f) az = (az > 0.0f) ? 1.0f : -1.0f;
59 float caz = (-cosf(xla) * sinf(delta) + sinf(xla) * cosf(delta) * cosf(ah))
60 / cosf(elev);
61 float azim = asinf(az);
62 if (caz <= 0.0f) azim = pi - azim;
63 if (caz > 0.0f && az <= 0.0f) azim = 2.0f * pi + azim;
64 azim += pi;
65 if (azim > 2.0f * pi) azim -= 2.0f * pi;
66
67 elev = elev * 180.0f / pi;
68 *asol = 90.0f - elev;
69 *phi0 = azim / fac;
70}
