11/* Expand ctx->polar.pha[] into Legendre coefficients ctx->polar.betal[].
12 * Always sets *coeff = 0 (no truncation in this 6SV2.1 version).
13 * ipol: 0 = scalar only; nonzero = also compute gammal/deltal (not implemented). */
14void sixs_trunca(SixsCtx *ctx, int ipol, float *coeff)
15{
16 (void)ipol; /* polarization expansion not implemented */
17
18 int nbmu = ctx->quad.nquad; /* e.g. 83 */
19 int nbmu_2 = (nbmu - 3) / 2; /* 40 for nbmu=83 */
20
21 /* Set up Gauss quadrature with special endpoints at -1, 0, +1 */
22 float cgaus_S[NQ_MAX];
23 float pdgs_S[NQ_MAX];
24 sixs_gauss_setup(nbmu, cgaus_S, pdgs_S);
25
26 /* Zero out Legendre coefficients */
27 int ncoeff = nbmu - 2; /* 0 .. nbmu-3 */
28 for (int k = 0; k < ncoeff; k++) ctx->polar.betal[k] = 0.0f;
29
30 /* Compute betal[k] = (2k+1)/2 * sum_j pha[j] * w[j] * P_k(cos_j) */
31 /* Use double precision to accumulate; pl uses offset pointer trick */
32 double pl_arr[NQ_MAX + 2]; /* pl_arr[k+1] = P_k */
33 double *pl = pl_arr + 1; /* so pl[-1], pl[0], pl[k] work */
34
35 for (int j = 0; j < nbmu; j++) {
36 double x = (double)ctx->polar.pha[j] * (double)pdgs_S[j];
37 double rm = (double)cgaus_S[j];
38 pl[-1] = 0.0;
39 pl[0] = 1.0;
40 for (int k = 0; k < ncoeff; k++) {
41 pl[k+1] = ((2*k + 1.0) * rm * pl[k] - k * pl[k-1]) / (k + 1.0);
42 ctx->polar.betal[k] += (float)(x * pl[k]);
43 }
44 }
45
46 /* Normalise and zero out negative coefficients */
47 for (int k = 0; k < ncoeff; k++) {
48 ctx->polar.betal[k] = (float)((2*k + 1.0) * 0.5 * ctx->polar.betal[k]);
49 if (ctx->polar.betal[k] < 0.0f) {
50 for (int j = k; j < ncoeff; j++) ctx->polar.betal[j] = 0.0f;
51 break;
52 }
53 }
54
55 *coeff = 0.0f; /* no truncation applied */
56 (void)nbmu_2;
57}
