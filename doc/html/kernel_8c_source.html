22 #define PSL(l, j) psl[((l)+1)*dim + ((j)+mu)]
23 #define XPL(j) xpl_off[(j)+mu]
24 #define BP(j, k) bp[(j)*dim + ((k)+mu)]

28 if (is == 0) {
29 for (int j = 0; j <= mu; j++) {
30 double c = RM(j);
31 PSL(0, -j) = 1.0; PSL(0, j) = 1.0;
32 PSL(1, j) = c; PSL(1, -j) = -c;
33 double xdb = (3.0 * c * c - 1.0) * 0.5;
34 if (fabs(xdb) < 1e-30) xdb = 0.0;
35 PSL(2, -j) = xdb; PSL(2, j) = xdb;
36 }
37 PSL(1, 0) = RM(0);
38 } else if (is == 1) {
39 for (int j = 0; j <= mu; j++) {
40 double c = RM(j);
41 double x = 1.0 - c * c;
42 PSL(0, j) = 0.0; PSL(0, -j) = 0.0;
43 double sq = sqrt(x * 0.5);
44 PSL(1, -j) = sq; PSL(1, j) = sq;
45 PSL(2, j) = c * PSL(1, j) * rac3;
46 PSL(2, -j) = -PSL(2, j);
47 }
48 PSL(2, 0) = -PSL(2, 0);
49 } else {
50 double a = 1.0;
51 for (int i = 1; i <= is; i++) {
52 double xi = i;
53 a *= sqrt((xi + is) / xi) * 0.5;
54 }
55 for (int j = 0; j <= mu; j++) {
56 double c = RM(j);
57 double xx = 1.0 - c * c;
58 PSL(is-1, j) = 0.0;
59 double xdb = a * pow(xx, is * 0.5);
60 if (fabs(xdb) < 1e-30) xdb = 0.0;
61 PSL(is, -j) = xdb; PSL(is, j) = xdb;
62 }
63 }
64
65 /* Recurrence relation for higher orders */
66 int k = (is > 2) ? is : 2;
67 if (k < ip1) {
68 int ig = (is == 1) ? 1 : -1;
69 for (int l = k; l <= ip1 - 1; l++) {
70 int lp = l + 1, lm = l - 1;
71 double aa = (2.0 * l + 1.0) / sqrt((double)((l + is + 1) * (l - is + 1)));
72 double bb = sqrt((double)((l + is) * (l - is))) / (2.0 * l + 1.0);
73 for (int j = 0; j <= mu; j++) {
74 double c = RM(j);
75 double xdb = aa * (c * PSL(l, j) - bb * PSL(lm, j));
76 if (fabs(xdb) < 1e-30) xdb = 0.0;
77 PSL(lp, j) = xdb;
78 if (j != 0) PSL(lp, -j) = ig * PSL(lp, j);
79 }
80 ig = -ig;
81 }
82 }
83
84 /* xpl = psl[2, :] */
85 for (int j = -mu; j <= mu; j++) XPL(j) = PSL(2, j);
86
87 /* Compute bp[j][k] = sum_l betal[l] * psl[l,j] * psl[l,k] */
88 for (int j = 0; j <= mu; j++) {
89 for (int k2 = -mu; k2 <= mu; k2++) {
90 double sbp = 0.0;
91 if (is <= ip1) {
92 for (int l = is; l <= ip1; l++) {
93 sbp += PSL(l, j) * PSL(l, k2) * ctx->polar.betal[l];
94 }
95 }
96 if (fabs(sbp) < 1e-30) sbp = 0.0;
97 BP(j, k2) = sbp;
98 }
99 }
