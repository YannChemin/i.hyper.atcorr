40float uncertainty_estimate_nedl(const float *rad, int npix)
41{
42 float *buf = malloc((size_t)npix * sizeof(float));
43 if (!buf) return 0.01f;
44
45 int n = 0;
46 for (int i = 0; i < npix; i++)
47 if (isfinite(rad[i]) && rad[i] > 0.0f) buf[n++] = rad[i];
48
49 if (n < 10) { free(buf); return 0.01f; }
50
51 qsort(buf, n, sizeof(float), cmp_float_asc);
52
53 /* Standard deviation of the darkest 5 % */
54 int p5 = (int)(0.05f * n);
55 if (p5 < 5) p5 = 5;
56
57 double sum = 0.0, sum2 = 0.0;
58 for (int i = 0; i < p5; i++) {
59 sum += buf[i];
60 sum2 += (double)buf[i] * buf[i];
61 }
62 double mean = sum / p5;
63 double var = sum2 / p5 - mean * mean;
64
65 free(buf);
66 return (var > 0.0) ? (float)sqrt(var) : 0.01f;
67}

71void uncertainty_compute_band(const float *rad_band, const float *refl_band,
72 int npix,
73 float E0, float d2, float cos_sza,
74 float T_down, float T_up,
75 float s_alb, float R_atm,
76 float nedl, float aod_sigma,
77 const LutConfig *cfg,
78 const LutArrays *lut,
79 float wl_um,
80 float aod_val, float h2o_val,
81 float *sigma_out)
82{
83 /* ── Instrument noise (scalar for this band) ── */
84 if (nedl <= 0.0f && rad_band != NULL)
85 nedl = uncertainty_estimate_nedl(rad_band, npix);
86 if (nedl <= 0.0f) nedl = 0.01f;
87
88 if (E0 <= 0.0f) E0 = 1.0f;
89 if (cos_sza <= 0.0f) cos_sza = 0.01f;
90
91 float sigma_rho_toa = (float)M_PI * nedl * (float)d2 / (E0 * cos_sza);
92 float T_total = T_down * T_up;
93 float sigma_noise = sigma_rho_toa / (T_total > 1e-6f ? T_total : 1e-6f);
94
95 /* ── AOD perturbation (scalar LUT values at ±aod_sigma) ── */
96 float aod_plus = aod_val + aod_sigma;
97 float aod_minus = (aod_val > aod_sigma + 0.001f)
98 ? aod_val - aod_sigma : 0.001f;
99
100 float R1, Td1, Tu1, s1;
101 float R0, Td0, Tu0, s0;
102 atcorr_lut_interp_pixel(cfg, lut, aod_plus, h2o_val, wl_um,
103 &R1, &Td1, &Tu1, &s1);
104 atcorr_lut_interp_pixel(cfg, lut, aod_minus, h2o_val, wl_um,
105 &R0, &Td0, &Tu0, &s0);
106
107 /* ── Per-pixel uncertainty ── */
108#ifdef _OPENMP
109#pragma omp parallel for schedule(static)
110#endif
111 for (int i = 0; i < npix; i++) {
112 float r = refl_band[i];
113 if (!isfinite(r)) { sigma_out[i] = NAN; continue; }
114
115 /* Reconstruct rho_toa from inverted result */
116 float rho_toa = R_atm + T_total * r / (1.0f - s_alb * r + 1e-10f);
117
118 /* Invert at AOD+sigma */
119 float T1 = Td1 * Tu1;
120 float y1 = (rho_toa - R1) / (T1 > 1e-6f ? T1 : 1e-6f);
121 float r1 = y1 / (1.0f + s1 * y1 + 1e-10f);
122
123 /* Invert at AOD-sigma */
124 float T0 = Td0 * Tu0;
125 float y0 = (rho_toa - R0) / (T0 > 1e-6f ? T0 : 1e-6f);
126 float r0 = y0 / (1.0f + s0 * y0 + 1e-10f);
127
128 float sigma_aod = fabsf(r1 - r0) * 0.5f;
129 sigma_out[i] = sqrtf(sigma_noise * sigma_noise + sigma_aod * sigma_aod);
130 }
