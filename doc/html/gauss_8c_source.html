8void sixs_gauss(float x1, float x2, float *x, float *w, int n) {
9 const double eps = 3.0e-14;
10 int m = (n + 1) / 2;
11 double xm = 0.5 * (x2 + x1);
12 double xl = 0.5 * (x2 - x1);
13
14 for (int i = 1; i <= m; i++) {
15 double z = cos(M_PI * (i - 0.25) / (n + 0.5));
16 double z1, p1, p2, p3, pp;
17 do {
18 p1 = 1.0; p2 = 0.0;
19 for (int j = 1; j <= n; j++) {
20 p3 = p2; p2 = p1;
21 p1 = ((2.0 * j - 1.0) * z * p2 - (j - 1.0) * p3) / j;
22 }
23 pp = n * (z * p1 - p2) / (z * z - 1.0);
24 z1 = z;
25 z = z1 - p1 / pp;
26 } while (fabs(z - z1) > eps);
27 if (fabs(z) < eps) z = 0.0;
28 x[i - 1] = (float)(xm - xl * z);
29 x[n - i] = (float)(xm + xl * z);
30 w[i - 1] = (float)(2.0 * xl / ((1.0 - z * z) * pp * pp));
31 w[n - i] = w[i - 1];
32 }
33}
