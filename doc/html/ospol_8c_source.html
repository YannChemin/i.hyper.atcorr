1/* Vector (Stokes I,Q,U) successive orders RT — ported from 6SV2.1 OSPOL.f
2 *
3 * Extends sixs_os() to propagate the full Stokes vector (I,Q,U).
4 * The main physical effect captured over the scalar version is Rayleigh
5 * polarisation: solar intensity generates Q and U via gamma2, and the
6 * multiply-scattered Q/U feed back into I (improving R_atm accuracy by
7 * 1–5 % in blue bands for typical geometries).
8 *
9 * Aerosol simplification: gammal = zetal = 0, alphal = betal (spherical
10 * particles). gr(j,k) = gt(j,k) = 0 for aerosol.
11 *
12 * Memory: three copies of i1/i2 (I,Q,U) → 6 × (nt+1) × (2mu+1) doubles.
13 * This is ~3× the scalar footprint. Temporary arrays are heap-
14 * allocated and freed on exit. */
15

141 double xmus = -(double)RM(0);

143 /* ── Layer grid setup (identical to sixs_os) ── */
144 if (ta <= accu2 && tr > ta) {
145 for (int j = 0; j <= ntp; j++) {
146 h[j] = (double)j * tr / ntp;
147 ch[j] = exp(-h[j] / xmus) * 0.5;
148 ydel[j] = 1.0; xdel[j] = 0.0;
149 altc[j] = (j == 0) ? 300.0 : -log(h[j] / tr) * hr;
150 }
151 } else if (tr <= accu2 && ta > tr) {
152 for (int j = 0; j <= ntp; j++) {
153 h[j] = (double)j * ta / ntp;
154 ch[j] = exp(-h[j] / xmus) * 0.5;
155 ydel[j] = 0.0; xdel[j] = piz;
156 altc[j] = (j == 0) ? 300.0 : -log(h[j] / ta) * ha;
157 }
158 } else {
159 ydel[0] = 1.0; xdel[0] = 0.0; h[0] = 0.0;
160 ch[0] = 0.5; altc[0] = 300.0;
161 double zx = 300.0;
162 for (int it = 0; it <= ntp; it++) {
163 double yy = (it > 0) ? h[it-1] : 0.0;
164 double dd = (it > 0) ? ydel[it-1] : 0.0;
165 int itp = it;
166 sixs_discre(ctx, ta, ha, tr, hr, itp, ntp, yy, dd, 300.0, 0.0, &zx);
167 if (ctx->err.ier) goto cleanup;
168 double xxx = -zx / ha;
169 double ca = (xxx < -18.0) ? 0.0 : ta * exp(xxx);
170 double cr = tr * exp(-zx / hr);
171 h[it] = ca + cr;
172 altc[it] = zx;
173 ch[it] = exp(-h[it] / xmus) * 0.5;
174 cr /= hr; ca /= ha;
175 double ratio = (cr + ca > 0.0) ? cr / (cr + ca) : 0.5;
176 xdel[it] = (1.0 - ratio) * piz;
177 ydel[it] = ratio;
178 }
179 }
180
181 /* ── Plane layer update ── */
182 if (ntp == nt - 1) {
183 double taup = tap + trp;
184 iplane = -1;
185 for (int i = 0; i <= ntp; i++) if (taup >= h[i]) iplane = i;
186 double th = 0.0005;
187 double xt1 = fabs(h[iplane] - taup);
188 double xt2 = fabs(h[iplane+1] - taup);
189 if (xt1 > th && xt2 > th) {
190 for (int i = nt; i >= iplane + 1; i--) {
191 xdel[i] = xdel[i-1]; ydel[i] = ydel[i-1];
192 h[i] = h[i-1]; altc[i] = altc[i-1]; ch[i] = ch[i-1];
193 }
194 } else {
195 nt = ntp;
196 if (xt2 < xt1) iplane++;
197 }
198 h[iplane] = taup;
199 if (tr > accu2 && ta > accu2) {
200 double ca = ta * exp(-palt / ha), cr = tr * exp(-palt / hr);
201 double hi = ca + cr;
202 h[iplane] = hi;
203 ch[iplane] = exp(-hi / xmus) * 0.5;
204 cr /= hr; ca /= ha;
205 double ratio = cr / (cr + ca);
206 xdel[iplane] = (1.0 - ratio) * piz;
207 ydel[iplane] = ratio;
208 altc[iplane] = palt;
209 } else if (tr > accu2) {
210 ydel[iplane] = 1.0; xdel[iplane] = 0.0; altc[iplane] = palt;
211 } else {
212 ydel[iplane] = 0.0; xdel[iplane] = piz; altc[iplane] = palt;
213 }
214 }
215
216 /* ── Initialise output arrays ── */
217 double pi = acos(-1.0);
218 for (int l = 0; l < np; l++)
219 for (int m = -mu; m <= mu; m++) {
220 XL(m, l) = 0.0f;
221 if (xlq) xlq[((m)+mu)*np + l] = 0.0f;
222 if (xlu) xlu[((m)+mu)*np + l] = 0.0f;
223 }
224 for (int ifi = 0; ifi < nfi; ifi++) xlphim[ifi] = 0.0f;
225
226 /* ── Rayleigh depolarisation parameters ── */
227 double aaaa = (double)ctx->del.delta / (2.0 - (double)ctx->del.delta);
228 double ron = (1.0 - aaaa) / (1.0 + 2.0 * aaaa);
229 double beta0 = 1.0;
230 double beta2 = 0.5 * ron;
231 double gamma2 = -ron * sqrt(1.5); /* Rayleigh I→Q coupling */
232 double alpha2 = 3.0 * ron; /* Rayleigh Q→Q strength */
233
234 for (int j = -mu; j <= mu; j++) {
235 I4(j) = 0.0; Q4(j) = 0.0; U4(j) = 0.0;
236 }
237
238 int iborm = ctx->quad.nquad - 3;
239 if (fabs(xmus - 1.0) < 1e-6) iborm = 0;
240
241 /* ══ Fourier decomposition loop ══════════════════════════════════════ */
242 for (int is = 0; is <= iborm; is++) {
243
244 int ig = 1;
245 /* roIavion etc.: path reflectance × xmus at observer plane */
246 double roI = 0.0, roI0 = 0.0, roI1 = 0.0, roI2 = 0.0;
247 double roQ = 0.0, roQ0 = 0.0, roQ1 = 0.0, roQ2 = 0.0;
248 double roU = 0.0, roU0 = 0.0, roU1 = 0.0, roU2 = 0.0;
249
250 for (int j = -mu; j <= mu; j++) {
251 I3(j) = 0.0; Q3(j) = 0.0; U3(j) = 0.0;
252 }
253
254 double beta0_is = (is > 0) ? 0.0 : beta0;
255
256 /* Polarised kernel for this Fourier order */
257 sixs_kernelpol(ctx, is, mu, rm_off,
258 xpl, xrl, xtl, psl, rsl, tsl,
259 bp, arr, art, att);
260
261 /* ── Primary scattering source functions ── */
262 double spl = XPL(0); /* xpl[0] = psl[2][0] */
263 double srl = XRL(0); /* xrl[0] = rsl[2][0] */
264 double stl = XTL(0); /* xtl[0] = tsl[2][0] */
265 (void)srl; (void)stl; /* Only spl enters the primary term */
266
267 for (int j = -mu; j <= mu; j++) {
268 double sa1, sa2, sb1, sb2, sc1, sc2;
269 if (is <= 2) {
270 sa1 = beta0_is + beta2 * XPL(j) * spl; /* Rayleigh I from I */
271 sa2 = BP(0, j); /* aerosol I from I */
272 sb1 = gamma2 * XRL(j) * spl; /* Rayleigh Q from I */
273 sb2 = 0.0; /* gr(0,j) = 0 */
274 sc1 = gamma2 * XTL(j) * spl; /* Rayleigh U from I */
275 sc2 = 0.0; /* gt(0,j) = 0 */
276 } else {
277 sa1 = 0.0; sa2 = BP(0, j);
278 sb1 = 0.0; sb2 = 0.0;
279 sc1 = 0.0; sc2 = 0.0;
280 }
281 for (int k = 0; k <= nt; k++) {
282 double c = ch[k];
283 double a = ydel[k];
284 double b = xdel[k];
285 I2(k, j) = c * (sa2 * b + sa1 * a);
286 Q2(k, j) = c * (sb2 * b + sb1 * a);
287 U2(k, j) = -c * (sc2 * b + sc1 * a); /* minus sign per OSPOL.f */
288 }
289 }
290
291 /* ── Vertical integration, primary upward (k = 1..mu) ── */
292 for (int k = 1; k <= mu; k++) {
293 I1(nt,k) = Q1(nt,k) = U1(nt,k) = 0.0;
294 double zi1 = 0.0, zq1 = 0.0, zu1 = 0.0;
295 double yy = (double)RM(k);
296 for (int i = nt - 1; i >= 0; i--) {
297 int jj = i + 1;
298 double f = h[jj] - h[i];
299 double c = exp(-f / yy), d = 1.0 - c, xx = h[i] - h[jj]*c;
300 #define VERT_INT_UP(Z, v2, v1, k_idx) \
301 { double _a = (v2(jj,k_idx) - v2(i,k_idx)) / f; \
302 double _b = v2(i,k_idx) - _a * h[i]; \
303 Z = c*Z + (d*(_b + _a*yy) + _a*xx) * 0.5; \
304 v1(i,k_idx) = Z; }
305 VERT_INT_UP(zi1, I2, I1, k)
306 VERT_INT_UP(zq1, Q2, Q1, k)
307 VERT_INT_UP(zu1, U2, U1, k)
308 #undef VERT_INT_UP
309 }
310 }
311
312 /* ── Vertical integration, primary downward (k = -mu..-1) ── */
313 for (int k = -mu; k <= -1; k++) {
314 I1(0,k) = Q1(0,k) = U1(0,k) = 0.0;
315 double zi1 = 0.0, zq1 = 0.0, zu1 = 0.0;
316 double yy = (double)RM(k);
317 for (int i = 1; i <= nt; i++) {
318 int jj = i - 1;
319 double f = h[i] - h[jj];
320 double c = exp(f / yy), d = 1.0 - c, xx = h[i] - h[jj]*c;
321 #define VERT_INT_DN(Z, v2, v1, k_idx) \
322 { double _a = (v2(i,k_idx) - v2(jj,k_idx)) / f; \
323 double _b = v2(i,k_idx) - _a * h[i]; \
324 Z = c*Z + (d*(_b + _a*yy) + _a*xx) * 0.5; \
325 v1(i,k_idx) = Z; }
326 VERT_INT_DN(zi1, I2, I1, k)
327 VERT_INT_DN(zq1, Q2, Q1, k)
328 VERT_INT_DN(zu1, U2, U1, k)
329 #undef VERT_INT_DN
330 }
331 }
332
333 /* ── Collect primary scattering ── */
334 for (int k = -mu; k <= mu; k++) {
335 if (k == 0) continue;
336 int idx = (k < 0) ? nt : 0;
337 INM1(k) = INM2(k) = I3(k) = I1(idx, k);
338 QNM1(k) = QNM2(k) = Q3(k) = Q1(idx, k);
339 UNM1(k) = UNM2(k) = U3(k) = U1(idx, k);
340 }
341 roI2 = roI = I1(iplane, mu);
342 roQ2 = roQ = Q1(iplane, mu);
343 roU2 = roU = U1(iplane, mu);
344
345 /* ════ Successive Orders of Scattering ════════════════════════════ */
346 for (;;) {
347 ig++;
348
349 /* ── Build source functions for next order ── */
350 if (is <= 2) {
351 /* Mixed Rayleigh + aerosol */
352 for (int k = 1; k <= mu; k++) {
353 double xpk = XPL( k), ypk = XPL(-k);
354 double xrk = XRL( k), yrk = XRL(-k);
355 double xtk = XTL( k), ytk = XTL(-k);
356 for (int i = 0; i <= nt; i++) {
357 double ii1 = 0.0, ii2 = 0.0;
358 double qq1 = 0.0, qq2 = 0.0;
359 double uu1 = 0.0, uu2 = 0.0;
360 double x = xdel[i], y = ydel[i];
361 for (int j = 1; j <= mu; j++) {
362 double z = (double)GB(j);
363 double xpj = XPL( j);
364 double xrj = XRL( j), yrj = XRL(-j);
365 double xtj = XTL( j), ytj = XTL(-j);
366 double xi1 = I1(i, j), xi2 = I1(i, -j);
367 double xq1 = Q1(i, j), xq2 = Q1(i, -j);
368 double xu1 = U1(i, j), xu2 = U1(i, -j);
369
370 /* Combined (aerosol×x + Rayleigh×y) kernels */
371 double bpjk = BP(j, k)*x + y*(beta0_is + beta2*xpj*xpk);
372 double bpjmk = BP(j,-k)*x + y*(beta0_is + beta2*xpj*ypk);
373 /* I→Q and Q→I Rayleigh coupling (gr=0 for aerosol) */
374 double grjk = y * gamma2 * xpj * xrk;
375 double grjmk = y * gamma2 * xpj * yrk;
376 double grkj = y * gamma2 * xpk * xrj;
377 double grkmj = y * gamma2 * xpk * yrj;
378 /* I→U and U→I Rayleigh coupling (gt=0 for aerosol) */
379 double gtjk = y * gamma2 * xpj * xtk;
380 double gtjmk = y * gamma2 * xpj * ytk;
381 double gtkj = y * gamma2 * xpk * xtj;
382 double gtkmj = y * gamma2 * xpk * ytj;
383 /* Q↔Q and U↔U (aerosol arr/att + Rayleigh alpha2) */
384 double arrjk = ARR(j, k)*x + y * alpha2 * xrj * xrk;
385 double arrjmk = ARR(j,-k)*x + y * alpha2 * xrj * yrk;
386 double artjk = ART(j, k)*x + y * alpha2 * xtj * xrk;
387 double artjmk = ART(j,-k)*x + y * alpha2 * xtj * yrk;
388 double artkj = ART(k, j)*x + y * alpha2 * xtk * xrj;
389 double artkmj = ART(k,-j)*x + y * alpha2 * xtk * yrj;
390 double attjk = ATT(j, k)*x + y * alpha2 * xtj * xtk;
391 double attjmk = ATT(j,-k)*x + y * alpha2 * xtj * ytk;
392
393 /* I source */
394 double xdb;
395 xdb = xi1*bpjk + xi2*bpjmk + xq1*grkj + xq2*grkmj
396 - xu1*gtkj - xu2*gtkmj;
397 ii2 += xdb * z;
398 xdb = xi1*bpjmk + xi2*bpjk + xq1*grkmj + xq2*grkj
399 + xu1*gtkmj + xu2*gtkj;
400 ii1 += xdb * z;
401
402 /* Q source */
403 xdb = xi1*grjk + xi2*grjmk + xq1*arrjk + xq2*arrjmk
404 - xu1*artjk + xu2*artjmk;
405 qq2 += xdb * z;
406 xdb = xi1*grjmk + xi2*grjk + xq1*arrjmk + xq2*arrjk
407 - xu1*artjmk + xu2*artjk;
408 qq1 += xdb * z;
409
410 /* U source (minus sign per OSPOL.f) */
411 xdb = xi1*gtjk - xi2*gtjmk + xq1*artkj + xq2*artkmj
412 - xu1*attjk - xu2*attjmk;
413 uu2 -= xdb * z;
414 xdb = xi1*gtjmk - xi2*gtjk - xq1*artkmj - xq2*artkj
415 - xu1*attjmk - xu2*attjk;
416 uu1 -= xdb * z;
417 }
418 if (fabs(ii2) < 1e-30) ii2 = 0.0;
419 if (fabs(ii1) < 1e-30) ii1 = 0.0;
420 if (fabs(qq2) < 1e-30) qq2 = 0.0;
421 if (fabs(qq1) < 1e-30) qq1 = 0.0;
422 if (fabs(uu2) < 1e-30) uu2 = 0.0;
423 if (fabs(uu1) < 1e-30) uu1 = 0.0;
424 I2(i, k) = ii2; I2(i, -k) = ii1;
425 Q2(i, k) = qq2; Q2(i, -k) = qq1;
426 U2(i, k) = uu2; U2(i, -k) = uu1;
427 }
428 }
429 } else {
430 /* Aerosol-only: no Rayleigh coupling, arr/att/art only */
431 for (int k = 1; k <= mu; k++) {
432 for (int i = 0; i <= nt; i++) {
433 double ii1 = 0.0, ii2 = 0.0;
434 double qq1 = 0.0, qq2 = 0.0;
435 double uu1 = 0.0, uu2 = 0.0;
436 double x = xdel[i];
437 for (int j = 1; j <= mu; j++) {
438 double z = (double)GB(j);
439 double xi1 = I1(i, j), xi2 = I1(i, -j);
440 double xq1 = Q1(i, j), xq2 = Q1(i, -j);
441 double xu1 = U1(i, j), xu2 = U1(i, -j);
442 double bpjk = BP (j, k) * x;
443 double bpjmk = BP (j, -k) * x;
444 double arrjk = ARR(j, k) * x;
445 double arrjmk = ARR(j, -k) * x;
446 double artjk = ART(j, k) * x;
447 double artjmk = ART(j, -k) * x;
448 double artkj = ART(k, j) * x;
449 double artkmj = ART(k, -j) * x;
450 double attjk = ATT(j, k) * x;
451 double attjmk = ATT(j, -k) * x;
452
453 double xdb;
454 xdb = xi1*bpjk + xi2*bpjmk;
455 ii2 += xdb * z;
456 xdb = xi1*bpjmk + xi2*bpjk;
457 ii1 += xdb * z;
458 xdb = xq1*arrjk + xq2*arrjmk - xu1*artjk + xu2*artjmk;
459 qq2 += xdb * z;
460 xdb = xq1*arrjmk + xq2*arrjk - xu1*artjmk + xu2*artjk;
461 qq1 += xdb * z;
462 xdb = xq1*artkj + xq2*artkmj - xu1*attjk - xu2*attjmk;
463 uu2 -= xdb * z;
464 xdb = -xq1*artkmj - xq2*artkj - xu1*attjmk - xu2*attjk;
465 uu1 -= xdb * z;
466 }
467 if (fabs(ii2) < 1e-30) ii2 = 0.0;
468 if (fabs(ii1) < 1e-30) ii1 = 0.0;
469 if (fabs(qq2) < 1e-30) qq2 = 0.0;
470 if (fabs(qq1) < 1e-30) qq1 = 0.0;
471 if (fabs(uu2) < 1e-30) uu2 = 0.0;
472 if (fabs(uu1) < 1e-30) uu1 = 0.0;
473 I2(i, k) = ii2; I2(i, -k) = ii1;
474 Q2(i, k) = qq2; Q2(i, -k) = qq1;
475 U2(i, k) = uu2; U2(i, -k) = uu1;
476 }
477 }
478 }
479
480 /* ── Vertical integration upward ── */
481 for (int k = 1; k <= mu; k++) {
482 I1(nt,k) = Q1(nt,k) = U1(nt,k) = 0.0;
483 double zi1 = 0.0, zq1 = 0.0, zu1 = 0.0;
484 double yy = (double)RM(k);
485 for (int i = nt - 1; i >= 0; i--) {
486 int jj = i + 1;
487 double f = h[jj] - h[i];
488 double c = exp(-f / yy), d = 1.0 - c, xx = h[i] - h[jj]*c;
489 #define VI_UP(Z, v2, v1, k_idx) \
490 { double _a = (v2(jj,k_idx) - v2(i,k_idx)) / f; \
491 double _b = v2(i,k_idx) - _a * h[i]; \
492 Z = c*Z + (d*(_b + _a*yy) + _a*xx) * 0.5; \
493 if (fabs(Z) < 1e-20) Z = 0.0; \
494 v1(i,k_idx) = Z; }
495 VI_UP(zi1, I2, I1, k)
496 VI_UP(zq1, Q2, Q1, k)
497 VI_UP(zu1, U2, U1, k)
498 #undef VI_UP
499 }
500 }
501
502 /* ── Vertical integration downward ── */
503 for (int k = -mu; k <= -1; k++) {
504 I1(0,k) = Q1(0,k) = U1(0,k) = 0.0;
505 double zi1 = 0.0, zq1 = 0.0, zu1 = 0.0;
506 double yy = (double)RM(k);
507 for (int i = 1; i <= nt; i++) {
508 int jj = i - 1;
509 double f = h[i] - h[jj];
510 double c = exp(f / yy), d = 1.0 - c, xx = h[i] - h[jj]*c;
511 #define VI_DN(Z, v2, v1, k_idx) \
512 { double _a = (v2(i,k_idx) - v2(jj,k_idx)) / f; \
513 double _b = v2(i,k_idx) - _a * h[i]; \
514 Z = c*Z + (d*(_b + _a*yy) + _a*xx) * 0.5; \
515 if (fabs(Z) < 1e-20) Z = 0.0; \
516 v1(i,k_idx) = Z; }
517 VI_DN(zi1, I2, I1, k)
518 VI_DN(zq1, Q2, Q1, k)
519 VI_DN(zu1, U2, U1, k)
520 #undef VI_DN
521 }
522 }
523
524 /* ── Collect scattering order n ── */
525 for (int k = -mu; k <= mu; k++) {
526 if (k == 0) continue;
527 int idx = (k < 0) ? nt : 0;
528 IN(k) = I1(idx, k);
529 QN(k) = Q1(idx, k);
530 UN(k) = U1(idx, k);
531 }
532 roI0 = I1(iplane, mu);
533 roQ0 = Q1(iplane, mu);
534 roU0 = U1(iplane, mu);
535
536 /* ── Convergence test (geometric series) ── */
537 if (ig > 2) {
538 double z = 0.0;
539 /* Check roI */
540 if (roI2 >= accu && roI1 >= accu && roI >= accu) {
541 double r = roI0 / roI1;
542 double y = fabs(((r - roI1/roI2) / ((1.0-r)*(1.0-r)))
543 * (roI0 / roI));
544 z = fmax(y, z);
545 }
546 /* Check roQ, roU similarly */
547 #define CHECK_CONV(a2, a1, a0, asum) \
548 if (fabs(a2) >= accu && fabs(a1) >= accu && fabs(asum) >= accu) { \
549 double _r = (a0) / (a1); \
550 double _y = fabs(((_r - (a1)/(a2)) / ((1.0-_r)*(1.0-_r))) \
551 * ((a0) / (asum))); \
552 z = fmax(_y, z); \
553 }
554 CHECK_CONV(roQ2, roQ1, roQ0, roQ)
555 CHECK_CONV(roU2, roU1, roU0, roU)
556 #undef CHECK_CONV
557 /* Check all quadrature points */
558 for (int l = -mu; l <= mu; l++) {
559 if (l == 0) continue;
560 #define CK3(n2,n1,n0,s3) \
561 if (fabs(n2(l)) >= accu && fabs(n1(l)) >= accu && fabs(s3(l)) >= accu) { \
562 double _r = (n0(l)) / (n1(l)); \
563 double _y = fabs(((_r - (n1(l))/(n2(l))) / ((1.0-_r)*(1.0-_r))) \
564 * ((n0(l)) / (s3(l)))); \
565 z = fmax(_y, z); \
566 }
567 CK3(INM2, INM1, IN, I3)
568 CK3(QNM2, QNM1, QN, Q3)
569 CK3(UNM2, UNM1, UN, U3)
570 #undef CK3
571 }
572
573 if (z < 0.01) {
574 /* Geometric series acceleration */
575 for (int l = -mu; l <= mu; l++) {
576 if (l == 0) continue;
577 #define ACCEL(n1,n0,s3) \
578 { double _d1 = n1(l), _g1 = n0(l); \
579 if (fabs(_d1) > accu && fabs(_g1 - _d1) > accu) \
580 s3(l) += _g1 / (1.0 - _g1 / _d1); }
581 ACCEL(INM1, IN, I3)
582 ACCEL(QNM1, QN, Q3)
583 ACCEL(UNM1, UN, U3)
584 #undef ACCEL
585 }
586 #define ACCEL_RO(ro1, ro0, ro) \
587 { if (fabs(ro1) >= accu && fabs(ro0 - ro1) >= accu) \
588 ro += ro0 / (1.0 - ro0 / ro1); }
589 ACCEL_RO(roI1, roI0, roI)
590 ACCEL_RO(roQ1, roQ0, roQ)
591 ACCEL_RO(roU1, roU0, roU)
592 #undef ACCEL_RO
593 goto done_sos;
594 }
595 /* Shift n-2 ← n-1 */
596 for (int k = -mu; k <= mu; k++) {
597 INM2(k) = INM1(k); QNM2(k) = QNM1(k); UNM2(k) = UNM1(k);
598 }
599 roI2 = roI1; roQ2 = roQ1; roU2 = roU1;
600 }
601 /* Shift n-1 ← n */
602 for (int k = -mu; k <= mu; k++) {
603 INM1(k) = IN(k); QNM1(k) = QN(k); UNM1(k) = UN(k);
604 }
605 roI1 = roI0; roQ1 = roQ0; roU1 = roU0;
606
607 /* Accumulate order n */
608 for (int l = -mu; l <= mu; l++) {
609 I3(l) += IN(l); Q3(l) += QN(l); U3(l) += UN(l);
610 }
611 roI += roI0; roQ += roQ0; roU += roU0;
612
613 /* Relative convergence: stop if last order < 1e-5 of sum */
614 {
615 double z = 0.0;
616 for (int l = -mu; l <= mu; l++) {
617 if (l == 0) continue;
618 if (fabs(I3(l)) >= accu) z = fmax(z, fabs(IN(l)/I3(l)));
619 if (fabs(Q3(l)) >= accu) z = fmax(z, fabs(QN(l)/Q3(l)));
620 if (fabs(U3(l)) >= accu) z = fmax(z, fabs(UN(l)/U3(l)));
621 }
622 if (z < 1e-5) goto done_sos;
623 }
624 if (ig >= ctx->multi.igmax) goto done_sos;
625 } /* end SOS loop */
626 done_sos:;
627
628 /* ── Accumulate Fourier component ── */
629 double delta0s = (is == 0) ? 1.0 : 2.0;
630 for (int l = -mu; l <= mu; l++) {
631 I4(l) += delta0s * I3(l);
632 Q4(l) += delta0s * Q3(l);
633 U4(l) += delta0s * U3(l);
634 }
635
636 /* xl: radiance field at each azimuth plane */
637 for (int l_idx = 0; l_idx < np; l_idx++) {
638 double phi_l = (double)rp[l_idx];
639 for (int m = -mum1; m <= mum1; m++) {
640 double cosfac = (m > 0) ? cos(is * (phi_l + pi)) : cos(is * phi_l);
641 double sinfac = (m > 0) ? sin(is * (phi_l + pi)) : sin(is * phi_l);
642 XL(m, l_idx) += (float)(delta0s * I3(m) * cosfac);
643 if (xlq) xlq[((m)+mu)*np + l_idx] += (float)(delta0s * Q3(m) * cosfac);
644 if (xlu) xlu[((m)+mu)*np + l_idx] += (float)(delta0s * U3(m) * sinfac);
645 }
646 }
647
648 /* xl(mu, 1): upward at specific view direction */
649 XL(mu, 0) += (float)(delta0s * I3(mu) * cos(is * (phirad + pi)));
650 if (xlq) xlq[(mu+mu)*np] += (float)(delta0s * Q3(mu) * cos(is * (phirad + pi)));
651 if (xlu) xlu[(mu+mu)*np] += (float)(delta0s * U3(mu) * sin(is * (phirad + pi)));
652
653 /* xl(-mu, 1): atmospheric path reflectance (×xmus) */
654 XL(-mu, 0) += (float)(delta0s * roI * cos(is * (phirad + pi)));
655 if (xlq) xlq[(0)*np] += (float)(delta0s * roQ * cos(is * (phirad + pi)));
656 if (xlu) xlu[(0)*np] += (float)(delta0s * roU * sin(is * (phirad + pi)));
657
658 /* xl(0, 1): downward flux (is=0 only) */
659 if (is == 0) {
660 for (int k = 1; k <= mum1; k++)
661 XL(0, 0) += (float)((double)RM(k) * (double)GB(k) * I3(-k));
662 }
663
664 /* xlphim: path reflectance vs azimuth */
665 for (int ifi = 0; ifi < nfi; ifi++) {
666 double phimul = (double)ifi * pi / (double)(nfi - 1);
667 xlphim[ifi] += (float)(delta0s * roI * cos(is * (phimul + pi)));
668 }
669
670 /* Fourier convergence */
671 {
672 double z = 0.0;
673 for (int l = -mu; l <= mu; l++) {
674 if (fabs(I4(l)) >= accu) z = fmax(z, fabs(I3(l)/I4(l)));
675 if (fabs(Q4(l)) >= accu) z = fmax(z, fabs(Q3(l)/Q4(l)));
676 if (fabs(U4(l)) >= accu) z = fmax(z, fabs(U3(l)/U4(l)));
677 }
678 if (z <= 0.0005) break;
679 }
680 } /* end Fourier loop */
681
682 nt = snt;
