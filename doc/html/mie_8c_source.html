1/* Mie scattering for custom aerosols — ported from 6SV2.1 MIE.f + EXSCPHASE.f.
2 *
3 * Implements Bohren-Huffman Mie theory for spherical particles over a
4 * log-normal size distribution, computing extinction, single-scatter albedo,
5 * asymmetry parameter, and phase function at the 20 reference wavelengths
6 * used by the 6SV2.1 RT solver.
7 *
8 * Usage:
9 * sixs_mie_init(ctx, r_mode, sigma_g, m_r, m_i) — fills ctx->aer
10 * sixs_aerosol_init then builds ctx->polar from ctx->aer as usual.
11 */

29/* ──────────────────────────────────────────────────────────────────────────
30 * EXSCPHASE — Mie extinction/scattering efficiencies and phase function for
31 * a single sphere at size parameter X = 2π r / λ with complex refractive
32 * index m = nr - i*ni.
33 *
34 * Outputs:
35 * Qext, Qsca : efficiency factors
36 * p11[nbmu] : scattering intensity at nbmu Gauss points
37 *
38 * cgaus_S[nbmu], pdgs_S[nbmu] : Gauss quadrature cosines and weights.
39 * ipol must be 0 (scalar only — polarisation terms Q/U not computed here).
40 * ────────────────────────────────────────────────────────────────────────── */
41static void exscphase(double X, double nr, double ni,
42 double *Qext_out, double *Qsca_out,
43 double *p11, int nbmu,
44 const float *cgaus_S, const float *pdgs_S)
45{
46 (void)pdgs_S; /* used only when asymmetry is needed externally */
47
48 /* Maximum series order — Wiscombe criterion on |m|*X */
49 double Y = X * sqrt(nr*nr + ni*ni);
50 int N = (int)(0.5 * (-1.0 + sqrt(1.0 + 4.0*Y*Y))) + 1;
51 if (N < 2) N = 2;
52
53 /* Upper bound for series truncation (Deirmendjian) */
54 double Up = 2.0*Y / (2.0*N + 1.0);
55 int mu1 = (int)(N + 30.0*(0.10 + 0.35*Up*(2 - Up*Up)/2.0/(1 - Up + 1e-30)));
56 int Np = (int)(Y - 0.5 + sqrt(30.0*0.35*Y));
57 int mu2 = mu1;
58 if (Np > N) {
59 Up = 2.0*Y / (2.0*Np + 1.0);
60 mu2 = (int)(Np + 30.0*(0.10 + 0.35*Up*(2 - Up*Up)/2.0/(1 - Up + 1e-30)));
61 }
62 int mu = (mu1 < mu2) ? mu1 : mu2;
63 if (mu < 2) mu = 2;
64 if (mu > 50000) mu = 50000; /* hard safety cap */
65
66 /* Dynamic allocation for Mie recursion arrays (0..mu) */
67 double *xj = (double*)calloc(mu + 2, sizeof(double));
68 double *xy = (double*)calloc(mu + 2, sizeof(double)); /* shifted: index 0=xy[-1], 1=xy[0], ... */
69 double *Rn = (double*)calloc(mu + 2, sizeof(double));
70 double *RDnX = (double*)calloc(mu + 2, sizeof(double));
71 double *RDnY = (double*)calloc(mu + 2, sizeof(double));
72 double *IDnY = (double*)calloc(mu + 2, sizeof(double));
73 double *RGnX = (double*)calloc(mu + 2, sizeof(double));
74 double *IGnX = (double*)calloc(mu + 2, sizeof(double));
75 double *RAn = (double*)calloc(mu + 2, sizeof(double));
76 double *IAn = (double*)calloc(mu + 2, sizeof(double));
77 double *RBn = (double*)calloc(mu + 2, sizeof(double));
78 double *IBn = (double*)calloc(mu + 2, sizeof(double));
79 double *TAUn = (double*)calloc(mu + 2, sizeof(double));
80 double *PIn = (double*)calloc(mu + 2, sizeof(double));
81
82 if (!xj || !xy || !Rn || !RDnX || !RDnY || !IDnY ||
83 !RGnX || !IGnX || !RAn || !IAn || !RBn || !IBn || !TAUn || !PIn) {
84 free(xj); free(xy); free(Rn); free(RDnX); free(RDnY); free(IDnY);
85 free(RGnX); free(IGnX); free(RAn); free(IAn); free(RBn); free(IBn);
86 free(TAUn); free(PIn);
87 *Qext_out = 2.0; *Qsca_out = 2.0;
88 for (int j = 0; j < nbmu; j++) p11[j] = 1.0;
89 return;
90 }
91
92 /* Inverse relative refractive index (for Dn recursion) */
93 double denom = nr*nr + ni*ni;
94 double Ren = nr / denom;
95 double Imn = ni / denom;
96
97 /* Downward recursion for Dn(X) and Dn(Y) */
98 RDnY[mu] = 0.0; IDnY[mu] = 0.0; RDnX[mu] = 0.0;
99 for (int k = mu; k >= 1; k--) {
100 RDnX[k-1] = (double)k/X - 1.0 / (RDnX[k] + (double)k/X);
101 double XnumR = RDnY[k] + Ren*(double)k/X;
102 double XnumI = IDnY[k] + Imn*(double)k/X;
103 double Xden = XnumR*XnumR + XnumI*XnumI;
104 RDnY[k-1] = (double)k*Ren/X - XnumR/Xden;
105 IDnY[k-1] = (double)k*Imn/X + XnumI/Xden;
106 }
107
108 /* Downward pass for Rn ratios + xj normalization seed */
109 Rn[mu] = 0.0;
110 int mub = mu;
111 for (int k = mu; k >= 2; k--) {
112 Rn[k-1] = X / (2.0*k + 1.0 - X*Rn[k]);
113 if (Rn[k-1] > 1.0) {
114 mub = k - 1;
115 xj[mub+1] = Rn[mub];
116 xj[mub] = 1.0;
117 goto seed_found;
118 }
119 }
120 /* Fell through: start from mu */
121 xj[mub+1] = 0.0;
122 xj[mub] = 1.0;
123
124seed_found:;
125 /* Upward recursion for xj (spherical Bessel j) */
126 for (int k = mub; k >= 1; k--)
127 xj[k-1] = (2.0*k + 1.0)*xj[k]/X - xj[k+1];
128
129 /* Normalize xj by the Wronskian coxj */
130 double coxj = (xj[0] - X*xj[1])*cos(X) + X*xj[0]*sin(X);
131
132 /* Initialise Bessel y (-1 and 0 indices stored offset by 1) */
133 /* xy_arr[k] = y(k-1): xy_arr[0]=y(-1)=sin(X)/X, xy_arr[1]=y(0)=-cos(X)/X */
134 xy[0] = sin(X)/X; /* y_{-1} */
135 xy[1] = -cos(X)/X; /* y_0 */
136
137 /* Upward recursion for Gn, An, Bn, Qsca, Qext */
138 RGnX[0] = 0.0;
139 IGnX[0] = -1.0;
140 double Qsca = 0.0, Qext = 0.0;
141
142 int mu_final = mu;
143 for (int k = 1; k <= mu; k++) {
144 /* Normalise xj */
145 double xjk;
146 if (k <= mub) {
147 xjk = xj[k] / coxj;
148 } else {
149 xjk = Rn[k-1] * xj[k-1] / coxj;
150 }
151 xj[k] = xjk;
152
153 /* Bessel y upward */
154 xy[k+1] = (2.0*k - 1.0)*xy[k]/X - xy[k-1]; /* y_k */
155
156 double yk = xy[k+1]; /* y_{k} */
157 double h2 = xjk*xjk + yk*yk;
158 double xJonH = (h2 > 0.0) ? xjk/h2 : 0.0;
159
160 /* Gn */
161 double den_G = (RGnX[k-1] - (double)k/X)*(RGnX[k-1] - (double)k/X)
162 + IGnX[k-1]*IGnX[k-1];
163 if (den_G > 0.0) {
164 RGnX[k] = ((double)k/X - RGnX[k-1])/den_G - (double)k/X;
165 IGnX[k] = IGnX[k-1]/den_G;
166 }
167
168 /* An */
169 double n1a = RDnY[k] - nr*RDnX[k];
170 double n2a = IDnY[k] + ni*RDnX[k];
171 double d1a = RDnY[k] - nr*RGnX[k] - ni*IGnX[k];
172 double d2a = IDnY[k] + ni*RGnX[k] - nr*IGnX[k];
173 double da = d1a*d1a + d2a*d2a;
174 double RAb = (da > 0.0) ? (n1a*d1a + n2a*d2a)/da : 0.0;
175 double IAb = (da > 0.0) ? (-n1a*d2a + n2a*d1a)/da : 0.0;
176 RAn[k] = xJonH*(xjk*RAb - yk*IAb);
177 IAn[k] = xJonH*(yk*RAb + xjk*IAb);
178
179 /* Bn */
180 double n1b = nr*RDnY[k] + ni*IDnY[k] - RDnX[k];
181 double n2b = nr*IDnY[k] - ni*RDnY[k];
182 double d1b = nr*RDnY[k] + ni*IDnY[k] - RGnX[k];
183 double d2b = nr*IDnY[k] - ni*RDnY[k] - IGnX[k];
184 double db = d1b*d1b + d2b*d2b;
185 double RBb = (db > 0.0) ? (n1b*d1b + n2b*d2b)/db : 0.0;
186 double IBb = (db > 0.0) ? (-n1b*d2b + n2b*d1b)/db : 0.0;
187 RBn[k] = xJonH*(xjk*RBb - yk*IBb);
188 IBn[k] = xJonH*(yk*RBb + xjk*IBb);
189
190 /* Convergence test */
191 double test = (RAn[k]*RAn[k] + IAn[k]*IAn[k]
192 + RBn[k]*RBn[k] + IBn[k]*IBn[k]) / (double)k;
193 if (test < 1.0e-14) {
194 mu_final = k;
195 break;
196 }
197
198 double xpond = 2.0/X/X * (2.0*k + 1.0);
199 Qsca += xpond * (RAn[k]*RAn[k] + IAn[k]*IAn[k]
200 + RBn[k]*RBn[k] + IBn[k]*IBn[k]);
201 Qext += xpond * (RAn[k] + RBn[k]);
202 }
203
204 /* Phase function at nbmu Gauss angles */
205 for (int j = 0; j < nbmu; j++) {
206 double xmud = (double)cgaus_S[j];
207 double RS1 = 0.0, RS2 = 0.0, IS1 = 0.0, IS2 = 0.0;
208 PIn[0] = 0.0;
209 PIn[1] = 1.0;
210 TAUn[1] = xmud;
211 for (int k = 1; k <= mu_final; k++) {
212 double co_n = (2.0*k + 1.0) / ((double)k * (k + 1.0));
213 RS1 += co_n * (RAn[k]*PIn[k] + RBn[k]*TAUn[k]);
214 RS2 += co_n * (RAn[k]*TAUn[k] + RBn[k]*PIn[k]);
215 IS1 += co_n * (IAn[k]*PIn[k] + IBn[k]*TAUn[k]);
216 IS2 += co_n * (IAn[k]*TAUn[k] + IBn[k]*PIn[k]);
217 PIn[k+1] = ((2.0*k+1)*xmud*PIn[k] - (k+1.0)*PIn[k-1]) / (double)k;
218 TAUn[k+1] = (k+1.0)*xmud*PIn[k+1] - (k+2.0)*PIn[k];
219 }
220 p11[j] = (RS1*RS1 + IS1*IS1 + RS2*RS2 + IS2*IS2) / (X*X*2.0);
221 }
222
223 *Qext_out = Qext;
224 *Qsca_out = Qsca;
225
226 free(xj); free(xy); free(Rn); free(RDnX); free(RDnY); free(IDnY);
227 free(RGnX); free(IGnX); free(RAn); free(IAn); free(RBn); free(IBn);
228 free(TAUn); free(PIn);
229}

231/* ──────────────────────────────────────────────────────────────────────────
232 * sixs_mie_init — fill ctx->aer for a custom log-normal aerosol.
233 *
234 * Parameters (log-normal size distribution):
235 * r_mode : mode radius (µm)
236 * sigma_g : geometric standard deviation (dimensionless, > 1)
237 * m_r_550 : real part of refractive index at 550 nm
238 * m_i_550 : imaginary part (positive, absorption)
239 *
240 * Uses a constant refractive index across all 20 reference wavelengths.
241 * For a spectrally varying index the caller should call aerosol_init with
242 * a custom mixture instead.
243 * ────────────────────────────────────────────────────────────────────────── */
244void sixs_mie_init(SixsCtx *ctx,
245 double r_mode, double sigma_g,
246 double m_r_550, double m_i_550)
247{
248 const double pi = M_PI;
249 const double rlogpas = 0.011; /* log10 step = 6SV default */
250 const double rmin = 0.001; /* minimum radius (µm) */
251 const double rmax = 25.0; /* maximum radius (µm) */
252
253 /* Build quadrature grid (same as aerosol.c / DISCOM) */
254 int nbmu = ctx->quad.nquad;
255 float *cgaus_S = (float*)malloc((size_t)nbmu * sizeof(float));
256 float *pdgs_S = (float*)malloc((size_t)nbmu * sizeof(float));
257 if (!cgaus_S || !pdgs_S) { free(cgaus_S); free(pdgs_S); return; }
258 sixs_gauss_setup(nbmu, cgaus_S, pdgs_S);
259
260 double *p11 = (double*)malloc((size_t)nbmu * sizeof(double));
261 double *ph_wl = (double*)calloc((size_t)(NWL_DISC * nbmu), sizeof(double));
262 if (!p11 || !ph_wl) {
263 free(cgaus_S); free(pdgs_S); free(p11); free(ph_wl);
264 return;
265 }
266
267 /* Accumulators for each reference wavelength */
268 double ext_acc[NWL_DISC] = {0};
269 double sca_acc[NWL_DISC] = {0};
270 double np_acc = 0.0;
271 double vol_acc = 0.0;
272
273 double log10_sigma = log10(sigma_g);
274
275 /* Integrate over particle radius */
276 double r = rmin;
277 double dr = r * (pow(10.0, rlogpas) - 1.0);
278
279 while (r < rmax) {
280 /* Log-normal dn/dr */
281 double logr_ratio = log10(r / r_mode) / log10_sigma;
282 double nr_dist = exp(-0.5 * logr_ratio * logr_ratio)
283 / (sqrt(2.0*pi) * log(10.0) * r * log10_sigma);
284 /* Projected area element */
285 double xndpr2 = nr_dist * dr * pi * r * r;
286 np_acc += nr_dist * dr;
287 vol_acc += r*r*r * nr_dist * dr;
288
289 for (int l = 0; l < NWL_DISC; l++) {
290 double wl = (double)wldisc_mie[l];
291 double X = 2.0 * pi * r / wl;
292 double Qext, Qsca;
293 exscphase(X, m_r_550, m_i_550, &Qext, &Qsca, p11, nbmu,
294 cgaus_S, pdgs_S);
295 ext_acc[l] += xndpr2 * Qext;
296 sca_acc[l] += xndpr2 * Qsca;
297 for (int k = 0; k < nbmu; k++)
298 ph_wl[l*nbmu + k] += 4.0 * p11[k] * xndpr2;
299 }
300
301 r += dr;
302 dr = r * (pow(10.0, rlogpas) - 1.0);
303 }
304
305 /* Normalise per-particle and store into ctx->aer */
306 /* 6SV normalises at 550 nm (index 7 in wldisc) */
307 double norm = (ext_acc[7] > 0.0) ? 1.0 / ext_acc[7] : 1.0;
308
309 for (int l = 0; l < NWL_DISC; l++) {
310 if (np_acc > 0.0) {
311 ext_acc[l] /= np_acc;
312 sca_acc[l] /= np_acc;
313 for (int k = 0; k < nbmu; k++)
314 ph_wl[l*nbmu + k] /= np_acc;
315 }
316 ctx->aer.ext[l] = (float)(ext_acc[l] * norm);
317 ctx->aer.ome[l] = (ext_acc[l] > 0.0) ? (float)(sca_acc[l] / ext_acc[l]) : 0.0f;
318
319 /* Asymmetry parameter from phase function */
320 double asy_n = 0.0, asy_d = 0.0;
321 for (int k = 0; k < nbmu; k++) {
322 double ph_k = (sca_acc[l] > 0.0) ? ph_wl[l*nbmu+k] / sca_acc[l] : 0.0;
323 asy_n += (double)cgaus_S[k] * ph_k * (double)pdgs_S[k];
324 asy_d += ph_k * (double)pdgs_S[k];
325 }
326 ctx->aer.gasym[l] = (float)((asy_d > 0.0) ? asy_n / asy_d : 0.0);
327
328 /* Phase function at the scattering angle (from ctx->polar.pha[0..nbmu-1]) */
329 /* Store at index 0 (forward scatter) for now; aerosol.c rebuilds polar */
330 ctx->aer.phase[l] = (sca_acc[l] > 0.0 && nbmu > 0)
331 ? (float)(ph_wl[l*nbmu + 0] / sca_acc[l])
332 : 0.0f;
333 }
334
335 /* Build phase function in ctx->polar (Legendre expansion at 550 nm) */
336 if (sca_acc[7] > 0.0) {
337 for (int k = 0; k < nbmu; k++) {
338 ctx->polar.pha[k] = (float)(ph_wl[7*nbmu + k] / sca_acc[7]);
339 }
340 /* Legendre coefficients */
341 float *betal = ctx->polar.betal;
342 for (int kk = 0; kk <= nbmu - 3; kk++) betal[kk] = 0.0f;
343 for (int j = 0; j < nbmu; j++) {
344 float x = ctx->polar.pha[j] * pdgs_S[j];
345 float rm_ = cgaus_S[j];
346 float plm1 = 0.0f, pl0 = 1.0f;
347 for (int k = 0; k <= nbmu - 3; k++) {
348 float plnew = ((2.0f*k + 1.0f)*rm_*pl0 - (float)k*plm1) / (float)(k+1);
349 betal[k] += x * pl0;
350 plm1 = pl0; pl0 = plnew;
351 }
352 }
353 for (int k = 0; k <= nbmu - 3; k++) {
354 betal[k] *= (2.0f*k + 1.0f) * 0.5f;
355 if (betal[k] < 0.0f) {
356 for (int j = k; j <= nbmu - 3; j++) betal[j] = 0.0f;
357 break;
358 }
359 }
360 }
361
362 free(cgaus_S); free(pdgs_S); free(p11); free(ph_wl);
363}
