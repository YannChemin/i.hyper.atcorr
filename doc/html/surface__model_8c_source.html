82/* Linear interpolation of a tabulated function at x. */
83static float interp1(const float *xp, const float *yp, int n, float x)
84{
85 if (x <= xp[0]) return yp[0];
86 if (x >= xp[n - 1]) return yp[n - 1];
87 for (int i = 0; i < n - 1; i++) {
88 if (xp[i + 1] >= x) {
89 float t = (x - xp[i]) / (xp[i + 1] - xp[i]);
90 return yp[i] + t * (yp[i + 1] - yp[i]);
91 }
92 }
93 return yp[n - 1];
94}

98SurfaceModelImpl *surface_model_alloc(const float *wl_um, int n_bands)
99{
100 SurfaceModelImpl *mdl = calloc(1, sizeof(*mdl));
101 if (!mdl) return NULL;
102
103 mdl->n_bands = n_bands;
104 mdl->means = malloc((size_t)SM_N_COMPONENTS * n_bands * sizeof(float));
105 mdl->variances = malloc((size_t)SM_N_COMPONENTS * n_bands * sizeof(float));
106 mdl->vnir_mask = malloc((size_t)n_bands * sizeof(int));
107
108 if (!mdl->means || !mdl->variances || !mdl->vnir_mask) {
109 surface_model_free(mdl);
110 return NULL;
111 }
112
113 for (int c = 0; c < SM_N_COMPONENTS; c++) {
114 for (int b = 0; b < n_bands; b++) {
115 float mean = interp1(REF_WL, REFL_TABLE[c], SM_N_REF, wl_um[b]);
116 float sd = VAR_SCALE[c] * mean;
117 if (sd < 1e-3f) sd = 1e-3f;
118 mdl->means [c * n_bands + b] = mean;
119 mdl->variances[c * n_bands + b] = sd * sd; /* store σ² */
120 }
121 }
122
123 for (int b = 0; b < n_bands; b++)
124 mdl->vnir_mask[b] = (wl_um[b] >= 0.400f && wl_um[b] <= 1.000f);
125
126 return mdl;
127}

129int surface_model_classify(const SurfaceModelImpl *mdl,
130 const float *refl, int n_bands)
131{
132 int best_comp = 1; /* default: soil */
133 float best_dist = 1e30f;
134
135 for (int c = 0; c < SM_N_COMPONENTS; c++) {
136 const float *ref = mdl->means + c * n_bands;
137 float dist = 0.0f;
138 int n = 0;
139 for (int b = 0; b < n_bands; b++) {
140 if (!mdl->vnir_mask[b]) continue;
141 if (!isfinite(refl[b])) continue;
142 float d = refl[b] - ref[b];
143 dist += d * d;
144 n++;
145 }
146 if (n > 0 && dist < best_dist) {
147 best_dist = dist;
148 best_comp = c;
149 }
150 }
151 return best_comp;
152}

154void surface_model_regularize(const SurfaceModelImpl *mdl,
155 float *refl_cube,
156 const float *sigma2,
157 int n_bands, int npix,
158 float weight)
159{
160 /* Stack buffer limit — sufficient for Tanager (426 bands) and beyond. */
161#define SM_MAX_BANDS 2048
162 if (n_bands > SM_MAX_BANDS) n_bands = SM_MAX_BANDS;
163
164#ifdef _OPENMP
165#pragma omp parallel for schedule(dynamic, 64)
166#endif
167 for (int p = 0; p < npix; p++) {
168 /* Copy pixel spectrum into a contiguous local buffer. */
169 float tmp[SM_MAX_BANDS];
170 for (int b = 0; b < n_bands; b++)
171 tmp[b] = refl_cube[(size_t)b * npix + p];
172
173 /* Classify to nearest component using VNIR bands. */
174 int comp = surface_model_classify(mdl, tmp, n_bands);
175
176 const float *xa = mdl->means + comp * mdl->n_bands;
177 const float *sa = mdl->variances + comp * mdl->n_bands;
178
179 /* MAP blend per band (diagonal covariance). */
180 for (int b = 0; b < n_bands; b++) {
181 float r = tmp[b];
182 if (!isfinite(r)) continue;
183
184 float inv_so;
185 if (sigma2 != NULL) {
186 float s2 = sigma2[(size_t)b * npix + p];
187 inv_so = (isfinite(s2) && s2 > 1e-20f) ? 1.0f / s2 : 1e20f;
188 } else {
189 /* Effective obs variance from weight parameter */
190 float sa_b = sa[b] > 1e-20f ? sa[b] : 1e-20f;
191 float s2_obs = sa_b * (1.0f - weight)
192 / (weight > 1e-10f ? weight : 1e-10f);
193 inv_so = 1.0f / (s2_obs > 1e-20f ? s2_obs : 1e-20f);
194 }
195 float inv_sa = 1.0f / (sa[b] > 1e-20f ? sa[b] : 1e-20f);
196
197 refl_cube[(size_t)b * npix + p] =
198 (r * inv_so + xa[b] * inv_sa) / (inv_so + inv_sa);
199 }
200 }
201#undef SM_MAX_BANDS
202}

206void surface_model_discrepancy(const float *wl_um, int n_wl, float *sigma_out)
207{
208 /* Empirical RT error floor — matches Python lib/uncertainty.py */
209 static const float centers[6] = {
210 0.720f, 0.760f, 0.940f, 1.135f, 1.380f, 1.850f
211 };
212
213 for (int i = 0; i < n_wl; i++) {
214 float s = 0.005f; /* baseline 0.5 % */
215 for (int k = 0; k < 6; k++) {
216 float dw = (wl_um[i] - centers[k]) / 0.030f;
217 s += 0.02f * expf(-0.5f * dw * dw);
218 }
219 if (wl_um[i] > 1.5f) s += 0.01f;
220 sigma_out[i] = s;
221 }
222}
