1/* Surface pressure / altitude adjustment — ported from 6SV2.1 PRESSURE.f.
2 *
3 * Adjusts the standard atmosphere profile stored in ctx->atm so that its
4 * surface (bottom) level corresponds to the given pressure or altitude.
5 * The lower layers are interpolated; the upper layers are preserved as-is.
6 *
7 * Convention (matching Fortran):
8 * sp > 0 : surface pressure in hPa (most common case for elevated terrain)
9 * sp < 0 : surface altitude in km (negative sign triggers alt-mode)
10 * sp == 0 : no-op (standard atmosphere unchanged)
11 */

16/* Compute integrated H2O and O3 columns after profile adjustment.
17 * Returns uw (g/cm²) and uo3 (atm-cm × 1000 Dobson). */
18static void compute_columns(const SixsCtx *ctx, float *uw, float *uo3)
19{
20 const float g = 98.1f;
21 const float air = 0.028964f / 0.0224f;
22 const float ro3 = 0.048f / 0.0224f;
23
24 float rmwh[NATM], rmo3[NATM];
25 for (int k = 0; k < NATM - 1; k++) {
26 float roair = air * 273.16f * ctx->atm.p[k]
27 / (1013.25f * ctx->atm.t[k]);
28 rmwh[k] = ctx->atm.wh[k] / (roair * 1000.0f);
29 rmo3[k] = ctx->atm.wo[k] / (roair * 1000.0f);
30 }
31
32 float uw_ = 0.0f, uo3_ = 0.0f;
33 for (int k = 1; k < NATM - 1; k++) {
34 float ds = (ctx->atm.p[k-1] - ctx->atm.p[k]) / ctx->atm.p[0];
35 uw_ += 0.5f * (rmwh[k] + rmwh[k-1]) * ds;
36 uo3_ += 0.5f * (rmo3[k] + rmo3[k-1]) * ds;
37 }
38 uw_ = uw_ * ctx->atm.p[0] * 100.0f / g;
39 uo3_ = uo3_ * ctx->atm.p[0] * 100.0f / g;
40 uo3_ = 1000.0f * uo3_ / ro3;
41
42 *uw = uw_;
43 *uo3 = uo3_;
44}

46/* Adjust atmosphere profile to surface pressure sp (hPa, positive)
47 * or surface altitude xps2 (km, negative sign → |sp|).
48 * Modifies ctx->atm in place. */
49void sixs_pressure(SixsCtx *ctx, float sp)
50{
51 if (sp == 0.0f) return;
52
53 int isup, iinf;
54 float ps, xalt;
55
56 if (sp < 0.0f) {
57 /* ── Altitude mode: sp = -altitude_km ──────────────────────────── */
58 float xps2 = -sp;
59 if (xps2 >= 100.0f) xps2 = 99.99f;
60
61 /* Find bracketing levels by altitude */
62 int i = 0;
63 while (i < NATM - 1 && ctx->atm.z[i] <= xps2) i++;
64 isup = i;
65 iinf = i - 1;
66
67 /* Log-linear pressure interpolation */
68 float xa = (ctx->atm.z[isup] - ctx->atm.z[iinf])
69 / logf(ctx->atm.p[isup] / ctx->atm.p[iinf]);
70 float xb = ctx->atm.z[isup] - xa * logf(ctx->atm.p[isup]);
71 ps = expf((xps2 - xb) / xa);
72 xalt = xps2;
73 } else {
74 /* ── Pressure mode: sp = surface_pressure_hPa ────────────────── */
75 if (sp >= 1013.0f) {
76 /* Surface pressure higher than standard — shift bottom layers */
77 float dps = sp - ctx->atm.p[0];
78 for (int i = 0; i < 9 && ctx->atm.p[i] > dps; i++)
79 ctx->atm.p[i] += dps;
80 return;
81 }
82 ps = sp;
83
84 /* Find first level where p < ps (pressure decreases upward) */
85 int i = 0;
86 while (i < NATM - 1 && ctx->atm.p[i] >= ps) i++;
87 isup = i;
88 iinf = i - 1;
89
90 /* Log-linear altitude interpolation */
91 float xa = (ctx->atm.z[isup] - ctx->atm.z[iinf])
92 / logf(ctx->atm.p[isup] / ctx->atm.p[iinf]);
93 float xb = ctx->atm.z[isup] - xa * logf(ctx->atm.p[isup]);
94 xalt = logf(ps) * xa + xb;
95 }
96
97 /* Linearly interpolate T, wh, wo at xalt between iinf and isup */
98 float dz = ctx->atm.z[isup] - ctx->atm.z[iinf];
99 float t_frac = (dz > 0.0f) ? (xalt - ctx->atm.z[iinf]) / dz : 0.0f;
100
101 float xtemp = ctx->atm.t[iinf] + t_frac * (ctx->atm.t[isup] - ctx->atm.t[iinf]);
102 float xwh = ctx->atm.wh[iinf] + t_frac * (ctx->atm.wh[isup] - ctx->atm.wh[iinf]);
103 float xwo = ctx->atm.wo[iinf] + t_frac * (ctx->atm.wo[isup] - ctx->atm.wo[iinf]);
104
105 /* Rebuild profile: new bottom = (xalt, ps, xtemp, xwh, xwo)
106 * then copy layers from iinf onward into positions 1, 2, ...
107 * Fortran: do i=2, 33-iinf+1 → 0-based: i=1 to 32-iinf */
108 ctx->atm.z[0] = xalt;
109 ctx->atm.p[0] = ps;
110 ctx->atm.t[0] = xtemp;
111 ctx->atm.wh[0] = xwh;
112 ctx->atm.wo[0] = xwo;
113
114 int n_copy = 33 - iinf; /* number of levels to copy */
115 for (int i = 1; i <= n_copy; i++) {
116 ctx->atm.z[i] = ctx->atm.z[i + iinf];
117 ctx->atm.p[i] = ctx->atm.p[i + iinf];
118 ctx->atm.t[i] = ctx->atm.t[i + iinf];
119 ctx->atm.wh[i] = ctx->atm.wh[i + iinf];
120 ctx->atm.wo[i] = ctx->atm.wo[i + iinf];
121 }
122
123 /* Fill remaining levels with linear extrapolation to TOA */
124 int l = n_copy; /* last filled index (0-based) */
125 for (int i = l + 1; i < NATM; i++) {
126 float frac = (float)(i - l) / (float)(NATM - 1 - l);
127 ctx->atm.z[i] = ctx->atm.z[l] + (ctx->atm.z[NATM-1] - ctx->atm.z[l]) * frac;
128 ctx->atm.p[i] = ctx->atm.p[l] + (ctx->atm.p[NATM-1] - ctx->atm.p[l]) * frac;
129 ctx->atm.t[i] = ctx->atm.t[l] + (ctx->atm.t[NATM-1] - ctx->atm.t[l]) * frac;
130 ctx->atm.wh[i] = ctx->atm.wh[l] + (ctx->atm.wh[NATM-1] - ctx->atm.wh[l]) * frac;
131 ctx->atm.wo[i] = ctx->atm.wo[l] + (ctx->atm.wo[NATM-1] - ctx->atm.wo[l]) * frac;
132 }
133}
