36/* ── Rahman-Pinty-Verstraete (RPV) ──────────────────────────────────────── */
37/* RAHMBRDF.f: rho0 intensity, af asymmetry factor, k structural parameter */
38static float brdf_rahman(const BrdfParams *p,
39 float mu1, float mu2, float fi)
40{
41 float rho0 = p->rahman.rho0;
42 float af = p->rahman.af;
43 float k = p->rahman.k;
44
45 float cospha = mu1*mu2 + sqrtf(1.0f - mu1*mu1)*sqrtf(1.0f - mu2*mu2)*cosf(fi);
46 float tante1 = sqrtf(1.0f - mu1*mu1) / (mu1 + 1e-10f);
47 float tante2 = sqrtf(1.0f - mu2*mu2) / (mu2 + 1e-10f);
48 float geofac = sqrtf(tante1*tante1 + tante2*tante2
49 - 2.0f*tante1*tante2*cosf(fi));
50
51 float coef1 = powf(mu1, k - 1.0f) * powf(mu2, k - 1.0f)
52 / powf(mu1 + mu2, 1.0f - k);
53 /* Phase function: HG with argument -cospha (backscatter convention) */
54 float phafun = (1.0f - af*af)
55 / powf(1.0f + af*af - 2.0f*af*(-cospha), 1.5f);
56 float coef2 = 1.0f + (1.0f - rho0) / (1.0f + geofac);
57
58 return rho0 * coef1 * phafun * coef2;
59}

61/* ── Roujean volumetric kernel ────────────────────────────────────────────── */
62/* ROUJBRDF.f: k0 isotropic, k1 geometric-optical, k2 volumetric */
63static float brdf_roujean(const BrdfParams *p,
64 float xmus, float xmuv, float fi)
65{
66 const float pi = (float)M_PI;
67
68 float tts = tanf(acosf(clampf(xmus, 1e-6f, 1.0f)));
69 float ttv = tanf(acosf(clampf(xmuv, 1e-6f, 1.0f)));
70
71 float cpsi = xmus*xmuv + sqrtf(1.0f - xmus*xmus)*sqrtf(1.0f - xmuv*xmuv)*cosf(fi);
72 cpsi = clampf(cpsi, -1.0f, 1.0f);
73 float psi = (cpsi < 1.0f) ? acosf(cpsi) : 0.0f;
74
75 /* fr = |fi| after acos(cos(fi)) — ensures range [0, pi] */
76 float fr = acosf(clampf(cosf(fi), -1.0f, 1.0f));
77
78 float f2 = 4.0f / (3.0f*pi*(xmus + xmuv))
79 * ((pi/2.0f - psi)*cpsi + sinf(psi)) - 1.0f/3.0f;
80
81 float f1 = 0.5f * ((pi - fr)*cosf(fr) + sinf(fr)) * tts * ttv
82 - tts - ttv
83 - sqrtf(tts*tts + ttv*ttv - 2.0f*tts*ttv*cosf(fr));
84 f1 /= pi;
85
86 return p->roujean.k0 + p->roujean.k1*f1 + p->roujean.k2*f2;
87}

89/* ── Hapke canopy model ───────────────────────────────────────────────────── */
90/* HAPKBRDF.f: om albedo, af asymmetry, s0 hotspot amplitude, h hotspot width */
91static float brdf_hapke(const BrdfParams *p,
92 float mu1, float mu2, float fi)
93{
94 float om = p->hapke.om;
95 float af = p->hapke.af;
96 float s0 = p->hapke.s0;
97 float h = p->hapke.h;
98
99 float cg = mu1*mu2 + sqrtf(1.0f - mu1*mu1)*sqrtf(1.0f - mu2*mu2)*cosf(fi);
100 cg = clampf(cg, -1.0f, 1.0f);
101
102 float f = om / 4.0f / (mu2 + mu1 + 1e-10f);
103 float sq = sqrtf(clampf(1.0f - om, 0.0f, 1.0f));
104 float h1 = (1.0f + 2.0f*mu1) / (1.0f + 2.0f*sq*mu1);
105 float h2 = (1.0f + 2.0f*mu2) / (1.0f + 2.0f*sq*mu2);
106
107 float pg = (1.0f - af*af) / powf(1.0f + af*af + 2.0f*af*cg, 1.5f);
108 float p0 = (1.0f - af*af) / powf(1.0f + af*af + 2.0f*af, 1.5f);
109
110 float g = acosf(cg);
111 float bg_d = 1.0f + tanf(g * 0.5f) / (h + 1e-10f);
112 float bg = (om*p0 > 0.0f) ? s0 / (om*p0) / bg_d : 0.0f;
113
114 return f * ((1.0f + bg)*pg + h1*h2 - 1.0f);
115}

117/* ── Ocean BRDF — simplified Cox-Munk + whitecaps + water body ───────────── */
118/* Based on OCEABRDF.f; uses simplified Cox-Munk slope distribution.
119 * Full treatment requires indwat/MORCASIWAT/gauss integration (not ported);
120 * this provides a reasonable first-order approximation. */
121static float brdf_ocean(const BrdfParams *p,
122 float mu1, float mu2, float fi)
123{
124 /* Cox-Munk slope variance from wind speed */
125 float wspd = p->ocean.wspd;
126 float azw = p->ocean.azw * (float)(M_PI / 180.0);
127
128 float sigma2 = 0.003f + 0.00512f * wspd; /* isotropic approximation */
129
130 /* Specular glint: find half-angle geometry */
131 float theta1 = acosf(clampf(mu1, 0.0f, 1.0f));
132 float theta2 = acosf(clampf(mu2, 0.0f, 1.0f));
133
134 /* Normal to the specularly reflecting facet */
135 float cos_n = (cosf(theta1) + cosf(theta2));
136 float sin1 = sinf(theta1), sin2 = sinf(theta2);
137 float norm_n = sqrtf(sin1*sin1 + sin2*sin2 + 2.0f*sin1*sin2*cosf(fi + azw)
138 + cos_n*cos_n);
139 /* cos(tilt) of specular facet from vertical */
140 float cos_tilt = (norm_n > 0.0f) ? cos_n / norm_n : 1.0f;
141 float cos_tilt2 = cos_tilt*cos_tilt;
142
143 /* Cox-Munk Gaussian slope distribution */
144 float tan_tilt2 = (cos_tilt2 > 1e-6f)
145 ? (1.0f - cos_tilt2) / cos_tilt2 : 0.0f;
146 float p_cox = (sigma2 > 0.0f)
147 ? expf(-tan_tilt2 / (2.0f*sigma2)) / ((float)M_PI * sigma2 * cos_tilt2*cos_tilt2)
148 : 0.0f;
149
150 /* Fresnel reflectance for seawater at ~550 nm, nr≈1.34 */
151 const float nr = 1.34f;
152 float cos_i = clampf((cosf(theta1) + cosf(theta2)) * 0.5f, 0.0f, 1.0f);
153 float sin_t = sqrtf(clampf(1.0f - cos_i*cos_i, 0.0f, 1.0f)) / nr;
154 float cos_t = sqrtf(clampf(1.0f - sin_t*sin_t, 0.0f, 1.0f));
155 float rs = (cos_i - nr*cos_t) / (cos_i + nr*cos_t + 1e-10f);
156 float rp = (nr*cos_i - cos_t) / (nr*cos_i + cos_t + 1e-10f);
157 float fresnel = 0.5f * (rs*rs + rp*rp);
158
159 /* Sunglint reflectance: π × fresnel × p_cox / (4 × μ1 × μ2) */
160 float rog = (mu1 > 0.0f && mu2 > 0.0f)
161 ? (float)M_PI * fresnel * p_cox / (4.0f * mu1 * mu2)
162 : 0.0f;
163
164 /* Whitecap reflectance (Koepke 1984) */
165 float W = 2.95e-6f * powf(wspd, 3.52f);
166 float Rwc = W * 0.22f; /* approximate whitecap reflectance ~0.22 */
167
168 /* Water body below-surface reflectance (Morel simplified) */
169 float C = p->ocean.pcl;
170 float wl = p->ocean.wl;
171 /* Approximate Rw (Gordon & Morel 1983) */
172 float kw = (wl < 0.5f) ? 0.01f : (wl < 0.7f) ? 0.05f : 1.0f;
173 float kc = 0.06f * powf(C + 1e-6f, 0.65f);
174 float Rw = 0.33f * kc / (kw + kc + 1e-10f);
175
176 /* Combine: total = whitecaps + (1-W)*glint + water body */
177 float nr2 = nr*nr;
178 float Rwb = (nr2 > 0.0f) ? Rw / nr2 : Rw;
179
180 return Rwc + (1.0f - W)*rog + (1.0f - Rwc)*Rwb;
181}

183/* ── Walthall polynomial ──────────────────────────────────────────────────── */
184/* WALTBRDF.f */
185static float brdf_walthall(const BrdfParams *p,
186 float xmus, float xmuv, float fi)
187{
188 float ts = acosf(clampf(xmus, -1.0f, 1.0f));
189 float tv = acosf(clampf(xmuv, -1.0f, 1.0f));
190 float phi = fi;
191
192 return p->walthall.a * (ts*ts*tv*tv)
193 + p->walthall.ap * (ts*ts + tv*tv)
194 + p->walthall.b * ts*tv*cosf(phi)
195 + p->walthall.c;
196}

198/* ── Minnaert ─────────────────────────────────────────────────────────────── */
199/* MINNBRDF.f: par1=k (exponent), par2=b (albedo normalisation) */
200static float brdf_minnaert(const BrdfParams *p,
201 float xmus, float xmuv)
202{
203 float k = p->minnaert.k;
204 float b = p->minnaert.b;
205 float prod = clampf(xmus * xmuv, 1e-10f, 1.0f);
206 return 0.5f * b * (k + 1.0f) * powf(prod, k - 1.0f);
207}

209/* ── Ross-Thick + Li-Sparse + Maignan hot-spot ────────────────────────────── */
210/* ROSSLIMAIGNANBRDF.f (rlmaignanbrdf): p1=f_iso, p2=f_vol, p3=f_geo */
211static float brdf_rosslimaignan(const BrdfParams *p,
212 float mu_s, float mu_v, float fi)
213{
214 const float pi = (float)M_PI;
215
216 /* Clamp sza to 75°, vza to 65° as in Fortran */
217 float rts = acosf(clampf(mu_s, cosf(75.0f*pi/180.0f), 1.0f));
218 float rtv = acosf(clampf(mu_v, cosf(65.0f*pi/180.0f), 1.0f));
219 float rfi = fabsf(fi);
220
221 float cts = cosf(rts), ctv = cosf(rtv);
222 float sts = sinf(rts), stv = sinf(rtv);
223 float cfi = cosf(rfi), sfi = sinf(rfi);
224
225 float cpha = cts*ctv + sts*stv*cfi;
226 cpha = clampf(cpha, -1.0f, 1.0f);
227 float rpha = acosf(cpha);
228
229 /* Ross-Thick kernel (Maignan hot-spot modification) */
230 float ct0 = 2.0f / (3.0f*pi);
231 float rosselt = ct0 * (((pi - 2.0f*rpha)*cpha + 2.0f*sinf(rpha)) / (cts + ctv + 1e-10f));
232 float rossthick = rosselt * (1.0f + 1.0f/(1.0f + rpha/(1.5f*pi/180.0f))) - 1.0f/3.0f;
233
234 /* Li-Sparse geometric kernel */
235 float tanti = tanf(rts);
236 float tantv = tanf(rtv);
237
238 float angdist = sqrtf(clampf(tanti*tanti + tantv*tantv - 2.0f*tanti*tantv*cfi, 0.0f, 1e10f));
239 float angtemp = 1.0f/(cts + 1e-10f) + 1.0f/(ctv + 1e-10f);
240
241 float cost = 2.0f * sqrtf(angdist*angdist + tanti*tanti*tantv*tantv*sfi*sfi)
242 / (angtemp + 1e-10f);
243 cost = clampf(cost, -1.0f, 1.0f);
244 float tvar = acosf(cost);
245 float sint = sqrtf(clampf(1.0f - cost*cost, 0.0f, 1.0f));
246 float angover = (tvar - sint*cost) * angtemp / pi;
247 float lispars = angover - angtemp + 0.5f*(1.0f + cpha) / (cts*ctv + 1e-10f);
248
249 float result = p->rosslimaignan.f_iso
250 + p->rosslimaignan.f_vol * rossthick
251 + p->rosslimaignan.f_geo * lispars;
252 return (result < 0.0f) ? 0.0f : result;
253}

255/* ══════════════════════════════════════════════════════════════════════════════
256 * sixs_brdf_eval — dispatcher
257 * ══════════════════════════════════════════════════════════════════════════════ */
258float sixs_brdf_eval(BrdfType type, const BrdfParams *params,
259 float cos_sza, float cos_vza, float raa_deg)
260{
261 /* Convert raa to radians; both kernels use fi = relative azimuth in radians */
262 float fi = raa_deg * (float)(M_PI / 180.0);
263
264 /* Ensure non-degenerate cos values */
265 float mu1 = clampf(cos_sza, 1e-4f, 1.0f);
266 float mu2 = clampf(cos_vza, 1e-4f, 1.0f);
267
268 switch (type) {
269 case BRDF_LAMBERTIAN:
270 /* Lambertian: caller uses params->lambertian.rho0 directly;
271 * the RT is handled analytically in atcorr_invert(). */
272 return params->lambertian.rho0;
273
274 case BRDF_RAHMAN:
275 return brdf_rahman(params, mu1, mu2, fi);
276
277 case BRDF_ROUJEAN:
278 return brdf_roujean(params, mu1, mu2, fi);
279
280 case BRDF_HAPKE:
281 return brdf_hapke(params, mu1, mu2, fi);
282
283 case BRDF_OCEAN:
284 return brdf_ocean(params, mu1, mu2, fi);
285
286 case BRDF_WALTHALL:
287 return brdf_walthall(params, mu1, mu2, fi);
288
289 case BRDF_MINNAERT:
290 return brdf_minnaert(params, mu1, mu2);
291
292 case BRDF_ROSSLIMAIGNAN:
293 return brdf_rosslimaignan(params, mu1, mu2, fi);
294
295 case BRDF_VERSFELD:
296 case BRDF_IAPI:
297 default:
298 /* Complex models with large external dependencies — stubs */
299 return 0.0f;
300 }
301}

303/* ══════════════════════════════════════════════════════════════════════════════
304 * sixs_brdf_albe — directional-hemispherical albedo by numerical integration
305 * ══════════════════════════════════════════════════════════════════════════════ */
306float sixs_brdf_albe(BrdfType type, const BrdfParams *params,
307 float cos_sza, int n_phi, int n_theta)
308{
309 if (n_phi < 2) n_phi = 48;
310 if (n_theta < 2) n_theta = 24;
311
312 const float pi = (float)M_PI;
313 const float dphi = 2.0f * pi / (float)n_phi;
314 const float dmu = 1.0f / (float)n_theta;
315
316 float albe = 0.0f;
317 for (int it = 0; it < n_theta; it++) {
318 float mu = (it + 0.5f) * dmu; /* midpoint rule in cos(vza) */
319 for (int ip = 0; ip < n_phi; ip++) {
320 float phi = (ip + 0.5f) * dphi; /* relative azimuth */
321 float raa_deg = phi * 180.0f / pi;
322 float brdf = sixs_brdf_eval(type, params, cos_sza, mu, raa_deg);
323 albe += brdf * mu * dmu * dphi;
324 }
325 }
326 albe /= pi; /* normalise by π for hemispherical reflectance */
327 return albe;
328}

330/* ══════════════════════════════════════════════════════════════════════════════
331 * NBAR normalisation — MODIS MCD43 standard Ross-Thick + Li-Sparse kernels
332 *
333 * These are the unmodified kernels from Lucht et al. (2000, IEEE TGRS 38:977).
334 * They differ from brdf_rosslimaignan() in that:
335 * - Ross-Thick: no Maignan hot-spot factor
336 * - Li-Sparse: b/r = 1 (no crown ellipsoid stretching), h/b = 2
337 *
338 * Input angles in degrees; uses cos/sin of those angles internally.
339 * Both kernels return 0 at nadir (vza=0), which is used for f_nbar.
340 * ══════════════════════════════════════════════════════════════════════════════ */
341
342/* Ross-Thick volumetric kernel (standard MODIS, no hot-spot).
343 * K_RT = [(π/2 − ξ)cos(ξ) + sin(ξ)] / (μs + μv) × (2/3π) − 1/3
344 * where ξ = arccos(μs μv + √(1−μs²)√(1−μv²) cos(Δφ)). */
345static float brdf_rossthick_std(float sza_deg, float vza_deg, float raa_deg)
346{
347 const float pi = (float)M_PI;
348
349 float rts = sza_deg * (pi / 180.0f);
350 float rtv = vza_deg * (pi / 180.0f);
351 float rfi = raa_deg * (pi / 180.0f);
352
353 float cts = cosf(rts), stv_s = sinf(rts);
354 float ctv = cosf(rtv), stv_v = sinf(rtv);
355
356 float cpha = clampf(cts*ctv + stv_s*stv_v*cosf(rfi), -1.0f, 1.0f);
357 float rpha = acosf(cpha);
358
359 float rosselt = (2.0f / (3.0f*pi))
360 * (((pi/2.0f - rpha)*cpha + sinf(rpha)) / (cts + ctv + 1e-10f));
361 return rosselt - 1.0f/3.0f;
362}

364/* Li-Sparse geometric kernel (standard MODIS, b/r=1, h/b=2).
365 * Uses the reciprocal overlap area O and the secant terms. */
366static float brdf_lisparse_std(float sza_deg, float vza_deg, float raa_deg)
367{
368 const float pi = (float)M_PI;
369
370 float rts = sza_deg * (pi / 180.0f);
371 float rtv = vza_deg * (pi / 180.0f);
372 float rfi = raa_deg * (pi / 180.0f);
373
374 float cts = cosf(rts), stv_s = sinf(rts);
375 float ctv = cosf(rtv), stv_v = sinf(rtv);
376 float cfi = cosf(rfi), sfi = sinf(rfi);
377
378 /* Phase angle for the (1 + cos ξ) term */
379 float cpha = clampf(cts*ctv + stv_s*stv_v*cfi, -1.0f, 1.0f);
380
381 /* tan(θs), tan(θv) [b/r = 1 so θs' = θs, θv' = θv] */
382 float tants = (cts > 1e-6f) ? stv_s / cts : stv_s / 1e-6f;
383 float tantv = (ctv > 1e-6f) ? stv_v / ctv : stv_v / 1e-6f;
384
385 float D2 = tants*tants + tantv*tantv - 2.0f*tants*tantv*cfi;
386 float D = sqrtf(clampf(D2, 0.0f, 1e10f));
387
388 float sec_s = (cts > 1e-6f) ? 1.0f / cts : 1.0f / 1e-6f;
389 float sec_v = (ctv > 1e-6f) ? 1.0f / ctv : 1.0f / 1e-6f;
390 float sec_sum = sec_s + sec_v;
391
392 /* cos(t): h/b = 2 factor */
393 float cost = 2.0f * sqrtf(clampf(D2 + tants*tants*tantv*tantv*sfi*sfi,
394 0.0f, 1e10f))
395 / (sec_sum + 1e-10f);
396 cost = clampf(cost, -1.0f, 1.0f);
397 float t = acosf(cost);
398 float sint = sqrtf(clampf(1.0f - cost*cost, 0.0f, 1.0f));
399
400 /* Overlap area O */
401 float O = (t - sint*cost) * sec_sum / pi;
402
403 return O - sec_sum + 0.5f*(1.0f + cpha) / (cts*ctv + 1e-10f);
404}

406/* ── Public: NBAR normalisation ─────────────────────────────────────────── */

408float atcorr_brdf_normalize(float rho_boa,
409 float f_iso, float f_vol, float f_geo,
410 float sza_obs, float vza_obs, float raa_obs,
411 float sza_nbar)
412{
413 /* Short-circuit: Lambertian (no anisotropy) */
414 if (f_vol == 0.0f && f_geo == 0.0f)
415 return rho_boa;
416
417 float K_vol_obs = brdf_rossthick_std(sza_obs, vza_obs, raa_obs);
418 float K_geo_obs = brdf_lisparse_std (sza_obs, vza_obs, raa_obs);
419 float K_vol_nbar = brdf_rossthick_std(sza_nbar, 0.0f, 0.0f);
420 float K_geo_nbar = brdf_lisparse_std (sza_nbar, 0.0f, 0.0f);
421
422 float f_obs = f_iso + f_vol * K_vol_obs + f_geo * K_geo_obs;
423 float f_nbar = f_iso + f_vol * K_vol_nbar + f_geo * K_geo_nbar;
424
425 /* Protect against division by zero */
426 if (fabsf(f_obs) < 1e-6f)
427 return rho_boa;
428
429 return rho_boa * (f_nbar / f_obs);
430}
