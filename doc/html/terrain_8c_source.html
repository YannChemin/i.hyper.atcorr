30/* ── Local illumination angle ────────────────────────────────────────────── */
31
32float cos_incidence(float sza_deg, float saa_deg,
33 float slope_deg, float aspect_deg)
34{
35 float sza = sza_deg * (float)(M_PI / 180.0);
36 float saa = saa_deg * (float)(M_PI / 180.0);
37 float slope = slope_deg * (float)(M_PI / 180.0);
38 float aspect = aspect_deg * (float)(M_PI / 180.0);
39
40 /* Standard topographic illumination model (Iqbal 1983) */
41 return cosf(sza) * cosf(slope)
42 + sinf(sza) * sinf(slope) * cosf(saa - aspect);
43}

53/* ── Effective downward transmittance on tilted surface ─────────────────── */
54
55float atcorr_terrain_T_down(float T_down, float T_down_dir,
56 float cos_sza, float cos_i, float V_d)
57{
58 float T_dif = T_down - T_down_dir;
59 if (T_dif < 0.0f) T_dif = 0.0f;
60
61 if (cos_i <= 0.0f)
62 /* Topographic shadow: only diffuse sky radiation reaches surface */
63 return T_dif * V_d;
64
65 /* Illuminated: direct component scaled by local/scene incidence ratio */
66 float cos_sza_safe = (cos_sza > 1e-6f) ? cos_sza : 1e-6f;
67 return T_down_dir * (cos_i / cos_sza_safe) + T_dif * V_d;
68}

70/* ── Per-pixel upward transmittance correction ───────────────────────────── */
71
72float atcorr_terrain_T_up(float T_up, float cos_vza_ref, float vza_pixel_deg)
73{
74 if (T_up <= 0.0f || T_up >= 1.0f) return T_up;
75
76 float cos_px = cosf(vza_pixel_deg * (float)(M_PI / 180.0));
77 if (cos_px < 1e-4f) cos_px = 1e-4f;
78 if (cos_vza_ref < 1e-4f) cos_vza_ref = 1e-4f;
79
80 /* T_up ∝ exp(−τ / cos_vza) → T_up(px) = T_up(ref) ^ (cos_ref / cos_px) */
81 return powf(T_up, cos_vza_ref / cos_px);
82}
