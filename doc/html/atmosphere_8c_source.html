210void sixs_init_atmosphere(SixsCtx *ctx, int atmo_model) {
211 const float *z, *p, *t, *wh, *wo;
212 switch (atmo_model) {
213 case 2: z=zms; p=pms; t=tms; wh=whms; wo=woms; break;
214 case 3: z=zmw; p=pmw; t=tmw; wh=whmw; wo=womw; break;
215 case 4: z=ztr; p=ptr; t=ttr; wh=whtr; wo=wotr; break;
216 case 5: z=zss; p=pss; t=tss; wh=whss; wo=woss; break;
217 case 6: z=zsw; p=psw; t=tsw; wh=whsw; wo=wosw; break;
218 default: z=z6; p=p6; t=t6; wh=wh6; wo=wo6; break;
219 }
220 for (int i = 0; i < NATM; i++) {
221 ctx->atm.z[i] = z[i];
222 ctx->atm.p[i] = p[i];
223 ctx->atm.t[i] = t[i];
224 ctx->atm.wh[i] = wh[i];
225 ctx->atm.wo[i] = wo[i];
226 }
227 /* Rayleigh depolarization factor for air (6SV standard) */
228 ctx->del.delta = 0.0279f;
229 ctx->del.sigma = 0.0f;
230 /* Default quadrature */
231 ctx->quad.nquad = NQ_P;
232 ctx->multi.igmax = 20;
233 ctx->err.ier = false;
234}
