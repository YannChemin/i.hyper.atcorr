139class TestIHyperAtcorrCorrect(TestCase):
140 """Full Raster3D atmospheric correction pipeline tests.
141
142 Creates a synthetic 5×5 Raster3D with 4 spectral bands, writes band
143 wavelength metadata to the map history, then runs i.hyper.atcorr to
144 produce a corrected output Raster3D.
145 """
146
147 input_map = "ihyper_test_input"
148 output_map = "ihyper_test_output"
149 lut_file = None
150
151 @classmethod
152 def setUpClass(cls):
153 cls.use_temp_region()
154 # 5×5 pixel region, 10 m resolution, 4 depth layers (= 4 bands)
155 cls.runModule(
156 "g.region",
157 n=50, s=0, e=50, w=0,
158 rows=5, cols=5,
159 t=_N_BANDS, b=0, tbres=1,
160 )
161 # Create uniform radiance Raster3D
162 cls.runModule(
163 "r3.mapcalc",
164 expression=f"{cls.input_map} = {_RADIANCE:.1f}",
165 overwrite=True,
166 )
167 # Write band wavelength metadata to history so main.c can parse it
168 # Each call to r3.support history= appends a line to the map history
169 for i, wl_nm in enumerate(_BAND_WL_NM, start=1):
170 cls.runModule(
171 "r3.support",
172 map=cls.input_mapinput_map,
173 history=f"Band {i}: {wl_nm:.1f} nm",
174 )
175 cls.lut_file = tempfile.mktemp(suffix=".lut", prefix="ihyper_corr_")
176
177 @classmethod
178 def tearDownClass(cls):
179 cls.runModule(
180 "g.remove", type="raster_3d", flags="f",
181 name=f"{cls.input_map},{cls.output_map}",
182 )
183 cls.del_temp_region()
184 if cls.lut_file and os.path.exists(cls.lut_file):
185 os.remove(cls.lut_file)
186
