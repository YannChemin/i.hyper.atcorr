37/* Fill ctx->polar.pha[] with the scattering-weighted mixed phase function
38 * for aerosol model iaer at reference wavelength index wl_idx. */
39static void set_mixed_pha(SixsCtx *ctx, int iaer, int wl_idx)
40{
41 int nquad = ctx->quad.nquad;
42 memset(ctx->polar.pha, 0, nquad * sizeof(float));
43 if (iaer == 0) return;
44
45 int mix_idx = (iaer == 2) ? 1 : (iaer == 3 ? 2 : 0);
46 const float *ci = std_mix_d[mix_idx];
47
48 float sigm = 0.0f;
49 for (int j = 0; j < 4; j++) if (ci[j] > 0.0f) sigm += ci[j] / vi_d[j];
50 if (sigm < 1e-10f) return;
51
52 /* Scattering-weighted sum */
53 float sca_mix = 0.0f;
54 for (int j = 0; j < 4; j++) {
55 if (ci[j] <= 0.0f) continue;
56 float cij = ci[j] / vi_d[j] / sigm;
57 sca_mix += cij * comp_sca_d[j][wl_idx];
58 }
59 if (sca_mix < 1e-10f) return;
60
61 for (int j = 0; j < 4; j++) {
62 if (ci[j] <= 0.0f) continue;
63 float cij = ci[j] / vi_d[j] / sigm;
64 float weight = cij * comp_sca_d[j][wl_idx] / sca_mix;
65 const float (*pha_j)[83] = comp_pha_d[j];
66 for (int k = 0; k < nquad; k++)
67 ctx->polar.pha[k] += weight * pha_j[wl_idx][k];
68 }
69}

71/* Set up the Gauss quadrature rm/gb arrays used by OS and SCATRA.
72 * rm[j+mu] = rm(j) in Fortran, j = -mu..mu.
73 * rm(1..mu-1): interior Gauss cosines; rm(0)=-xmus; rm(mu)=xmuv; rm(-k)=-rm(k).
74 * gb(1..mu-1): Gauss weights; gb(0)=gb(mu)=0. */
75static void setup_rm_gb(float xmus, float xmuv, int mu,
76 float *rm_off, float *gb_off)
77{
78 int nbmu = mu - 1; /* interior Gauss points per hemisphere */
79 float cgaus[MU_P], wgaus[MU_P];
80 sixs_gauss(0.0f, 1.0f, cgaus, wgaus, nbmu);
81
82 /* Zero entire array */
83 memset(rm_off, 0, (2*mu+1) * sizeof(float));
84 memset(gb_off, 0, (2*mu+1) * sizeof(float));
85
86 /* Interior positive Gauss points: rm(1..mu-1), gb(1..mu-1) */
87 for (int k = 1; k <= nbmu; k++) {
88 rm_off[k + mu] = cgaus[k - 1]; /* rm(k) */
89 rm_off[-k + mu] = -cgaus[k - 1]; /* rm(-k) */
90 gb_off[k + mu] = wgaus[k - 1]; /* gb(k) */
91 gb_off[-k + mu] = wgaus[k - 1]; /* gb(-k) */
92 }
93 /* Special: solar (0) and view (±mu) directions */
94 rm_off[0 + mu] = -xmus; /* rm(0) = -xmus */
95 rm_off[mu + mu] = xmuv; /* rm(mu) = xmuv */
96 rm_off[-mu + mu] = -xmuv; /* rm(-mu)= -xmuv */
97 gb_off[0 + mu] = 0.0f;
98 gb_off[mu + mu] = 0.0f;
99 gb_off[-mu + mu] = 0.0f;
100}

102/* ------------------------------------------------------------------ *
103 * sixs_discom: compute scattering properties at 20 reference wavelengths.
104 *
105 * ctx - context, must have atm/aer/del/quad/multi initialized
106 * idatmp - observation type: 0=ground sensor, 4=plane at palt km, else=satellite
107 * iaer - aerosol model (0=none, 1=continental, 2=maritime, 3=urban)
108 * xmus - cosine of solar zenith angle
109 * xmuv - cosine of view zenith angle
110 * phi - relative azimuth angle (degrees, for phase function init)
111 * taer55 - aerosol optical depth at 550nm (total)
112 * taer55p - aerosol optical depth at 550nm above sensor plane
113 * palt - sensor altitude km (>900 = satellite)
114 * phirad - relative azimuth angle (radians)
115 * nt - number of atmospheric layers
116 * mu - Gauss quadrature points per hemisphere (= MU_P = 25)
117 * np - azimuth planes in xl (1 for scalar output)
118 * ftray - fraction of Rayleigh OD above sensor (for idatmp=1,2,3)
119 * ipol - 0 = scalar (only mode implemented)
120 * ------------------------------------------------------------------ */
121void sixs_discom(SixsCtx *ctx,
122 int idatmp, int iaer,
123 float xmus, float xmuv, float phi,
124 float taer55, float taer55p, float palt, float phirad,
125 int nt, int mu, int np,
126 float ftray, int ipol)
127{
128 (void)phi; /* phi used for phase init (already done) */
129
130 /* Save reference wavelengths */
131 for (int l = 0; l < NWL_DISC; l++) ctx->disc.wldis[l] = wldisc[l];
132
133 /* Set up Gauss quadrature arrays (rm/gb) */
134 float rm_full[2*MU_P + 1], gb_full[2*MU_P + 1];
135 setup_rm_gb(xmus, xmuv, mu, rm_full, gb_full);
136
137 /* Azimuth planes (just one: phirad) */
138 float rp[1] = { phirad };
139 int nfi = 1; /* minimum for xlphim */
140
141 /* Allocate temporary xl/xlphim for os/ospol calls */
142 size_t xl_sz = (size_t)(2 * mu + 1) * np;
143 float *xl = (float*)calloc(xl_sz, sizeof(float));
144 float *xlphim = (float*)calloc(nfi, sizeof(float));
145 /* Q/U arrays for vector RT (only used when ipol=1) */
146 float *xlq = ipol ? (float*)calloc(xl_sz, sizeof(float)) : NULL;
147 float *xlu = ipol ? (float*)calloc(xl_sz, sizeof(float)) : NULL;
148 if (!xl || !xlphim || (ipol && (!xlq || !xlu))) {
149 free(xl); free(xlphim); free(xlq); free(xlu); return;
150 }
151
152 /* ===== Loop over 20 reference wavelengths ===== */
153 for (int l = 0; l < NWL_DISC; l++) {
154 float wl = wldisc[l];
155
156 /* Rayleigh optical depth */
157 float tray, trayp;
158 sixs_odrayl(ctx, wl, &tray);
159
160 /* Rayleigh OD above sensor */
161 if (idatmp == 4)
162 trayp = tray;
163 else if (idatmp == 0)
164 trayp = 0.0f;
165 else
166 trayp = tray * ftray;
167
168 ctx->disc.trayl[l] = tray;
169 ctx->disc.traypl[l] = trayp;
170
171 /* Aerosol optical depth at this wavelength */
172 float taer = taer55 * ctx->aer.ext[l];
173 float taerp = taer55p * ctx->aer.ext[l];
174 float piza = ctx->aer.ome[l];
175
176 /* Truncated values (coeff = 0 always in this 6SV version) */
177 float coeff = 0.0f;
178 float tamoy = taer, tamoyp = taerp, pizmoy = piza;
179
180 if (iaer != 0) {
181 /* Load mixed phase function for this wavelength → polar.pha[] */
182 set_mixed_pha(ctx, iaer, l);
183 /* Decompose into Legendre betal[] */
184 sixs_trunca(ctx, 0, &coeff);
185 /* coeff = 0 always, so tamoy/tamoyp/pizmoy unchanged */
186 tamoy = taer * (1.0f - piza * coeff);
187 tamoyp = taerp * (1.0f - piza * coeff);
188 pizmoy = (fabsf(1.0f - piza * coeff) > 1e-10f) ?
189 piza * (1.0f - coeff) / (1.0f - piza * coeff) : piza;
190 } else {
191 /* No aerosol: reset betal to Rayleigh values */
192 for (int k = 0; k <= ctx->quad.nquad - 3; k++) ctx->polar.betal[k] = 0.0f;
193 ctx->polar.betal[0] = 1.0f;
194 ctx->polar.betal[2] = 0.1f; /* Rayleigh: simple 2-term approximation */
195 }
196
197 /* ---- Atmospheric reflectances via sixs_os or sixs_ospol ---- */
198 float rorayl = 0.0f, roaero = 0.0f, romix = 0.0f;
199 float rorayl_q = 0.0f, roaero_q = 0.0f, romix_q = 0.0f;
200 float rorayl_u = 0.0f, roaero_u = 0.0f, romix_u = 0.0f;
201
202 /* Helper macro: reset xl (and xlq/xlu when ipol) and call the right RT */
203 #define RT_CALL(ta, tr, tap, trp) do { \
204 memset(xl, 0, xl_sz * sizeof(float)); \
205 memset(xlphim, 0, nfi * sizeof(float)); \
206 if (ipol) { \
207 memset(xlq, 0, xl_sz * sizeof(float)); \
208 memset(xlu, 0, xl_sz * sizeof(float)); \
209 sixs_ospol(ctx, 0, (ta), (tr), pizmoy, (tap), (trp), palt, \
210 phirad, nt, mu, np, nfi, \
211 rm_full, gb_full, rp, xl, xlq, xlu, xlphim); \
212 } else { \
213 sixs_os(ctx, 0, (ta), (tr), pizmoy, (tap), (trp), palt, \
214 phirad, nt, mu, np, nfi, \
215 rm_full, gb_full, rp, xl, xlphim, NULL); \
216 } \
217 } while (0)
218
219 /* Rayleigh-only reflectance: tamoy=0, trmoy=tray */
220 RT_CALL(0.0f, tray, 0.0f, trayp);
221 rorayl = xl[0] / xmus;
222 rorayl_q = xlq ? xlq[0] / xmus : 0.0f;
223 rorayl_u = xlu ? xlu[0] / xmus : 0.0f;
224
225 if (iaer != 0 && taer > 1e-6f) {
226 /* Aerosol-only reflectance: tamoy=taer, trmoy=0 */
227 RT_CALL(tamoy, 0.0f, tamoyp, 0.0f);
228 roaero = xl[0] / xmus;
229 roaero_q = xlq ? xlq[0] / xmus : 0.0f;
230 roaero_u = xlu ? xlu[0] / xmus : 0.0f;
231
232 /* Combined (Rayleigh + aerosol) reflectance */
233 RT_CALL(tamoy, tray, tamoyp, trayp);
234 romix = xl[0] / xmus;
235 romix_q = xlq ? xlq[0] / xmus : 0.0f;
236 romix_u = xlu ? xlu[0] / xmus : 0.0f;
237 } else {
238 roaero = 0.0f; roaero_q = 0.0f; roaero_u = 0.0f;
239 romix = rorayl;
240 romix_q = rorayl_q;
241 romix_u = rorayl_u;
242 }
243
244 #undef RT_CALL
245
246 ctx->disc.roatm[0][l] = rorayl;
247 ctx->disc.roatm[1][l] = romix;
248 ctx->disc.roatm[2][l] = roaero;
249 ctx->disc.roatmq[0][l] = rorayl_q;
250 ctx->disc.roatmq[1][l] = romix_q;
251 ctx->disc.roatmq[2][l] = roaero_q;
252 ctx->disc.roatmu[0][l] = rorayl_u;
253 ctx->disc.roatmu[1][l] = romix_u;
254 ctx->disc.roatmu[2][l] = roaero_u;
255
256 /* ---- Scattering transmittances ---- */
257 float ddirtt, ddiftt, udirtt, udiftt, sphalbt;
258 float ddirtr, ddiftr, udirtr, udiftr, sphalbr;
259 float ddirta, ddifta, udirta, udifta, sphalba;
260
261 sixs_scatra(ctx,
262 tamoy, tamoyp, tray, trayp,
263 pizmoy, palt, nt, mu,
264 rm_full, gb_full,
265 xmus, xmuv,
266 &ddirtt, &ddiftt, &udirtt, &udiftt, &sphalbt,
267 &ddirtr, &ddiftr, &udirtr, &udiftr, &sphalbr,
268 &ddirta, &ddifta, &udirta, &udifta, &sphalba);
269
270 /* Store: [0]=Rayleigh, [1]=total, [2]=aerosol */
271 ctx->disc.dtdir[0][l] = ddirtr; ctx->disc.dtdif[0][l] = ddiftr;
272 ctx->disc.utdir[0][l] = udirtr; ctx->disc.utdif[0][l] = udiftr;
273 ctx->disc.sphal[0][l] = sphalbr;
274
275 ctx->disc.dtdir[1][l] = ddirtt; ctx->disc.dtdif[1][l] = ddiftt;
276 ctx->disc.utdir[1][l] = udirtt; ctx->disc.utdif[1][l] = udiftt;
277 ctx->disc.sphal[1][l] = sphalbt;
278
279 ctx->disc.dtdir[2][l] = ddirta; ctx->disc.dtdif[2][l] = ddifta;
280 ctx->disc.utdir[2][l] = udirta; ctx->disc.utdif[2][l] = udifta;
281 ctx->disc.sphal[2][l] = sphalba;
282 }
283
284 free(xl);
285 free(xlphim);
286 free(xlq);
287 free(xlu);
288}
