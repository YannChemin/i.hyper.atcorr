10/* Get the 8 absorption coefficients for a given gas, band, interval.
11 * gas_id: 1=H2O, 2=CO2, 3=O2, 5=N2O, 6=CH4, 7=CO
12 * band: 1-6, inu: 1-256 */
13static void get_gas_coef(int gas_id, int band, int inu, float a[8]) {
14 /* Map gas_id to table index (0-based) */
15 const float (*tbl)[256][8] = NULL;
16 switch (gas_id) {
17 case 1: tbl = gas_acr_wava; break;
18 case 2: tbl = gas_acr_dica; break;
19 case 3: tbl = gas_acr_oxyg; break;
20 case 5: tbl = gas_acr_niox; break;
21 case 6: tbl = gas_acr_meth; break;
22 case 7: tbl = gas_acr_moca; break;
23 default: memset(a, 0, 8 * sizeof(float)); return;
24 }
25 int b = band - 1;
26 int i = inu - 1;
27 if (b < 0 || b >= 6 || i < 0 || i >= 256) {
28 memset(a, 0, 8 * sizeof(float)); return;
29 }
30 for (int k = 0; k < 8; k++) a[k] = tbl[b][i][k];
31}

33/* Compute gas transmittance for a single gas species.
34 * Uses the 6SV Curtis-Godson approximation.
35 * Returns transmittance (0-1). */
36static float compute_gas_trans(const SixsCtx *ctx, const float a[8],
37 double ud, double upd, int gas_id)
38{
39 if (a[0] == 0.0f && a[1] == 0.0f) return 1.0f;
40 double udt = (ud == 0.0 && upd == 0.0) ? 1.0 : ud;
41 double updt = (ud == 0.0 && upd == 0.0) ? 1.0 : upd;
42 double atest = (a[1] == 0.0f && a[0] == 0.0f) ? 1.0 : (double)a[1];
43 double y;
44 if (gas_id == 1) {
45 /* H2O: special formula */
46 y = -(double)a[0] * ud / sqrt(1.0 + ((double)a[0] / atest) * (ud * ud / updt));
47 } else {
48 double tn = (double)a[1] * upd / (2.0 * udt);
49 double tt = 1.0 + 4.0 * ((double)a[0] / atest) * (ud * ud / updt);
50 y = -tn * (sqrt(tt) - 1.0);
51 }
52 if (y < -86.0) return 0.0f;
53 return (float)exp(y);
54}

56/* Compute gas column amounts (u, up) for a given gas species and direction.
57 * Returns u (total column) and up (pressure-weighted column). */
58static void compute_column(const SixsCtx *ctx, int gas_id, const float a[8],
59 float xmus, double *u_out, double *up_out)
60{
61 const float *z = ctx->atm.z;
62 const float *p = ctx->atm.p;
63 const float *t = ctx->atm.t;
64 const float *wh = ctx->atm.wh;
65 const float *wo = ctx->atm.wo;
66
67 const float t0 = 250.0f;
68 const float p0 = 1013.25f;
69 const float g = 98.1f;
70 const float air = 0.028964f / 0.0224f;
71 const float roco2 = 0.044f / 0.0224f;
72 const float rmo2 = 0.032f / 0.0224f;
73 const float rmo3 = 0.048f / 0.0224f;
74 const float rmn2o = 0.044f / 0.0224f;
75 const float rmch4 = 0.016f / 0.0224f;
76 const float rmco = 0.028f / 0.0224f;
77
78 double uu = 0.0, u = 0.0, up = 0.0;
79 for (int k = 1; k < NATM - 1; k++) {
80 float tp = (t[k] + t[k+1]) / 2.0f;
81 float roair = air * 273.16f * p[k] / (1013.25f * t[k]);
82 float te = tp - t0;
83 float te2 = te * te;
84 float phi = expf((float)(a[2]) * te + (float)(a[3]) * te2);
85 float psi = expf((float)(a[4]) * te + (float)(a[5]) * te2);
86 float rm_k;
87 switch (gas_id) {
88 case 1: rm_k = wh[k] / (roair * 1000.0f); break;
89 case 2: rm_k = 3.3e-4f * roco2 / air; break;
90 case 3: rm_k = 0.20947f * rmo2 / air; break;
91 case 4: rm_k = wo[k] / (roair * 1000.0f); break;
92 case 5: rm_k = 310.0e-9f * rmn2o / air; break;
93 case 6: rm_k = 1.72e-6f * rmch4 / air; break;
94 case 7: rm_k = 1.00e-9f * rmco / air; break;
95 default: rm_k = 0.0f; break;
96 }
97 float r2 = rm_k * phi;
98 float r3 = rm_k * psi;
99 float ds = (p[k] - p[k+1]) / p[0];
100 float ds2 = (p[k] * p[k] - p[k+1] * p[k+1]) / (2.0f * p[0] * p0);
101 uu += rm_k * ds;
102 u += r2 * ds;
103 up += r3 * ds2;
104 }
105 uu *= (double)p[0] * 100.0 / g;
106 u *= (double)p[0] * 100.0 / g;
107 up *= (double)p[0] * 100.0 / g;
108
109 /* Unit conversions */
110 if (gas_id == 4) uu *= 1000.0 / rmo3;
111 if (gas_id == 2) uu *= 1000.0 / roco2;
112 if (gas_id == 5) uu *= 1000.0 / rmn2o;
113 if (gas_id == 6) uu *= 1000.0 / rmch4;
114 if (gas_id == 7) uu *= 1000.0 / rmco;
115
116 *u_out = u / xmus;
117 *up_out = up / xmus;
118}

120/* Main gas absorption function.
121 * Returns downward transmittance for all gases combined. */
122float sixs_gas_transmittance(const SixsCtx *ctx, float wl, float xmus, float xmuv,
123 float uw, float uo3)
124{
125 if (wl <= 0.0f) return 1.0f;
126 float v = 1.0e4f / wl;
127 int iv = (int)(v / 5.0f) * 5;
128 /* Band index */
129 int id = ((iv - 2500) / 10) / 256 + 1;
130 if (id < 1 || id > 6) return 1.0f;
131 int inu = (iv - ivli[id - 1]) / 10 + 1;
132 if (inu < 1 || inu > 256) return 1.0f;
133
134 float tgas_down = 1.0f;
135
136 /* Loop over gas species (ABSTRA loops idgaz=1..7) */
137 for (int gas_id = 1; gas_id <= 7; gas_id++) {
138 /* Check if this gas/band combination exists */
139 int has_data = 0;
140 if (gas_id == 1) has_data = 1; /* H2O: all bands */
141 if (gas_id == 2 && id <= 3) has_data = 1; /* CO2: bands 1-3 */
142 if (gas_id == 3 && id >= 3) has_data = 1; /* O2: bands 3-6 */
143 if (gas_id == 4) { /* O3: special treatment below */ continue; }
144 if (gas_id >= 5) has_data = 1; /* N2O/CH4/CO: all bands */
145 if (!has_data) continue;
146
147 float a[8];
148 get_gas_coef(gas_id, id, inu, a);
149 if (a[0] == 0.0f) continue;
150
151 /* H2O: scale by uw/uwus (uwus=1.424 g/cm²) */
152 float scale = 1.0f;
153 if (gas_id == 1 && uw > 0.0f) scale = uw / 1.424f;
154
155 double ud, upd;
156 compute_column(ctx, gas_id, a, xmus, &ud, &upd);
157 ud *= scale;
158 upd *= scale;
159
160 float t_d = compute_gas_trans(ctx, a, ud, upd, gas_id);
161 tgas_down *= t_d;
162 }
163
164 /* Ozone absorption (separate table: co3[102]) */
165 if (iv >= 13000 && iv <= 27400) {
166 extern const float gas_co3_ozon[102];
167 float xi;
168 if (iv <= 23400) xi = (v - 13000.0f) / 200.0f + 1.0f;
169 else xi = (v - 27500.0f) / 500.0f + 57.0f;
170 int n = (int)(xi + 1.001f);
171 float xd = xi - (float)n;
172 if (n >= 1 && n <= 102) {
173 float ako3 = gas_co3_ozon[n - 1] + xd * (gas_co3_ozon[n - 1] - (n > 1 ? gas_co3_ozon[n - 2] : gas_co3_ozon[0]));
174 /* Ozone column: uo3 in Dobson units / 1000 = atm-cm */
175 float uud = (uo3 > 0.0f ? uo3 / 1000.0f : 0.344f) / xmus;
176 float test = ako3 * uud;
177 if (test > 86.0f) test = 86.0f;
178 tgas_down *= expf(-test);
179 }
180 }
181
182 return tgas_down;
183}
