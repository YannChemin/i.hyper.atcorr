31/* ── Beer-Lambert direct transmittance ──────────────────────────────────────── */
32
33float adjacency_T_dir(float wl_um, float aod550, float pressure,
34 float sza_deg, float vza_deg)
35{
36 /* Rayleigh optical depth (Hansen & Travis 1974) */
37 float wl2 = wl_um * wl_um;
38 float wl4 = wl2 * wl2;
39 float tau_r = 0.008569f / wl4 * (1.0f + 0.0113f / wl2)
40 * (pressure / 1013.25f);
41
42 /* Aerosol optical depth (power-law with alpha = 1.3) */
43 float tau_a = aod550 * powf(wl_um / 0.55f, -1.3f);
44
45 float tau = tau_r + tau_a;
46 float us = cosf(sza_deg * (float)(M_PI / 180.0));
47 float uv = cosf(vza_deg * (float)(M_PI / 180.0));
48
49 return expf(-tau / us) * expf(-tau / uv);
50}

54void adjacency_correct_band(float *r_boa, int nrows, int ncols,
55 float psf_radius_km, float pixel_size_m,
56 float T_scat, float s_alb,
57 float wl_um, float aod550, float pressure,
58 float sza_deg, float vza_deg)
59{
60 int npix = nrows * ncols;
61
62 /* Filter half-width: PSF radius in pixels (minimum 1) */
63 int filter_half = (int)(psf_radius_km * 1000.0f / pixel_size_m + 0.5f);
64 if (filter_half < 1) filter_half = 1;
65
66 float *r_env = malloc((size_t)npix * sizeof(float));
67 if (!r_env) return;
68
69 adjacency_r_env(r_boa, r_env, nrows, ncols, filter_half);
70
71 float T_dir = adjacency_T_dir(wl_um, aod550, pressure, sza_deg, vza_deg);
72 float T_diff = T_scat - T_dir;
73 if (T_diff < 0.0f) T_diff = 0.0f;
74
75#ifdef _OPENMP
76#pragma omp parallel for schedule(static)
77#endif
78 for (int i = 0; i < npix; i++) {
79 float r = r_boa[i];
80 float re = r_env[i];
81 if (!isfinite(r) || !isfinite(re)) continue;
82 float denom = 1.0f - s_alb * re;
83 if (fabsf(denom) < 1e-10f) continue;
84 float corr = T_diff * s_alb * (r - re) / denom;
85 r_boa[i] = r + corr;
86 }
87
88 free(r_env);
89}
