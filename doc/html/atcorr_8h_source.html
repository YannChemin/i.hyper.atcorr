163static inline float atcorr_invert(float rho_toa, float R_atm,
164 float T_down, float T_up, float s_alb)
165{
166 float y = (rho_toa - R_atm) / (T_down * T_up + 1e-10f);
167 return y / (1.0f + s_alb * y + 1e-10f);
168}

190static inline float atcorr_invert_brdf(float rho_toa, float R_atm,
191 float T_down, float T_up,
192 float s_alb, float rho_albe)
193{
194 float y = (rho_toa - R_atm) / (T_down * T_up + 1e-10f);
195 return y * (1.0f - s_alb * rho_albe);
196}
