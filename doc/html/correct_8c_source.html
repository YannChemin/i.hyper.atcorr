3 * for use in the atmospheric correction inversion loop. */

6/* Fill output arrays [n_wl] with bilinear interpolation of the LUT at
7 * (aod_val, h2o_val). Values outside the grid are clamped to the boundary.
8 *
9 * Outputs Rs, Tds, Tus, ss must be pre-allocated with cfg->n_wl floats. */
10void atcorr_lut_slice(const LutConfig *cfg, const LutArrays *lut,
11 float aod_val, float h2o_val,
12 float *Rs, float *Tds, float *Tus, float *ss,
13 float *Tdds)
14{
15 int na = cfg->n_aod, nh = cfg->n_h2o, nw = cfg->n_wl;
16
17 /* AOD bracket */
18 int ia0 = 0;
19 while (ia0 < na - 2 && cfg->aod[ia0 + 1] <= aod_val) ia0++;
20 int ia1 = (ia0 < na - 1) ? ia0 + 1 : ia0;
21 float ta = (ia1 != ia0)
22 ? (aod_val - cfg->aod[ia0]) / (cfg->aod[ia1] - cfg->aod[ia0])
23 : 0.0f;
24 if (ta < 0.0f) ta = 0.0f;
25 if (ta > 1.0f) ta = 1.0f;
26
27 /* H2O bracket */
28 int ih0 = 0;
29 while (ih0 < nh - 2 && cfg->h2o[ih0 + 1] <= h2o_val) ih0++;
30 int ih1 = (ih0 < nh - 1) ? ih0 + 1 : ih0;
31 float th = (ih1 != ih0)
32 ? (h2o_val - cfg->h2o[ih0]) / (cfg->h2o[ih1] - cfg->h2o[ih0])
33 : 0.0f;
34 if (th < 0.0f) th = 0.0f;
35 if (th > 1.0f) th = 1.0f;
36
37#define IDX(ia, ih, iw) ((size_t)(ia) * nh * nw + (size_t)(ih) * nw + (iw))
38#define BI4(arr) \
39 ((arr)[IDX(ia0,ih0,iw)] * (1.f-ta)*(1.f-th) + \
40 (arr)[IDX(ia1,ih0,iw)] * ta *(1.f-th) + \
41 (arr)[IDX(ia0,ih1,iw)] * (1.f-ta)* th + \
42 (arr)[IDX(ia1,ih1,iw)] * ta * th)
43
44 for (int iw = 0; iw < nw; iw++) {
45 Rs[iw] = BI4(lut->R_atm);
46 Tds[iw] = BI4(lut->T_down);
47 Tus[iw] = BI4(lut->T_up);
48 ss[iw] = BI4(lut->s_alb);
49 if (Tdds && lut->T_down_dir)
50 Tdds[iw] = BI4(lut->T_down_dir);
51 else if (Tdds)
52 Tdds[iw] = Tds[iw]; /* fallback: no direct/diffuse split */
53 }
54#undef BI4
55#undef IDX
56}
