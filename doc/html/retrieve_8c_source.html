23/* ─── H2O 940 nm retrieval ────────────────────────────────────────────────── */
24
25void retrieve_h2o_940(const float *L_865, const float *L_940,
26 const float *L_1040, int npix,
27 float sza_deg, float vza_deg,
28 float *out_wvc)
29{
30 /* Fraction of the 865→1040 range occupied by 940-865 = 75/175 ≈ 0.4286 */
31 static const float FRAC = (0.940f - 0.865f) / (1.040f - 0.865f);
32 static const float K_940 = 0.036f; /* cm²/g; empirical at 940 nm */
33 static const float WVC_DEF = 2.0f; /* fallback for invalid pixels */
34 static const float WVC_MIN = 0.1f;
35 static const float WVC_MAX = 8.0f;
36
37 float cos_sza = cosf(sza_deg * (float)(M_PI / 180.0));
38 float cos_vza = cosf(vza_deg * (float)(M_PI / 180.0));
39 float mu_s = (cos_sza > 0.05f) ? cos_sza : 0.05f;
40 float mu_v = (cos_vza > 0.05f) ? cos_vza : 0.05f;
41 float km = K_940 * (1.0f / mu_s + 1.0f / mu_v);
42
43 for (int i = 0; i < npix; i++) {
44 float l8 = L_865[i], l9 = L_940[i], l10 = L_1040[i];
45 if (!(l8 > 0.0f) || !(l9 > 0.0f) || !(l10 > 0.0f)) {
46 out_wvc[i] = WVC_DEF;
47 continue;
48 }
49 float L_cont = l8 * (1.0f - FRAC) + l10 * FRAC;
50 if (L_cont <= 0.0f) { out_wvc[i] = WVC_DEF; continue; }
51
52 float D = 1.0f - l9 / L_cont;
53 if (D < 0.0f) D = 0.0f;
54
55 float wvc = D / km;
56 out_wvc[i] = (wvc < WVC_MIN) ? WVC_MIN : (wvc > WVC_MAX) ? WVC_MAX : wvc;
57 }
58}

60/* ─── AOD DDV retrieval ────────────────────────────────────────────────────── */
61
62float retrieve_aod_ddv(const float *L_470, const float *L_660,
63 const float *L_860, const float *L_2130,
64 int npix, int doy, float sza_deg,
65 float *out_aod)
66{
67 static const float G = 0.65f; /* HG asymmetry parameter */
68 static const float OMEGA0 = 0.89f; /* single-scattering albedo */
69 static const float AOD_FALLBACK = 0.15f;
70 static const float AOD_MAX = 3.0f;
71
72 float E0_470 = sixs_E0(0.470f);
73 float E0_660 = sixs_E0(0.660f);
74 float E0_2130 = sixs_E0(2.130f);
75 float d2f = (float)sixs_earth_sun_dist2(doy);
76
77 float cos_sza = cosf(sza_deg * (float)(M_PI / 180.0));
78 float mu_s = (cos_sza > 0.05f) ? cos_sza : 0.05f;
79
80 /* Henyey-Greenstein phase function for nadir view (cos Θ = −μs):
81 * P_HG = (1 − g²) / (1 + g² − 2g cos Θ)^1.5 with cos Θ = −μs */
82 float cos_theta = -mu_s;
83 float denom_hg = 1.0f + G*G - 2.0f*G*cos_theta;
84 float P_HG = (1.0f - G*G) / (denom_hg * sqrtf(denom_hg));
85 if (P_HG < 1e-4f) P_HG = 1e-4f;
86
87 /* ρ_toa = π × L × d² / (E0 × μs) → scale = π × d² / μs */
88 float pi_d2_over_mus = (float)M_PI * d2f / mu_s;
89
90 /* τ = ρ_path × 4 μs / (ω₀ P_HG) with μv = 1 (nadir) */
91 float tau_factor = 4.0f * mu_s / (OMEGA0 * P_HG);
92
93 double sum_aod = 0.0;
94 int n_ddv = 0;
95
96 for (int i = 0; i < npix; i++) {
97 float l4 = L_470[i], l6 = L_660[i], l8 = L_860[i], l21 = L_2130[i];
98 if (!(l4 > 0.0f) || !(l6 > 0.0f) || !(l8 > 0.0f) || !(l21 > 0.0f)) {
99 out_aod[i] = -1.0f; /* invalid → fill with mean later */
100 continue;
101 }
102
103 /* TOA reflectance at 2130 nm for DDV mask */
104 float rho_2130 = pi_d2_over_mus * l21 / E0_2130;
105 if (rho_2130 < 0.01f || rho_2130 > 0.25f) {
106 out_aod[i] = -1.0f;
107 continue;
108 }
109
110 /* NDVI from 860/660 radiance (E0 factor cancels in ratio) */
111 float ndvi = (l8 - l6) / (l8 + l6 + 1e-10f);
112 if (ndvi <= 0.1f) { out_aod[i] = -1.0f; continue; }
113
114 /* TOA reflectance at 470/660 nm */
115 float rho_470 = pi_d2_over_mus * l4 / E0_470;
116 float rho_660 = pi_d2_over_mus * l6 / E0_660;
117
118 /* DDV surface reflectance prediction */
119 float rho_surf_470 = 0.25f * rho_2130;
120 float rho_surf_660 = 0.50f * rho_2130;
121
122 /* Path reflectance (ensure positive) */
123 float rho_path_470 = rho_470 - rho_surf_470;
124 float rho_path_660 = rho_660 - rho_surf_660;
125 if (rho_path_470 < 1e-5f) rho_path_470 = 1e-5f;
126 if (rho_path_660 < 1e-5f) rho_path_660 = 1e-5f;
127
128 /* Single-scattering aerosol optical depth at 470/660 nm */
129 float tau_470 = rho_path_470 * tau_factor;
130 float tau_660 = rho_path_660 * tau_factor;
131
132 /* Ångström exponent and scale to 550 nm */
133 float alpha = 0.0f;
134 if (tau_470 > 1e-4f && tau_660 > 1e-4f) {
135 alpha = -logf(tau_470 / tau_660) / logf(0.470f / 0.660f);
136 if (alpha < -1.0f) alpha = -1.0f;
137 if (alpha > 3.0f) alpha = 3.0f;
138 }
139 float tau_550 = tau_470 * powf(0.550f / 0.470f, -alpha);
140 if (tau_550 < 0.0f) tau_550 = 0.0f;
141 if (tau_550 > AOD_MAX) tau_550 = AOD_MAX;
142
143 out_aod[i] = tau_550;
144 sum_aod += tau_550;
145 n_ddv++;
146 }
147
148 float aod_mean = (n_ddv > 0) ? (float)(sum_aod / n_ddv) : AOD_FALLBACK;
149
150 /* Fill invalid pixels (marked -1) with scene mean */
151 for (int i = 0; i < npix; i++)
152 if (out_aod[i] < 0.0f) out_aod[i] = aod_mean;
153
154 return aod_mean;
155}

157/* ─── O3 Chappuis band retrieval ───────────────────────────────────────────── */
158
159float retrieve_o3_chappuis(const float *L_540, const float *L_600,
160 const float *L_680, int npix,
161 float sza_deg, float vza_deg)
162{
163 /* Fraction of 540→680 range at 600 nm: (600-540)/(680-540) = 3/7 */
164 static const float FRAC = (0.600f - 0.540f) / (0.680f - 0.540f);
165 /* Chappuis cross section at 600 nm; effective DU⁻¹ absorption coefficient.
166 * Derived from Bogumil et al. (2003): σ ≈ 3.8e-21 cm²/mol;
167 * 1 DU = 2.687e16 mol/cm² → σ_DU = 3.8e-21 × 2.687e16 ≈ 1.02e-4 DU⁻¹ */
168 static const float SIGMA_O3 = 1.0e-4f;
169 static const float O3_FALLBACK = 300.0f;
170 static const float O3_MIN = 50.0f;
171 static const float O3_MAX = 800.0f;
172
173 float cos_sza = cosf(sza_deg * (float)(M_PI / 180.0));
174 float cos_vza = cosf(vza_deg * (float)(M_PI / 180.0));
175 float mu_s = (cos_sza > 0.05f) ? cos_sza : 0.05f;
176 float mu_v = (cos_vza > 0.05f) ? cos_vza : 0.05f;
177 float sigma_m = SIGMA_O3 * (1.0f / mu_s + 1.0f / mu_v);
178
179 double sum_o3 = 0.0;
180 int n_valid = 0;
181
182 for (int i = 0; i < npix; i++) {
183 float l5 = L_540[i], l6 = L_600[i], l7 = L_680[i];
184 if (!(l5 > 0.0f) || !(l6 > 0.0f) || !(l7 > 0.0f)) continue;
185
186 float L_cont = l5 * (1.0f - FRAC) + l7 * FRAC;
187 if (L_cont <= 0.0f || l6 >= L_cont) continue;
188
189 float D = 1.0f - l6 / L_cont;
190 float o3 = D / sigma_m;
191
192 if (o3 >= O3_MIN && o3 <= O3_MAX) {
193 sum_o3 += o3;
194 n_valid++;
195 }
196 }
197
198 if (n_valid == 0) return O3_FALLBACK;
199
200 float result = (float)(sum_o3 / n_valid);
201 return (result < O3_MIN) ? O3_MIN : (result > O3_MAX) ? O3_MAX : result;
202}

204/* ─── ISA surface pressure ─────────────────────────────────────────────────── */
205
206float retrieve_pressure_isa(float elev_m)
207{
208 /* ICAO International Standard Atmosphere (1993)
209 * P = 1013.25 × (1 − 2.2558×10⁻⁵ × h)^5.2559 [hPa] */
210 if (elev_m < 0.0f) elev_m = 0.0f;
211 if (elev_m > 11000.0f) elev_m = 11000.0f;
212 return 1013.25f * powf(1.0f - 2.2558e-5f * elev_m, 5.2559f);
213}

215/* ─── O₂-A surface pressure retrieval ─────────────────────────────────────── */
216
217void retrieve_pressure_o2a(const float *L_740, const float *L_760,
218 const float *L_780, int npix,
219 float sza_deg, float vza_deg,
220 float *out_pressure)
221{
222 /* Continuum interpolation between 740 nm and 780 nm; feature at 760 nm.
223 *
224 * Physical model (Beer-Lambert, linearised):
225 * τ_O2(P) = K_O2 × (P / P0)
226 * D_760 = max(0, 1 − L_760 / L_cont) ≈ τ_O2 × m (first-order)
227 * P = P0 × D_760 / (K_O2 × m)
228 *
229 * K_O2 = 0.25: calibrated to ~10 nm FWHM sensors; derived from
230 * Schläpfer et al. (1998): at P=P0, SZA=0 (m=2), band-depth D ≈ 0.50.
231 * For a 5 nm sensor set K_O2 ≈ 0.40; for 20 nm use K_O2 ≈ 0.15.
232 *
233 * Reference:
234 * Schläpfer, D., Borel, C.C., Keller, J., Itten, K.I. (1998).
235 * Atmospheric pre-processing of DAIS airborne data. SPIE 3502.
236 */
237 static const float FRAC = 0.5f; /* (760−740)/(780−740) = 0.5 */
238 static const float K_O2 = 0.25f; /* effective OD per unit air mass at P0 */
239 static const float P0 = 1013.25f; /* sea-level pressure [hPa] */
240 static const float P_MIN = 200.0f; /* ~11 km altitude */
241 static const float P_MAX = 1100.0f;
242 static const float P_DEF = P0;
243
244 float cos_sza = cosf(sza_deg * (float)(M_PI / 180.0));
245 float cos_vza = cosf(vza_deg * (float)(M_PI / 180.0));
246 float mu_s = (cos_sza > 0.05f) ? cos_sza : 0.05f;
247 float mu_v = (cos_vza > 0.05f) ? cos_vza : 0.05f;
248 float m = 1.0f / mu_s + 1.0f / mu_v; /* two-way air mass */
249
250 for (int i = 0; i < npix; i++) {
251 float l74 = L_740[i], l76 = L_760[i], l78 = L_780[i];
252 if (!(l74 > 0.0f) || !(l76 > 0.0f) || !(l78 > 0.0f)) {
253 out_pressure[i] = P_DEF;
254 continue;
255 }
256 float L_cont = l74 * (1.0f - FRAC) + l78 * FRAC;
257 if (L_cont <= 0.0f || l76 >= L_cont) {
258 out_pressure[i] = P_DEF;
259 continue;
260 }
261 float D = 1.0f - l76 / L_cont;
262 if (D < 0.0f) D = 0.0f;
263 if (D > 0.95f) D = 0.95f; /* cap: near-zero pressure is unphysical */
264
265 float P = P0 * D / (K_O2 * m);
266 out_pressure[i] = (P < P_MIN) ? P_MIN : (P > P_MAX) ? P_MAX : P;
267 }
268}

270/* ─── Cloud / shadow / water / snow quality bitmask ───────────────────────── */

272#include <stdint.h>
273
274void retrieve_quality_mask(const float *L_blue, const float *L_red,
275 const float *L_nir, const float *L_swir,
276 int npix, int doy, float sza_deg,
277 uint8_t *out_mask)
278{
279 /* Thresholds are applied to TOA reflectance (from TOA radiance).
280 *
281 * Bitmask: MASK_CLOUD=0x01 MASK_SHADOW=0x02 MASK_WATER=0x04 MASK_SNOW=0x08
282 *
283 * Cloud: TOA_blue > 0.25 AND NDVI < 0.2, OR TOA_nir > 0.60
284 * Shadow: TOA_blue < 0.04 AND TOA_red < 0.04 AND TOA_nir < 0.04
285 * Water: TOA_nir < 0.05 AND NDVI < 0.0
286 * Snow/ice: NDSI > 0.40 AND TOA_nir > 0.10 (requires L_swir)
287 *
288 * References:
289 * Cloud / shadow: Braaten et al. (2015), Remote Sens. 7:15745
290 * Water: McFeeters (1996), Int. J. Remote Sens. 17:1425
291 * Snow: Hall et al. (1995), Remote Sens. Environ. 54:127
292 */
293 float E0_blue = sixs_E0(0.470f);
294 float E0_red = sixs_E0(0.660f);
295 float E0_nir = sixs_E0(0.860f);
296 float E0_swir = sixs_E0(1.600f);
297 float d2f = (float)sixs_earth_sun_dist2(doy);
298
299 float cos_sza = cosf(sza_deg * (float)(M_PI / 180.0));
300 float mu_s = (cos_sza > 0.05f) ? cos_sza : 0.05f;
301 float scale = (float)M_PI * d2f / mu_s;
302
303 for (int i = 0; i < npix; i++) {
304 out_mask[i] = 0;
305 float lb = L_blue[i], lr = L_red[i], ln = L_nir[i];
306 if (!(lb > 0.0f) || !(lr > 0.0f) || !(ln > 0.0f)) continue;
307
308 float rb = scale * lb / E0_blue;
309 float rr = scale * lr / E0_red;
310 float rn = scale * ln / E0_nir;
311
312 float ndvi = (rn - rr) / (rn + rr + 1e-10f);
313
314 /* NDSI requires SWIR */
315 float ndsi = -2.0f;
316 if (L_swir && L_swir[i] > 0.0f) {
317 float rs = scale * L_swir[i] / E0_swir;
318 ndsi = (rb - rs) / (rb + rs + 1e-10f);
319 }
320
321 if ((rb > 0.25f && ndvi < 0.2f) || rn > 0.60f)
322 out_mask[i] |= RETRIEVE_MASK_CLOUD;
323
324 if (rb < 0.04f && rr < 0.04f && rn < 0.04f)
325 out_mask[i] |= RETRIEVE_MASK_SHADOW;
326
327 if (rn < 0.05f && ndvi < 0.0f)
328 out_mask[i] |= RETRIEVE_MASK_WATER;
329
330 if (ndsi > 0.40f && rn > 0.10f)
331 out_mask[i] |= RETRIEVE_MASK_SNOW;
332 }
333}

343void retrieve_aod_maiac(float *aod_data, int nrows, int ncols, int patch_sz)
344{
345 /* Segment image into non-overlapping patch_sz × patch_sz blocks.
346 * Each patch with ≥1 valid pixel uses the patch-median AOD (robust to
347 * DDV outliers). Invalid patches are filled by inverse-distance
348 * weighting from the nearest valid patch centroids.
349 *
350 * Reference:
351 * Lyapustin, A. et al. (2011). Multiangle implementation of atmospheric
352 * correction (MAIAC) — Part 1. JGR 116, D05203. This implementation
353 * uses only the spatial regularization concept (patch-median + IDW fill),
354 * not the full MAIAC surface model or multi-angle retrieval.
355 */
356 if (patch_sz < 2) patch_sz = 2;
357
358 int np_rows = (nrows + patch_sz - 1) / patch_sz;
359 int np_cols = (ncols + patch_sz - 1) / patch_sz;
360 int np_total = np_rows * np_cols;
361 int buf_max = patch_sz * patch_sz;
362
363 float *patch_aod = malloc((size_t)np_total * sizeof(float));
364 float *patch_row = malloc((size_t)np_total * sizeof(float));
365 float *patch_col = malloc((size_t)np_total * sizeof(float));
366 int *patch_valid = malloc((size_t)np_total * sizeof(int));
367 float *patch_buf = malloc((size_t)buf_max * sizeof(float));
368 if (!patch_aod || !patch_row || !patch_col || !patch_valid || !patch_buf) {
369 free(patch_aod); free(patch_row); free(patch_col);
370 free(patch_valid); free(patch_buf);
371 return; /* silently skip on OOM */
372 }
373
374 /* ── Step 1: per-patch median ── */
375 for (int pr = 0; pr < np_rows; pr++) {
376 for (int pc = 0; pc < np_cols; pc++) {
377 int pi = pr * np_cols + pc;
378 int r0 = pr * patch_sz, r1 = r0 + patch_sz; if (r1 > nrows) r1 = nrows;
379 int c0 = pc * patch_sz, c1 = c0 + patch_sz; if (c1 > ncols) c1 = ncols;
380
381 patch_row[pi] = 0.5f * (float)(r0 + r1);
382 patch_col[pi] = 0.5f * (float)(c0 + c1);
383
384 int n = 0;
385 for (int r = r0; r < r1; r++)
386 for (int c = c0; c < c1; c++) {
387 float v = aod_data[r * ncols + c];
388 if (isfinite(v) && v >= 0.0f)
389 patch_buf[n++] = v;
390 }
391
392 if (n == 0) {
393 patch_valid[pi] = 0;
394 patch_aod[pi] = -1.0f;
395 } else {
396 qsort(patch_buf, (size_t)n, sizeof(float), _float_cmp);
397 patch_aod[pi] = patch_buf[n / 2];
398 patch_valid[pi] = 1;
399 }
400 }
401 }
402
403 /* ── Step 2: IDW fill for invalid patches ── */
404 for (int pi = 0; pi < np_total; pi++) {
405 if (patch_valid[pi]) continue;
406 double sum_w = 0.0, sum_wa = 0.0;
407 for (int pj = 0; pj < np_total; pj++) {
408 if (!patch_valid[pj]) continue;
409 float dr = patch_row[pi] - patch_row[pj];
410 float dc = patch_col[pi] - patch_col[pj];
411 double d2 = (double)(dr * dr + dc * dc);
412 if (d2 < 1.0) d2 = 1.0;
413 double w = 1.0 / d2;
414 sum_w += w;
415 sum_wa += w * (double)patch_aod[pj];
416 }
417 patch_aod[pi] = (sum_w > 0.0) ? (float)(sum_wa / sum_w) : 0.1f;
418 }
419
420 /* ── Step 3: write patch medians back to all pixels in each patch ── */
421 for (int pr = 0; pr < np_rows; pr++) {
422 for (int pc = 0; pc < np_cols; pc++) {
423 int pi = pr * np_cols + pc;
424 float aod_p = patch_aod[pi];
425 int r0 = pr * patch_sz, r1 = r0 + patch_sz; if (r1 > nrows) r1 = nrows;
426 int c0 = pc * patch_sz, c1 = c0 + patch_sz; if (c1 > ncols) c1 = ncols;
427 for (int r = r0; r < r1; r++)
428 for (int c = c0; c < c1; c++)
429 aod_data[r * ncols + c] = aod_p;
430 }
431 }
432
433 free(patch_aod); free(patch_row); free(patch_col);
434 free(patch_valid); free(patch_buf);
435}
