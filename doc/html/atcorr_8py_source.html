1"""Python ctypes bindings for ``libatcorr.so`` (6SV2.1-based atmospheric correction).
2
3This module loads ``libatcorr.so`` from the GRASS add-on installation directory
4(``$GISBASE/lib/``) or from the build tree. It exposes the full public C API
5as Python functions backed by NumPy arrays.
6
7Public API
8----------
9- :class:`LutConfig` — LUT grid specification and scene geometry.
10- :class:`LutArrays` — LUT output arrays shaped ``[n_aod, n_h2o, n_wl]``.
11- :func:`compute_lut` — Build the 3-D atmospheric correction LUT.
12- :func:`lut_slice` — Bilinear interpolation at a fixed (AOD, H₂O) point.
13- :func:`invert` — Lambertian BOA reflectance inversion.
14- :func:`solar_E0` — Thuillier solar irradiance spectrum.
15- :func:`earth_sun_dist2` — Seasonal Earth–Sun distance.
16- :func:`apply_srf_correction` — SRF gas-transmittance correction via libRadtran.
17- :func:`retrieve_pressure_isa` — ISA surface pressure from elevation.
18- :func:`retrieve_h2o_940` — H₂O column from 940 nm band depth.
19- :func:`retrieve_o3_chappuis` — O₃ column from Chappuis band depth.
20- :func:`retrieve_aod_ddv` — AOD from MODIS dark-target DDV.
21- :func:`retrieve_pressure_o2a` — Surface pressure from O₂-A band depth.
22- :func:`retrieve_quality_mask` — Cloud/shadow/water/snow bitmask.
23- :func:`retrieve_aod_maiac` — Patch-median AOD spatial regularisation.
24- :func:`oe_invert_aod_h2o` — Joint AOD + H₂O optimal-estimation retrieval.
25
26Quick-start example::
27
28 import numpy as np
29 from atcorr import LutConfig, compute_lut, lut_slice, invert
30
31 cfg = LutConfig(
32 wl=np.array([0.45, 0.55, 0.65, 0.87], dtype=np.float32),
33 aod=np.array([0.0, 0.2, 0.6], dtype=np.float32),
34 h2o=np.array([1.0, 3.0], dtype=np.float32),
35 sza=35.0, vza=5.0, raa=90.0,
36 altitude_km=1000.0, # satellite orbit
37 atmo_model=1, # US Standard 1962
38 aerosol_model=1, # continental
39 )
40 lut = compute_lut(cfg) # LutArrays [3, 2, 4]
41 sl = lut_slice(cfg, lut, aod_val=0.15, h2o_val=2.0) # 1-D [4]
42 rho = invert(rho_toa, sl.R_atm, sl.T_down, sl.T_up, sl.s_alb)
43"""

151def solar_E0(wl_um):
152 """Solar irradiance E0 in W/(m² µm) from 6SV2.1 Thuillier spectrum.
153
154 Parameters
155 ----------
156 wl_um : float or array-like
157 Wavelength in µm.
158 """
159 wl_arr = np.asarray(wl_um, dtype=np.float32)
160 scalar = wl_arr.ndim == 0
161 wl_arr = np.atleast_1d(wl_arr)
162 out = np.array([_lib.sixs_E0(float(w)) for w in wl_arr], dtype=np.float32)
163 return float(out[0]) if scalar else out

235def compute_lut(cfg: LutConfig) -> LutArrays:
236 """Compute a full 3-D atmospheric correction LUT via DISCOM.
237
238 Allocates output arrays and calls :c:func:`atcorr_compute_lut` (OpenMP
239 parallelised over the AOD grid).
240
241 Parameters
242 ----------
243 cfg : LutConfig
244 LUT grid specification and scene geometry.
245
246 Returns
247 -------
248 LutArrays
249 ``R_atm``, ``T_down``, ``T_up``, ``s_alb`` shaped ``[n_aod, n_h2o, n_wl]``.
250 ``R_atmQ`` and ``R_atmU`` are populated when ``cfg.enable_polar=1``,
251 ``None`` otherwise.
252
253 Raises
254 ------
255 RuntimeError
256 If the C library returns a non-zero error code.
257 """
258 n_aod = len(cfg.aod)
259 n_h2o = len(cfg.h2o)
260 n_wl = len(cfg.wl)
261 n = n_aod * n_h2o * n_wl
262
263 # Allocate output arrays
264 R_atm = np.zeros(n, dtype=np.float32)
265 T_down = np.ones(n, dtype=np.float32)
266 T_up = np.ones(n, dtype=np.float32)
267 s_alb = np.zeros(n, dtype=np.float32)
268
269 # Polarization Q/U arrays (only when enable_polar=1)
270 enable_polar = getattr(cfg, 'enable_polar', 0)
271 R_atmQ = np.zeros(n, dtype=np.float32) if enable_polar else None
272 R_atmU = np.zeros(n, dtype=np.float32) if enable_polar else None
273
274 # Build C structs
275 c_cfg = _LutConfigC(
276 wl = cfg.wl.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
277 n_wl = n_wl,
278 aod = cfg.aod.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
279 n_aod = n_aod,
280 h2o = cfg.h2o.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
281 n_h2o = n_h2o,
282 sza = cfg.sza,
283 vza = cfg.vza,
284 raa = cfg.raa,
285 altitude_km = cfg.altitude_km,
286 atmo_model = cfg.atmo_model,
287 aerosol_model = cfg.aerosol_model,
288 surface_pressure = cfg.surface_pressure,
289 ozone_du = cfg.ozone_du,
290 enable_polar = enable_polar,
291 )
292 c_out = _LutArraysC(
293 R_atm = R_atm.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
294 T_down = T_down.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
295 T_up = T_up.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
296 s_alb = s_alb.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
297 T_down_dir = None, # not requested in Python API
298 R_atmQ = R_atmQ.ctypes.data_as(ctypes.POINTER(ctypes.c_float)) if R_atmQ is not None else None,
299 R_atmU = R_atmU.ctypes.data_as(ctypes.POINTER(ctypes.c_float)) if R_atmU is not None else None,
300 )
301
302 ret = _lib.atcorr_compute_lut(ctypes.byref(c_cfg), ctypes.byref(c_out))
303 if ret != 0:
304 raise RuntimeError(f"atcorr_compute_lut failed with code {ret}")
305
306 shape = (n_aod, n_h2o, n_wl)
307 return LutArrays(
308 R_atm = R_atm.reshape(shape),
309 T_down = T_down.reshape(shape),
310 T_up = T_up.reshape(shape),
311 s_alb = s_alb.reshape(shape),
312 R_atmQ = R_atmQ.reshape(shape) if R_atmQ is not None else None,
313 R_atmU = R_atmU.reshape(shape) if R_atmU is not None else None,
314 )

317def lut_slice(cfg: LutConfig, arrays: LutArrays,
318 aod_val: float, h2o_val: float) -> LutArrays:
319 """Bilinear interpolation of the LUT at a fixed (aod_val, h2o_val).
320
321 Returns a LutArrays object whose four arrays are 1-D [n_wl] spectral
322 curves at the requested atmospheric state. Useful for pre-computing
323 scene-average correction parameters before pixel-level inversion.
324 """
325 n_wl = len(cfg.wl)
326 Rs = np.zeros(n_wl, dtype=np.float32)
327 Tds = np.zeros(n_wl, dtype=np.float32)
328 Tus = np.zeros(n_wl, dtype=np.float32)
329 ss = np.zeros(n_wl, dtype=np.float32)
330
331 n_aod = len(cfg.aod)
332 n_h2o = len(cfg.h2o)
333 n = n_aod * n_h2o * n_wl
334
335 # Flatten LutArrays to 1-D for the C call
336 Ra = arrays.R_atm.ravel().astype(np.float32)
337 Td = arrays.T_down.ravel().astype(np.float32)
338 Tu = arrays.T_up.ravel().astype(np.float32)
339 sa = arrays.s_alb.ravel().astype(np.float32)
340 assert len(Ra) == n
341
342 c_cfg = _LutConfigC(
343 wl=cfg.wl.ctypes.data_as(ctypes.POINTER(ctypes.c_float)), n_wl=n_wl,
344 aod=cfg.aod.ctypes.data_as(ctypes.POINTER(ctypes.c_float)), n_aod=n_aod,
345 h2o=cfg.h2o.ctypes.data_as(ctypes.POINTER(ctypes.c_float)), n_h2o=n_h2o,
346 sza=cfg.sza, vza=cfg.vza, raa=cfg.raa,
347 altitude_km=cfg.altitude_km,
348 atmo_model=cfg.atmo_model, aerosol_model=cfg.aerosol_model,
349 surface_pressure=cfg.surface_pressure, ozone_du=cfg.ozone_du,
350 )
351 c_arr = _LutArraysC(
352 R_atm =Ra.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
353 T_down =Td.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
354 T_up =Tu.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
355 s_alb =sa.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
356 T_down_dir=None,
357 )
358 _lib.atcorr_lut_slice(
359 ctypes.byref(c_cfg), ctypes.byref(c_arr),
360 ctypes.c_float(aod_val), ctypes.c_float(h2o_val),
361 Rs.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
362 Tds.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
363 Tus.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
364 ss.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
365 None, # Tdds: not requested
366 )
367 return LutArrays(Rs, Tds, Tus, ss)

370def invert(rho_toa, R_atm, T_down, T_up, s_alb):
371 """Lambertian BOA reflectance inversion (Python/NumPy implementation).
372
373 Equivalent to :c:func:`atcorr_invert` but operates on NumPy arrays.
374 All inputs must be broadcastable to a common shape.
375
376 Parameters
377 ----------
378 rho_toa : array-like
379 TOA reflectance.
380 R_atm : array-like
381 Atmospheric path reflectance (LUT value).
382 T_down : array-like
383 Downward transmittance.
384 T_up : array-like
385 Upward transmittance.
386 s_alb : array-like
387 Spherical albedo.
388
389 Returns
390 -------
391 numpy.ndarray
392 Surface (BOA) reflectance.
393 """
394 y = (rho_toa - R_atm) / (T_down * T_up + 1e-10)
395 return y / (1.0 + s_alb * y + 1e-10)

424def apply_srf_correction(cfg: LutConfig, arrays: LutArrays,
425 fwhm_um, threshold_nm: float = 5.0) -> LutArrays:
426 """Apply per-band Gaussian SRF gas-transmittance correction.
427
428 Replaces the coarse 6SV gas parameterisation (10 cm⁻¹) with libRadtran
429 reptran fine (~0.05 nm) convolved with a Gaussian SRF for bands where
430 FWHM < threshold_nm. T_down and T_up are modified in the arrays.
431
432 Parameters
433 ----------
434 cfg : LutConfig used to compute `arrays`.
435 arrays : LutArrays to correct in place (T_down, T_up modified).
436 fwhm_um : per-band FWHM in µm, 1-D array aligned with cfg.wl.
437 Pass None or zeros to correct all bands (delta-SRF).
438 threshold_nm : only correct bands with FWHM < threshold_nm (default 5).
439
440 Returns
441 -------
442 arrays : same LutArrays with corrected T_down, T_up (modified in place).
443
444 Notes
445 -----
446 Requires libRadtran (uvspec) in PATH or $LIBRADTRAN_DIR/bin.
447 libRadtran data directory must be accessible via $LIBRADTRAN_DATA or the
448 standard installation path /usr/local/share/libRadtran/data.
449 """
450 n_wl = len(cfg.wl)
451 n_aod = len(cfg.aod)
452 n_h2o = len(cfg.h2o)
453 n = n_aod * n_h2o * n_wl
454
455 fwhm_arr = (np.asarray(fwhm_um, dtype=np.float32)
456 if fwhm_um is not None
457 else np.zeros(n_wl, dtype=np.float32))
458 if len(fwhm_arr) != n_wl:
459 raise ValueError(f"fwhm_um length {len(fwhm_arr)} != n_wl {n_wl}")
460
461 c_srf_cfg = _SrfConfigC(
462 fwhm_um = fwhm_arr.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
463 threshold_um = float(threshold_nm) * 1e-3, # nm → µm
464 )
465 c_cfg = _LutConfigC(
466 wl=cfg.wl.ctypes.data_as(ctypes.POINTER(ctypes.c_float)), n_wl=n_wl,
467 aod=cfg.aod.ctypes.data_as(ctypes.POINTER(ctypes.c_float)), n_aod=n_aod,
468 h2o=cfg.h2o.ctypes.data_as(ctypes.POINTER(ctypes.c_float)), n_h2o=n_h2o,
469 sza=cfg.sza, vza=cfg.vza, raa=cfg.raa,
470 altitude_km=cfg.altitude_km,
471 atmo_model=cfg.atmo_model, aerosol_model=cfg.aerosol_model,
472 surface_pressure=cfg.surface_pressure, ozone_du=cfg.ozone_du,
473 )
474
475 # Flatten LutArrays to 1-D C arrays
476 Td = np.ascontiguousarray(arrays.T_down.ravel(), dtype=np.float32)
477 Tu = np.ascontiguousarray(arrays.T_up.ravel(), dtype=np.float32)
478 Ra = np.ascontiguousarray(arrays.R_atm.ravel(), dtype=np.float32)
479 sa = np.ascontiguousarray(arrays.s_alb.ravel(), dtype=np.float32)
480 assert len(Td) == n
481
482 c_arr = _LutArraysC(
483 R_atm =Ra.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
484 T_down =Td.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
485 T_up =Tu.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
486 s_alb =sa.ctypes.data_as(ctypes.POINTER(ctypes.c_float)),
487 T_down_dir =None,
488 )
489
490 srf = _lib.atcorr_srf_compute(ctypes.byref(c_srf_cfg), ctypes.byref(c_cfg))
491 if not srf:
492 raise RuntimeError(
493 "atcorr_srf_compute returned NULL — "
494 "is uvspec installed? Check LIBRADTRAN_DIR / LIBRADTRAN_DATA."
495 )
496
497 _lib.atcorr_srf_apply(srf, ctypes.byref(c_cfg), ctypes.byref(c_arr))
498 _lib.atcorr_srf_free(srf)
499
500 shape = (n_aod, n_h2o, n_wl)
501 return LutArrays(
502 R_atm = Ra.reshape(shape),
503 T_down = Td.reshape(shape),
504 T_up = Tu.reshape(shape),
505 s_alb = sa.reshape(shape),
506 )
